public static void swapChild( BTNode root, int lvl, int M ){
       if(root==null){
        return;
       }
       if(lvl==M || lvl>M){
        BTNode curr=root.left;
            root.left = root.right;


            root.right = curr;
       }
       swapChild( root.left, lvl+1,  M );
       swapChild( root.right,  lvl+1,  M );
    }

        if(box==null){
            return 0;
        }

        }
        else{
            return (int)box.elem+clvlsum( box.left, level+1)+clvlsum( box.right, level+1);
        }
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        int total=0;
        if(root==null){
            return 0;
        }


        if(root.elem>=low&&root.elem<=high){
            total+=root.elem;
        }
        if(root.elem>low){
            total+=rangeSum(root.left,low,high);
        }
        if(root.elem<high){
            total+=rangeSum(root.right,low,high);
        }
        return total;
    }

public static Integer mirrorSum( BSTNode root ){
        if(root==null){
            return 0;
        }
        return totalnum(root.left,root.right);
    }
public static Integer totalnum(BSTNode baam,BSTNode daan){
    if(baam==null || daan==null){
        return 0;
    }
    int total=baam.elem+daan.elem;
    total+=totalnum(baam.left,daan.right);
    total+=totalnum(baam.right,daan.left);
    return total;
}

public static Boolean isBST( BSTNode root ){
        if(root==null){
            return true;
        }
        if((root.left != null && root.left.elem >= root.elem)||(root.right != null && root.right.elem <= root.elem)){


            return false;
        }
        return isBST(root.left)&&isBST(root.right);
