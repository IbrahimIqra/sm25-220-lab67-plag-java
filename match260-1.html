public class Task4 {

    public static void swapChild( BTNode root, int lvl, int M ){
        if(root==null){
            return;
        }

        if(lvl>=M){
        BTNode temp=root.left;
        root.left=root.right;
        root.right=temp;            
        }

        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }
}

public class Task5 {
    public static Integer subtractSummation( BTNode root ){
        if(root==null){
            return 0;
        }

        int l_sum=sumOfSubTree(root.left);
        int r_sum=sumOfSubTree(root.right);

        return l_sum-r_sum; 
    }

    private static int sumOfSubTree(BTNode root){ //helper
        if(root==null){
            return 0;
        }


        return (int)root.elem+sumOfSubTree(root.left)+sumOfSubTree(root.right);
    }

}


//TASK06
public class Task06 {

    public static Integer levelSum( BTNode root ){

        return lvlSum(root,1); 
    }
    private static Integer lvlSum(BTNode root,int lvl){
        if(root==null){
            return 0;
        }
        int val=(Integer)root.elem;
        int l=(lvl%2==1)? -val : val;//odd -val and even +val


        return l+lvlSum(root.left, lvl+1)+lvlSum(root.right, lvl+1);
    }

}

public class Task4 {

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null){
            return 0;
        }     
        int val=(int)root.elem;
        if(val<low){
            return rangeSum(root.right, low, high);
        }
        if(val>high){
            return rangeSum(root.left, low, high);
        }
        return val+rangeSum(root.left, low, high)+rangeSum(root.right, low, high);
    }

}

public class Task5 {
    
    public static Integer mirrorSum( BSTNode root ){
        if(root==null){
            return 0;
        }
        
        return mirrorSumHelper(root.left,root.right);
    }
    
    private static Integer mirrorSumHelper(BSTNode l,BSTNode r){
        if(l==null||r==null){
            return 0;
        }
        int l_val=(int)l.elem;
        int r_val=(int)r.elem;


        int sum=l_val+r_val;
        
        sum+=mirrorSumHelper(l.left,r.right);
        sum+=mirrorSumHelper(l.right,r.left);
        
        return sum;
    }

}

public class Task6 {

    public static Boolean isBST( BSTNode root ){
        
        return isBSTHelper(root,0,0);
    }
    private static boolean isBSTHelper(BSTNode root, int min, int max){
        if(root==null){
            return true;
        }
        int val=(int)root.elem;
        if((min!=0&&val<=min)||(max!=0&&val>=max)){
            return false;
        }
        return isBSTHelper(root.left,min,val) && isBSTHelper(root.right,val,max);
    }
}
