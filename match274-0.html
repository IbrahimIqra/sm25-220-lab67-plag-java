class Task4{
    public static void swapChild(BTNode root, int level, int M){
        swapping(root, level, M, level <M);
    }
    private static void swapping(BTNode node, int level, int M, boolean flag){
      if(node ==null){
        return;
      }
        if(flag){
            BTNode val =node.left;
            node.left =node.right;
            node.right =val;
        }
        boolean cldflag=(level +1)<M;
        swapping(node.left, level +1, M, cldflag);
        swapping(node.right,level +1, M, cldflag);
    }
}
class Task5{


    public static int subtractSummation(BTNode root){
      if(root == null){
        return 0;
      }
        int leftSum =sumLeft(root.left);
        int rightSum =sumRight(root.right);
        return leftSum-rightSum;
    }
    private static int sumLeft(BTNode node){
      if(node==null){
        return 0;
      }
        int tot=(Integer)node.elem; 
        return tot+sumLeft(node.left)+sumLeft(node.right);
    }
    private static int sumRight(BTNode node){
      if(node == null){
        return 0;
      }
        int tot=(Integer)node.elem; 
        return tot+sumRight(node.left)+sumRight(node.right);
    }
}

        return rangeSumHelper(root, low, high);
    }


    public static Integer rangeSumHelper(BSTNode root,Integer low, Integer high){
        if(root == null){
            return 0;
        }
        int sum = 0;
        if(root.elem >= low &&root.elem<= high){


            sum =root.elem;
        }
        if(root.elem>low){
            sum += rangeSumHelper(root.left, low, high);
        }
       if (root.elem<high){ 
            sum += rangeSumHelper(root.right, low, high);
        }
        return sum;
    }
}



public class Task5{
  public static Integer mirrorSum(BSTNode root){
        if(root == null){
            return 0;
        }
        return mirrorSumHelper(root.left,root.right);
    }
    public static Integer mirrorSumHelper(BSTNode left, BSTNode right){
        if(left ==null &&right==null){
            return 0;
        }


        if(left ==null||right == null){
            return 0;
        }
        int sum=left.elem +right.elem;
        sum +=mirrorSumHelper(left.left, right.right);  
        sum +=mirrorSumHelper(left.right, right.left);  
        return sum;
    }
}
