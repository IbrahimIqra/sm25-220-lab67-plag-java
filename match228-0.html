    public static Integer subtractSummation( BTNode root ){
      if(root==null){
        return null;
      }
      else{


        int sumOfLft=sumFunction(root.left);
        int sumOfRht=sumFunction(root.right);
        return sumOfLft-sumOfRht;
      }
    }
    
    private static int sumFunction(BTNode node){
      if(node==null){
        return 0;
        
      }
      else{
        int sumOfLft=sumFunction(node.left);
        int sumOfRht=sumFunction(node.right);
        return sumOfLft + sumOfRht + (Integer) node.elem;
      }
    }

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
      if(root==null){
        return 0;
      }
      int sum=0;
      
        
        if( low<=root.elem && high>=root.elem){
          sum+=root.elem;
        }
        if(root.elem<high){
          sum+=rangeSum(root.right,low,high);
        }
        if(root.elem>low){


          sum+=rangeSum(root.left,low,high);
        }
        
      
      return sum;
    }
