public class Task4 {
  public static void swapChild( BTNode root, int lvl, int M ){
    if (root==null) 
      return;
    if(lvl >= M){
      BTNode temp = root.left;
      root.left= root.right;
      root.right =temp;}
    swapChild(root.left,lvl +1,M);
    swapChild(root.right, lvl+1, M);
  } 
}

public class Task5 {
  public static Integer subtractSummation( BTNode root ){
    if (root == null) return 0;
    int lftSm = sumTree(root.left);
    int ritSm = sumTree(root.right);
    return lftSm- ritSm;
  }
  private static int sumTree(BTNode root) {
    if(root ==null) 
      return 0;
    return (Integer) root.elem +sumTree(root.left)+ sumTree(root.right);
  }
}

public class Task6 {
  public static Integer levelSum( BTNode root ){
    return levelSum(root, 1);}
  public static int levelSum(BTNode root,int lvl) {
    if (root == null) return 0;


    int vlu = (Integer) root.elem;
    if (lvl % 2== 1) { 
      return vlu +levelSum(root.left, lvl +1)+ levelSum(root.right, lvl + 1);
    } else { 
      return -vlu + levelSum(root.left, lvl + 1)+levelSum(root.right, lvl + 1);
    }
  }
}

public class Task4 {
  public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
    if(root== null) 
      return 0;
    int sm =0;
    if(root.elem>=low &&root.elem<= high){
      sm +=root.elem;
    }
    if(root.elem>low)
    {
      sm +=rangeSum(root.left,low,high);
    }
    if(root.elem< high){
      sm +=rangeSum(root.right,low,high);
    }
    return sm;
  }
  
}

public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
         if(root== null) return 0;
        return mirrorHelper(root.left,root.right);
    }
    private static int mirrorHelper(BSTNode left, BSTNode right){
        if(left ==null || right ==null) 
          return 0;
        int sum =left.elem + right.elem;
        sum +=mirrorHelper(left.left, right.right);
        sum +=mirrorHelper(left.right,right.left);
        return sum;
    }
}

public class Task6 {
  public static Boolean isBST( BSTNode root ){
    return isBSTUtil(root, Integer.MIN_VALUE,Integer.MAX_VALUE);
  }
  private static boolean isBSTUtil(BSTNode node, int min, int max){
    if(node == null)
      return true;


    if(node.elem <= min || node.elem >= max){
      return false;
    }
    return isBSTUtil(node.left,min,node.elem)&& isBSTUtil(node.right, node.elem,max);
  }
}
