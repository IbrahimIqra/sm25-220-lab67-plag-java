        sum += root.elem;

        sum += rangeSum(root.left, low, high);
    }

        sum += rangeSum(root.right, low, high);

    private static int mirrorSumHelper(BSTNode left, BSTNode right) {
        if (left == null || right == null){
             return 0;
        }
        int sum = left.elem + right.elem;
       
        sum += mirrorSumHelper(left.left, right.right);
        sum += mirrorSumHelper(left.right, right.left);
        return sum;
    }

public static Boolean isBST( BSTNode root ){
        return isBSTUtil(root, null, null);
    }

    // Helper method with boundary checks
    private static boolean isBSTUtil(BSTNode root, Integer min, Integer max) {
        if (root == null){
         return true;
        }
        if ((min != null && root.elem <= min) || (max != null && root.elem >= max)){


            return false;
        }
        return isBSTUtil(root.left, min, root.elem) &&
               isBSTUtil(root.right, root.elem, max);
    }
