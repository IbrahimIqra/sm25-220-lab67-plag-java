public class Task4 {
 public static void swapChild( BTNode root, int lvl, int M ){
        if(root==null) return;
        
        if(lvl < M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }
   
}

  public static Integer leftSubtree(BTNode root){
        if(root==null) return 0;

        return leftsum;

    }
     public static Integer rightSubtree(BTNode root){
      if(root==null)return 0;
       int rightsum = sum(root.right);
       return rightsum;
     }


      public static Integer sum(BTNode root){
           if(root==null) return 0;
           
        return (Integer) root.elem + sum(root.left) + sum(root.right);
      }

    public static Integer subtractSummation(BTNode root ){
        if(root==null) return 0;
        int left = leftSubtree(root);
        int right = rightSubtree(root) ;
        return left - right;
    }
    

    

}

   }

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        
    if(root==null) return 0;

      int sum = 0; 

      if ((int)root.value >= low && (int)root.value <= high){

    
     sum += (int)root.value;

    }

  if(root.value > low){

    sum += rangeSum(root.left, low, high);

   }

  if(root.value < high){
    sum += rangeSum(root.right, low, high);

  }

    return sum;
 }

  }

    public static int mirrorSumHelper(BSTNode left, BSTNode right){

          if (left == null || right == null) return 0;



      int sum = (int)left.value + (int)right.value;

    sum += mirrorSumHelper(left.left, right.right);
    sum += mirrorSumHelper(left.right, right.left);

    return sum;
    }
    
         public static Integer mirrorSum( BSTNode root ){

            if (root == null) return 0;
           return mirrorSumHelper(root.left, root.right);

         }
  

}

public class Task6 {

       public static Boolean isBST( BSTNode root ){

          return isBSTHelper(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
     }

 public static boolean isBSTHelper(BSTNode root, int min, int max) {

   if (root == null) return true;


   if ((int)root.value <= min || (int)root.value >= max) return false;

  return isBSTHelper(root.left, min, root.value) && isBSTHelper(root.right, root.value, max);
}
