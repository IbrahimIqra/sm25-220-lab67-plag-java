 public static void swapChild( BTNode root, int lvl, int M ){
        if (root == null) return;
        if (lvl >= M) return;

        BTNode tmp = root.left;
        root.left = root.right;


        root.right = tmp;

        swapChild(root.left,  lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }

public static int sum(BTNode root) {
        if (root == null){
            return 0;
        }
        return (Integer)root.elem + sum(root.left) + sum(root.right);

    }
    public static Integer subtractSummation(BTNode root) {


        int LeftSum = sum(root.left);
        int RightSum = sum(root.right);
        return LeftSum - RightSum;
    }

public static int sum(BTNode root, int lvl) {
        if (root == null){
            return 0;
        }
        int value = (Integer)root.elem;
        if (lvl%2==1){
            value=-value;
        }
        else{
            value=value;
        }


        return value + sum(root.left, lvl + 1) + sum(root.right, lvl + 1);
    }

    public static Integer levelSum( BTNode root ){
        return sum(root, 1);
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        //TO DO
        if (root == null) return 0;

        int sum = 0;

        if (root.elem >= low && root.elem <= high) {
            sum += root.elem;
        }

        if (root.elem >= low) {
            sum += rangeSum(root.left, low, high);
        }

        if (root.elem <= high) {
            sum += rangeSum(root.right, low, high);
        }

        return sum;
    }

//Task 5
public static Integer mirrorSum( BSTNode root ){
        //TO DO
        if (root == null) return 0;

        return mirrorSumHelper(root.left, root.right);
    }
    


    private static int mirrorSumHelper(BSTNode left, BSTNode right) {
        if (left == null || right == null) return 0;
        int sum = left.elem + right.elem;
        sum += mirrorSumHelper(left.left, right.right);
        sum += mirrorSumHelper(left.right, right.left);

        return sum;
    }

//Task 6
public static Boolean isBST( BSTNode root ){
        //TO DO
        return checkBST(root, null, null);
    }


    private static Boolean checkBST(BSTNode node, Integer min, Integer max) {
        if (node == null) {
            return true;
        }
        if (min != null && node.elem <= min) {
            return false;
        }
        if (max != null && node.elem >= max) {
            return false;
        }

        boolean leftCheck = checkBST(node.left, min, node.elem);
        boolean rightCheck = checkBST(node.right, node.elem, max);

        return leftCheck && rightCheck;
    }
