public class Task4 {

    


    public static void swapChild( BTNode root, int lvl, int M ){
        if (root==null) {
            return;
        }
        if(lvl<M){
            BTNode t=root.left;
            root.left=root.right;
            root.right=t;
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
        
    }
   
}

public class Task4 {

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null){
        return 0; 
    }
   int s=0;


   if(low<=root.elem && high>= root.elem){
    s+=root.elem;
   }
   if(root.elem>low){
    s+=rangeSum(root.left, low, high);
   }
    if(root.elem<high){
    s+=rangeSum(root.right, low, high);
   }
return s;
    }

}

public class Task5 {

    
    public static Integer mirrorSum( BSTNode root ){
        if(root==null){
        return 0; 
    }
    return helper(root.left,root.right);
}
private static Integer helper(BSTNode l,BSTNode r){
    if(l==null || r==null){
        return 0;
    }
    int s=r.elem+l.elem;

    s+=helper(l.left, r.right);
    s+=helper(l.right, r.left);
    return s;
}


    }

public class Task6 {

    public static Boolean isBST( BSTNode root ){
        
        return helper(root, null, null); 
    }
    private static Boolean helper(BSTNode r,Integer s,Integer l){
        if (r==null) {
           return true; 
        }
        if((s!=null && r.elem<=s)||(l!=null && r.elem>=l) ){
            return false;
        }

        Boolean b=helper(r.left, s, r.elem)&& helper(r.right, r.elem, l);
        return b;
    }

}
