public class Task4 {

    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) {
            return 0;
        }

        int sum = 0;

       
        if (root.elem >= low && root.elem <= high) {
            sum += root.elem;
        }

      
        if (root.elem > low) {
            sum += rangeSum(root.left, low, high);
        }

        if (root.elem < high) {
            sum += rangeSum(root.right, low, high);
        }

        return sum;
    }
}

public class Task5 {

    public static int mirrorSum(BSTNode root) {
        if (root == null) return 0;
        return mirrorSumHelper(root.left, root.right);
    }

    private static int mirrorSumHelper(BSTNode left, BSTNode right) {
        if (left == null || right == null) return 0;

        int sum = left.elem + right.elem;

        sum += mirrorSumHelper(left.left, right.right);   
        sum += mirrorSumHelper(left.right, right.left);   
        return sum;
    }
}

public class Task6 {

    public static boolean isBST(BSTNode root) {
        return isBSTHelper(root, null, null);
    }

    private static boolean isBSTHelper(BSTNode node, Integer min, Integer max) {
        if (node == null) return true;

        if ((min != null && node.elem <= min) || (max != null && node.elem >= max)) {
            return false;
        }

        return isBSTHelper(node.left, min, node.elem) &&
               isBSTHelper(node.right, node.elem, max);
    }
