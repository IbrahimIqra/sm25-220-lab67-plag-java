public static void swapChild( BTNode root, int lvl, int M ){
         //base
        if(lvl>=M || root==null) return;

        BTNode a = root.left;
        root.left = root.right;


        root.right = a;
        //lvl++;
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
}

//task5
public static Integer subtractSummation( BTNode root ){
        Integer a = sum(root.left);
        Integer b = sum(root.right);
        return a-b;
}
 private static Integer sum(BTNode root){
       //base
       if(root.left==null && root.right==null) return (Integer)root.elem;



       return (Integer)root.elem + sum(root.left) + sum(root.right);
 }

//task6
public static Integer levelSum( BTNode root ){
      return sum(root,0);
}
 private static Integer sum(BTNode root, int lvl){
     //base
     if(root==null) return 0;

       Integer sum = 0;
       if(lvl%2==0) sum-=(Integer) root.elem;


       else if(lvl%2!=0) sum+=(Integer) root.elem;

       sum+=sum(root.left,lvl+1)+sum(root.right,lvl+1);

       return sum;
}

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
    //base
    if(root==null) return 0;
        
    Integer sum = 0;
    if(root.elem>=low && root.elem<=high) sum+= root.elem;


    sum+= rangeSum(root.left,low,high)+rangeSum(root.right,low,high);
   
    return sum;
}

public static Integer mirrorSum( BSTNode root ){


        return msum(root.left,root.right);
}
 private static Integer msum(BSTNode leftmost, BSTNode rightmost){
        //base
        if(leftmost==null || rightmost==null) return 0;

        Integer sum = leftmost.elem+rightmost.elem;


        sum = sum + msum(leftmost.left,rightmost.right)+ msum(leftmost.right,rightmost.left);

        return sum;
 }

public static Boolean isBST( BSTNode root ){
        Integer mini = -10000, maxi = 10000;


        return BST(root,mini,maxi);
}
 private static Boolean BST(BSTNode root, Integer min, Integer max){
        if(root == null) return true; //base

        if(root.elem>=max || root.elem<=min) return false;

        Boolean flag = BST(root.left,min,root.elem) && BST(root.right,root.elem,max);
        return flag;
 }
