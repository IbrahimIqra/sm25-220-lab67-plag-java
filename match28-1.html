    public static Integer subtractSummation( BTNode root ){
        //TO DO
        if(root == null){
        return 0;
        }
        int left_tree = sum(root.left);
        int right_tree = sum(root.right);
        return left_tree-right_tree;
    }
    private static int sum(BTNode node){
        if(node == null){
        return 0;
        }
        return (Integer)node.elem + sum(node.left) + sum(node.right);
    }

    private static Integer sum(BTNode node, int level){
        if(node == null){
            return 0;
        }
        int val = (Integer)node.elem;
        if(level%2 == 1){
            return val + sum(node.left, level + 1) + sum(node.right, level + 1);
        } 
        else{
            return -val + sum(node.left, level + 1) + sum(node.right, level + 1);
        }
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        //TO DO
      int sum = 0;
      if(root == null){
        return 0;        
      }
      if(root.elem >= low && root.elem <= high){
        sum += root.elem;
      }
      if(root.elem >= low){
        sum += rangeSum(root.left, low, high);
      }
      if(root.elem <= high){
        sum += rangeSum(root.right, low, high);
      }
        return sum;
    }

public static Integer mirrorSum(BSTNode root){
        //TO DO
        if(root == null){
          return 0;
        }
        return traverse(root.left, root.right);
    }


    private static int traverse(BSTNode left, BSTNode right) {
        if(left == null || right == null){
        return 0;
        }
        int sum = left.elem + right.elem;
        sum += traverse(left.left, right.right);
        sum += traverse(left.right, right.left);
        return sum;
    }

  public static Boolean isBST( BSTNode root ){
        //TO DO
        return check(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }
     private static Boolean check(BSTNode val, int low, int high) {
       if(val == null){
         return true;
       }
       if(val.elem <= low || val.elem >= high){
          return false;
       }
        return check(val.left, low, val.elem) && check(val.right, val.elem, high);
    }
