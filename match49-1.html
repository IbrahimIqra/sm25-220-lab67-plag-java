    public static void swapChild(BTNode root, int lvl, int M) {


        if (root == null || lvl == M) {
            return;
        }


        if (lvl < M) {
            BTNode tem = root.left;
            root.left = root.right;
            root.right = tem;
        }


        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);


    }
    //============================================================================
}

public class Task5 {


    //===================================TASK#5==================================
    public static Integer subtractSummation(BTNode root) {
        if (root == null) {
            return 0;
        }
        int sum_left = dosum(root.left);


        int sum_right = dosum(root.right);


        return sum_left - sum_right;


    }
   
    private static Integer dosum(BTNode r) {
        if (r == null) {
            return 0;
        }
        return (Integer) r.elem + dosum(r.left) + dosum(r.right);
    }
    //============================================================================


}

public class Task6 {


    //===================================TASK#6===================================
    public static Integer levelSum(BTNode root) {
        int count = 0;
        int result = doSum(root, count);
        return result;
    }


    private static Integer doSum(BTNode r, int c) {


        if (r == null) {
            return 0;
        }


        if (c % 2 != 0) {


            return (Integer) r.elem - doSum(r.left, c + 1) - doSum(r.right, c + 1);


        } else {
            return -(Integer) r.elem + doSum(r.left, c + 1) + doSum(r.right, c + 1);
        }


    }
    //============================================================================


}

public class Task4 {


    //===================================TASK#4======================
   
    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {


        Integer sum = 0;
        if(root == null) {
            return 0;
        }


        if(root.elem >= low && root.elem <= high) {


            sum += root.elem;
        }


        if(root.elem > low) {
            sum += rangeSum(root.left,  low, high);
        }


        if(root.elem < high) {
            sum +=  rangeSum(root.right, low, high);
        }


        return sum;


    }


    //===============================================================
}

public class Task5 {


    //===================================TASK#5======================
    public static Integer mirrorSum(BSTNode root) {
        if(root == null) {
            return 0;
        }
        int sum = do_mirror(root.left, root.right);
        return sum;
    }




    private static Integer do_mirror(BSTNode l, BSTNode r) {
        if (l == null || r == null) {
            return 0;
        }


        int sum = l.elem + r.elem;
        sum += do_mirror(l.left, r.right);
        sum += do_mirror(l.right, r.left);


        return sum;


    }
    //===============================================================


}

    public static Boolean isBST(BSTNode root) {
        if(root == null) {
            return true;
        }


        if(root.left == null && root.right == null) {


            return true;


        }
        if(root.left != null && root.left.elem >= root.elem) {
            return false;
        }
        if(root.right != null &&  root.right.elem <= root.elem) {
            return false;
        }


        isBST(root.left);
        isBST(root.right);
        return true;


    }
