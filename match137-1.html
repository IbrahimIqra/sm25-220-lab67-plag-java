public static void swapChild( BTNode root, int lvl, int M ){

                if (root == null){
                         return;}
        if (lvl <= M) {
            BTNode altr = root.left;
            root.left = root.right;
            root.right = altr;
        }

        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }

 public static Integer subtractSummation( BTNode root ){
        if (root == null){ return 0;}

        int l_Sum = sumTree(root.left);
        int r_Sum = sumTree(root.right);

        return l_Sum - r_Sum;
    }

        private static int sumTree(BTNode rt) {
        if (rt == null){ return 0;}

        int val = (Integer) rt.elem;  
        int l = sumTree(rt.left);
        int r = sumTree(rt.right);

        return val + l + r;
    }

public static Integer levelSum(BTNode root) {
        return S_Helper(root,0); 
    }

    private static int S_Helper(BTNode rt, int lvl) {
        if (rt == null){ return 0;}

        int value = (Integer) rt.elem;

        int l = S_Helper(rt.left, lvl + 1);
        int r = S_Helper(rt.right, lvl + 1);

        if (lvl % 2 == 0) {
            return -value + l + r;
        } else {
            return value + l + r;
        }
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        
        if (root == null){return 0;}



        int val = (Integer) root.elem;

        if (val < low) {
            
            return rangeSum(root.right, low, high);

        }
        else if (val > high) {
            return rangeSum(root.left, low, high);
        } 
       else {
            
            return val + rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
        }
    
    }

public static Integer mirrorSum( BSTNode root ){
      
    if(root == null){ return 0;}
        return ms_Helper(root.left, root.right);
    }

    private static int ms_Helper(BSTNode left, BSTNode right){
        if(left == null || right == null) return 0;

        int sum = 0;
        


        sum += (int)left.elem + (int)right.elem;

        sum += ms_Helper(left.left, right.right);
        sum += ms_Helper(left.right, right.left);

        return sum;
    }

public static Boolean isBST( BSTNode root ){
        
        return is_BST_Helper(root, null, null);
    }

    private static boolean is_BST_Helper(BSTNode rt, Integer min_than_rt, Integer max_than_rt) {
        if (rt == null){ return true;}

        int val = (int) rt.elem;

        
        if (min_than_rt != null && val <= min_than_rt) {return false;}
        if (max_than_rt != null && val >= max_than_rt) {return false;}
       
        // checking children count 
        int c_Count = 0;
        if (rt.left != null) {c_Count++;}
        if (rt.right != null) {c_Count++;}
        
        if (c_Count > 2) {
            
            return false;
        }

        
        return is_BST_Helper(rt.left, min_than_rt, val) &&
               is_BST_Helper(rt.right, val, max_than_rt);
    }
