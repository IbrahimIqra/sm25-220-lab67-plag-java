public class Task6 {
    public static Boolean isBST(BSTNode root){
        return checkBST(root,null,null);
    }


    private static Boolean checkBST(BSTNode cur,Integer low,Integer high){
        if(cur==null)
            return true;


        int v=cur.elem;


        if(low!=null&&v<=low)  
            return false;


        if(high!=null&&v>=high)
            return false;
       
        if(!checkBST(cur.left,low,v))  
            return false;


        if(!checkBST(cur.right,v,high) )
            return false;


        return true;
    }
}
