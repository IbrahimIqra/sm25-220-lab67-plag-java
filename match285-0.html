public class Task4 {
  public static void swapChild( BTNode root, int lvl, int M ){
    if(root == null) 
      return;
    if(lvl >= M){
      BTNode temp = root.left;
      root.left = root.right;


      root.right = temp;
    }
    swapChild(root.left, lvl + 1, M);
    swapChild(root.right, lvl + 1, M);
  }
}

public class Task5 {
  public static Integer subtractSummation( BTNode root ){
    if(root == null) 
      return 0;
    int leftsum = sumTree(root.left);
    int rightsum = sumTree(root.right);
    return leftsum - rightsum;
  }
  private static int sumTree(BTNode node){
    if(node == null) 
      return 0;
    int val = (Integer) node.elem; 


    return val + sumTree(node.left) + sumTree(node.right);
  }
}

public class Task6 {
  public static Integer levelSum( BTNode root ){
    return helper(root, 1);
  }
  private static int helper(BTNode node, int level){
    if(node == null) 
      return 0;
    int value = (Integer)node.elem;
    if(level % 2 == 1){
      return value + helper(node.left, level + 1) + helper(node.right, level + 1);
    } 
    else{


      return -value + helper(node.left, level + 1) + helper(node.right, level + 1);
    }
  }
}

public class Task4 {
  public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
    if(root == null){
      return 0;
    }
    int sum = 0;
    if(root.elem >= low && root.elem <= high){


      sum += root.elem;
    }
    if(root.elem > low){
      sum += rangeSum(root.left, low, high);
    }
    if(root.elem < high){
      sum += rangeSum(root.right, low, high);
    }
    return sum;
  }
}

//Assignment07_Task05
public class Task5 {
  public static Integer mirrorSum( BSTNode root ){
    if(root == null){
      return 0;
    }
    return mirror_helper(root.left, root.right);
  }
  private static int mirror_helper(BSTNode left, BSTNode right){
    if(right == null || left == null){
      return 0;
    }
    int sum = left.elem + right.elem;
    sum += mirror_helper(left.left, right.right);
    sum += mirror_helper(left.right, right.left);
    return sum;
  }
}

public class Task6 {
  public static Boolean isBST( BSTNode root ){
    if(root == null){
      return true;
    }
    if(!smaller(root.left, root.elem)){
      return false;
    }
    if(!greater(root.right, root.elem)){
      return false;
    }
    return isBST(root.left) && isBST(root.right);
  }
  private static boolean smaller(BSTNode node, int val){
    if(node == null) 
      return true;
    if(node.elem >= val) 
      return false;
    return smaller(node.left, val) && smaller(node.right, val);
  }
  private static boolean greater(BSTNode node, int val){
    if(node == null) 
      return true;
    if (node.elem <= val) 
      return false;
    return greater(node.left, val) && greater(node.right, val);
  }
}
