public class Task4 {

    public static void swapChild( BTNode root, int lvl, int M ){
        if (root==null|| lvl>=M){
            return;
        }
            BTNode current = root.left;
            root.left=root.right;
            root.right=current;   

            

        swapChild( root.left, lvl+1, M );
        swapChild( root.right, lvl+1, M );
    }
    }

public class Task5 {

    public static Integer subtractSummation( BTNode root ){
        if (root==null){
            return 0;
        }
        int leftSum=sumTree(root.left);
        int rightSum=sumTree(root.right);
        return leftSum-rightSum; //remove this line
    }
    private static int sumTree(BTNode node){
        if (node==null){
            return 0;
        }
        return (Integer)node.elem+sumTree(node.left)+sumTree(node.right);
    }
    
}

public class Task6 {

    public static Integer levelSum( BTNode root ){
        return helper (root,0);
        
    }
    private static int helper (BTNode node, int lvl){
        if (node==null){
            return 0;
        }
        int val =(int)node.elem;
        if(lvl%2==0){
            val=-val;
        }
        return val
        +helper(node.left,lvl+1)
        +helper(node.right,lvl+1);
    }
    //============================================================================

}

public class Task4 {

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if (root == null) {
            return 0;
        }
        int sum=0;
        if (root.elem>=low&& root.elem<=high){
            sum+=root.elem;
        }
        if(root.elem>low){
            sum+=rangeSum(root.left,low,high);
        }
        if(root.elem<high){
            sum+=rangeSum(root.right,low,high);
        }
        return sum; // remove this line
    }
    //===============================================================


}

public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
        if (root== null){
            return 0;
        }

        return sumM(root.left,root.right);
    }
    private static int sumM(BSTNode left , BSTNode right){
        if (left == null || right==null){
            return 0;
        }
        int sum = left.elem + right.elem;

        sum += sumM(left.left, right.right);
        sum += sumM(left.right, right.left);

        return sum;
    }
    //===============================================================


}

public class Task6 {

      public static Boolean isBST( BSTNode root ){
        return helper (root,null,null);

    }
    private static boolean helper (BSTNode node, Integer minimum, Integer maximum){
        if (node==null){
            return true;

        }


        if ((minimum != null && node.elem <= minimum)||(maximum != null && node.elem >= maximum)){
            return false;
        }
        return helper (node.left, minimum, node.elem)&&
        helper (node.right, node.elem, maximum);
    }
    //===============================================================


}
