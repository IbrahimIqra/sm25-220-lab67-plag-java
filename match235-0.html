public class Task4 {



    public static void swapChild(BTNode root, int l, int X){
        if (root == null) return;
        if (l >= X) {
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left, l + 1, X);
        swapChild(root.right, l + 1, X);
    }
}

public class Task5 {
    public static Integer subtractSummation(BTNode root){
        if(root == null) return 0;
        return addNodes(root.left) - addNodes(root.right);
    }

    private static int addNodes(BTNode node){
        if(node == null) return 0;
        int t = (Integer) node.elem;
        t += addNodes(node.left);
        t += addNodes(node.right);
        return t;
    }
  
}

public class Task6 {



    public static Integer levelSum(BTNode root){
        return lvlDiff(root, 1);
    }

    private static int lvlDiff(BTNode node, int l){
        if(node == null) return 0;

        int v = (Integer) node.elem;
        int sign = 1;
        if(l % 2 == 0) {
            sign = -1;
        }

        int leftSum = lvlDiff(node.left, l + 1);
        int rightSum = lvlDiff(node.right, l + 1);

        return sign * v + leftSum + rightSum;
    }

}

public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
    if (root == null) return 0;

    int sum = 0;

      if (root.elem > low) {
        sum += rangeSum(root.left, low, high);
    }
    if (root.elem >= low && root.elem <= high) {
        sum += root.elem;
    }
    if (root.elem < high) {
        sum += rangeSum(root.right, low, high);
    }

    return sum;
}

public class Task5 {


    public static Integer mirrorSum(BSTNode root) {
        if (root == null) return 0;
        return mirrorSumHelper(root.left, root.right);
    }

    private static int mirrorSumHelper(BSTNode left, BSTNode right) {
        if (left == null || right == null) 
            return 0;

        return left.elem + right.elem
               + mirrorSumHelper(left.left, right.right)
               + mirrorSumHelper(left.right, right.left);
    }

}

public class Task6 {
    public static Boolean isBST(BSTNode root) {
        if (root == null) return true;


        return isBSTHelper(root, null, null);
    }
   private static boolean isBSTHelper(BSTNode node, Integer min, Integer max) {
        if (node == null) return true;

        if (min != null && node.elem <= min) return false;
        if (max != null && node.elem >= max) return false;

        return isBSTHelper(node.left, min, node.elem) &&
               isBSTHelper(node.right, node.elem, max);
    }
}
