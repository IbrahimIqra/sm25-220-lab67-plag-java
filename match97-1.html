public class Task5 {



    public static Integer subtractSummation( BTNode root ){
        if (root == null) {
            return 0;
        }
        int leftSum = sumSubtree(root.left);
        int rightSum = sumSubtree(root.right);
        return leftSum - rightSum;
    }
    private static int sumSubtree(BTNode node) {
        if (node == null) {
            return 0;
        }
        return (Integer) node.elem + sumSubtree(node.left) + sumSubtree(node.right);
    }
    }

public class Task6 {



    public static Integer levelSum( BTNode root ){
        return levelSumHelper(root, 0);
    }
    private static int levelSumHelper(BTNode node, int level) {
        if (node == null) {
            return 0;
        }
        int leftSubtreeSum = levelSumHelper(node.left, level + 1);
        int rightSubtreeSum = levelSumHelper(node.right, level + 1);
        int currentValue = (Integer) node.elem;
        if (level % 2 == 0) { 
            return leftSubtreeSum + rightSubtreeSum - currentValue;
        } else { 
            return leftSubtreeSum + rightSubtreeSum + currentValue;
        }
    }
   
   }


public class Task4 {

       public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if (root == null) {
            return 0;
        }

        int sum = 0;
        int currentValue = root.elem;
        if (currentValue >= low && currentValue <= high) {
            sum += currentValue;
            sum += rangeSum(root.left, low, high);
            sum += rangeSum(root.right, low, high);
        }
        else if (currentValue < low) {
            sum += rangeSum(root.right, low, high);
        }
        else { 
            sum += rangeSum(root.left, low, high);
        }

        return sum;
    }
}


public class Task5 {



        public static Integer mirrorSum( BSTNode root ){
        if (root == null || root.left == null || root.right == null) {
            return 0;
        }
        return mirrorSumHelper(root.left, root.right);
    }
    private static int mirrorSumHelper(BSTNode node1, BSTNode node2) {
        if (node1 == null || node2 == null) {
            return 0;
        }
        int currentSum = node1.elem + node2.elem;
        int outerSum = mirrorSumHelper(node1.left, node2.right);
        int innerSum = mirrorSumHelper(node1.right, node2.left);
        return currentSum + outerSum + innerSum;
    }
   

}


public class Task6 {



        public static Boolean isBST( BSTNode root ){
        return isBSTHelper(root, null, null);
    }
    private static boolean isBSTHelper(BSTNode node, Integer min, Integer max) {
        if (node == null) {
            return true;
        }
        if (min != null && node.elem <= min) {
            return false;
        }
        if (max != null && node.elem >= max) {
            return false;
        }
        return isBSTHelper(node.left, min, node.elem) && isBSTHelper(node.right, node.elem, max);
    }
