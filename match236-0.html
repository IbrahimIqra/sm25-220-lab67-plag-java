public class Task4 {
    public static void swapChild( BTNode root, int lvl, int M ){
        if (root == null)
            return;
        if (lvl < M) {
            BTNode temp = root.left;
            root.left  = root.right;
            root.right = temp;
        }
        swapChild(root.left,  lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }
}

public class Task5 {

    public static Integer subtractSummation( BTNode root ){
        if (root == null)
            return 0;
        return sum(root.left) - sum(root.right);
    }
    private static int sum(BTNode node) {
        if (node == null)
            return 0;
        return (Integer) node.elem + sum(node.left) + sum(node.right);
    }

}

public class Task6 {

    public static Integer levelSum( BTNode root ){

        return level_Sum(root, 0);
    }
    private static Integer level_Sum(BTNode node, int level) {
        if (node == null)
            return 0;
        int val = (Integer) node.elem;

        int sum;
        if (level % 2 == 0) {
            sum = -val;
        } else {
            sum = val;
        }
        return sum + level_Sum(node.left,  level + 1) + level_Sum(node.right, level + 1);
    }

}

public class Task4 {

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if (root == null) return 0;

        int sum = 0;



        if (root.elem >= low && root.elem <= high) {
            sum += root.elem;
        }
        if (root.elem > low) {
            sum += rangeSum(root.left, low, high);
        }
        if (root.elem < high) {
            sum += rangeSum(root.right, low, high);
        }

        return sum;
    }
}


Assignment Task = 05




public class Task5 {

    public static Integer mirrorSum(BSTNode root) {
        if (root == null)
            return 0;
        return new_mirrorSumH(root.left, root.right);
    }
    private static Integer new_mirrorSumH(BSTNode node1, BSTNode node2) {

        if (node1 == null || node2 == null)
            return 0;
        int tmp = node1.elem + node2.elem;

        int l = new_mirrorSumH(node1.left, node2.right);
        int r = new_mirrorSumH(node1.right, node2.left);

        return tmp + l + r;
    }
}

public class Task6 {
    


    public static Boolean isBST(BSTNode root) {
        return checkBST(root, null, null);
    }

    private static Boolean checkBST(BSTNode root, Integer m, Integer n) {
        if (root == null)
            return true;
        if ((m != null && root.elem <= m) || (n != null && root.elem >= n))
            return false;

        if (!checkBST(root.left, m, root.elem))
            return false;
        if (!checkBST(root.right, root.elem, n))
            return false;
        return true;
    }
}
