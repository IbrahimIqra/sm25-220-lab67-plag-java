public static void swapChild(BTNode root, int lvl, int M) {
  if (root == null) {
    return;
  }
  swapHelper(root, M);
}

private static void swapHelper(BTNode node, int M) {
  if (node == null) {
    return;
  }
  
  if (M <= 0) {
    BTNode temp = node.left;
    node.left = node.right;
    node.right = temp;
  }
  
  swapHelper(node.left, M - 1);
  swapHelper(node.right, M - 1);
}

public static Integer subtractSummation(BTNode root) {
  if (root == null) {
    return 0;
  }
  return sum(root.left) - sum(root.right);
}



private static int sum(BTNode node) {
  if (node == null) {
    return 0;
  }
  return (Integer) node.elem + sum(node.left) + sum(node.right);
}



  


  public static int levelSum(BTNode node, int level) {
    if (node == null){ 
      return 0;
    }
    int val = (Integer) node.elem;
    if (level % 2 == 0){
      val = -val;
    }
    return val + levelSum(node.left, level + 1) + levelSum(node.right, level + 1);
  }
  
  public static int levelSum(BTNode root) {
    return levelSum(root, 1);
  }
}

   public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
  if (root == null) {
    return 0;
  }
  int sum = 0;
  if (root.elem >= low && root.elem <= high) {
    sum += root.elem;
  }
  sum += rangeSum(root.left, low, high);


  sum += rangeSum(root.right, low, high);
  return sum;
}

//Task05

public static Integer mirrorSum(BSTNode root) {
  if (root == null) {
    return 0;
  }
  return mirrorSumHelper(root.left, root.right);
}

private static int mirrorSumHelper(BSTNode left, BSTNode right) {
  if (left == null || right == null){
    return 0;
  }
  int sum = left.elem + right.elem;
  sum += mirrorSumHelper(left.left, right.right);
  sum += mirrorSumHelper(left.right, right.left);
  return sum;
}

  public static Boolean isBST(BSTNode root) {
    return isBST(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
  }
  
  private static Boolean isBST(BSTNode node, int min, int max) {
    if (node == null) {
      return true;
    }
    if (node.elem <= min || node.elem >= max) {
      return false;
    }
    return isBST(node.left, min, node.elem) && isBST(node.right, node.elem, max);
  }
