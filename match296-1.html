public class Task4 {

    public static void swapChild( BTNode root, int lvl, int M ){
        if (root == null) 
            return;
        if (lvl==M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
            
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }
}

        public static int Sum(BTNode root,int level) {
            if(root== null){
                return 0;}
            int val =(Integer)root.elem;

            if (level%2== 0){
                return -val+Sum(root.left,level+1)+Sum(root.right,level+1);
            }
            else{
                return val+Sum(root.left,level+1)+Sum(root.right,level+1);
            }
        }
    }

public class Task4 {
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
      if(root == null){
        return 0;}
      int s= 0;
      if(root.elem>= low && root.elem<=high){
        s+= root.elem;
      }
      if(root.elem>low){
            s+= rangeSum(root.left,low,high);
        }
      if(root.elem<high) {
            s+= rangeSum(root.right, low, high);
        }
    return s;
  
    }

public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
        if(root==null){
            return 0;}
        int sum= mirror(root.left,root.right);
        return sum;
        }


        private static int mirror(BSTNode left,BSTNode right){
           if(left==null || right==null){
              return 0;}

               int sum= left.elem+right.elem;
               sum += mirror(left.left, right.right);
               sum += mirror(left.right, right.left);
        return sum;

}}

public class Task6 {
    public static Boolean isBST( BSTNode root ){
       return helper(root,null,null);}


    private static Boolean helper(BSTNode root, Integer x, Integer y){
        if(root== null){
            return true;}
        if((x!=null && root.elem<=x)||(y!= null &&root.elem >=y)){
           return true;
        }
        return helper(root.left,x,root.elem)&& helper(root,root.elem,y); 
    }}
