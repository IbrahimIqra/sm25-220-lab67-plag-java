public static void swapChild( BTNode root, int lvl, int M ){
        if(root==null){
            return;
        }
        if(lvl<M){
            BTNode temp=root.left;
            root.left=root.right;


            root.right=temp;
            
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }

public static Integer subtractSummation( BTNode root ){
        
       if(root==null){
        return -1;
       }
       Integer leftSum=sum(root.left);
       Integer rightSum=sum(root.right);
       return leftSum-rightSum;
    }
       public static Integer sum(BTNode node){
        if(node==null){
            return -1;
        }


      return (Integer)node.elem+sum(node.left)+sum(node.right);
       
    }

public static Integer levelSum( BTNode root ){
        return Helper(root,0);
    }
    public static int Helper(BTNode root, int lvl){
        if(root==null){
            return 0;
        }
        int val=(Integer)root.elem;
        int curr=0;
        if(lvl%2==0){
            curr-=val;
        }
        else{
            curr+=val;
        }
        int right=Helper(root.right,lvl+1);
        int left=Helper(root.left,lvl+1);
        return curr+right+left;
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        int sum=0;
        if(root==null){
        return 0;
        }
       int val=(Integer)root.elem;
       if(val>=low && val<=high){
        sum+=root.elem;
         
       }
       if(val>low){
         sum+=rangeSum(root.left,low,high);
       }


       if(val<high){
         sum+=rangeSum(root.right,low,high);    
       }
      
       return sum;
    }

 public static Integer mirrorSum( BSTNode root ){
        if(root==null){
            return 0;
        }
        return Helper(root.left, root.right);
    }
    

    public static Integer Helper(BSTNode node1, BSTNode node2){
        if(node1==null || node2==null){
            return 0;
        }
        int sum=node1.elem+node2.elem;
        sum+=Helper(node1.left,node2.right);
        sum+=Helper(node1.right,node2.left);
        return sum;
    }

public static Boolean isBST( BSTNode root ){
        if(root==null){
            return true;
        }


        return Helper(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }
    public static boolean Helper(BSTNode node, int min,int max){
        if(node==null){
            return true;
        }
        if(node.elem<=min || node.elem>=max){
            return false;
        }
        return Helper(node.left,min,node.elem) && Helper(node.right,node.elem,max);
    }
