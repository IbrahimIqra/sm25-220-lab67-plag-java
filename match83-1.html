public static void swapChild( BTNode root, int lvl, int M ){
        if(root == null){
            return;
        }
        if(lvl<M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }

public static Integer subtractSummation( BTNode root ){
        if(root == null){
            return 0;
        }


        int sumL = calculateSum(root.left);
        int sumR = calculateSum(root.right);
        return sumL - sumR; 
    }
    public static int calculateSum(BTNode node){
        if(node == null){
            return 0;
        }
        int curr = (Integer)node.elem;
        int leftSum = calculateSum(node.left);
        int rightSum = calculateSum(node.right);
        return curr + leftSum + rightSum;
    }


public static Integer levelSum( BTNode root ){
        if(root == null){
            return 0;
        }


        int [] sum = new int[2];
        calculateSum(root, 0, sum);
        return sum[1] - sum[0];
    }
    public static void calculateSum(BTNode node, int level, int [] sum){
        if(node == null){
            return;
        }
        int val = (Integer)node.elem;
        if(level%2 == 0){
            sum[0] += val;
        }
        else{


            sum[1]+= val;
        }
        calculateSum(node.left, level+1, sum);
        calculateSum(node.right, level+1, sum);
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root == null){
            return 0;
        }
        int sum = 0;
        if(root.elem>= low && root.elem<= high){
            sum+= root.elem;
        }
        sum+= rangeSum(root.left, low, high);


        sum+= rangeSum(root.right, low, high);

        return sum;
    }

public static Integer mirrorSum( BSTNode root ){
        if(root == null){
            return 0;
        }
        return sumHelper(root.left, root.right);
    }
    public static Integer sumHelper(BSTNode left, BSTNode right){
        if(left == null || right == null){
            return 0;
        }
        int sum = left.elem + right.elem + sumHelper(left.left, right.right) + sumHelper(left.right, right.left);
        return sum;
    }

public static Boolean isBST( BSTNode root ){
        if(root == null){
            return false;
        }
        if(root.right!= null && root.right.elem<root.elem){
            return false;
        }
        if(root.left!= null && root.left.elem>root.elem){
            return false;
        }
        isBST(root.left);
        isBST(root.right);
        return true;
    }
