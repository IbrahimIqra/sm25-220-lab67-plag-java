public class Task4 {
    public static void swapChild( BTNode root, int lvl, int M ){
        mirror(root,lvl,M);
    }
     public static BTNode mirror( BTNode root, int lvl, int M ){
        if(root == null){
            return null;
        }

        if(lvl >= M){


            root.left = mirror(root.left,lvl+1,M);
            root.right = mirror(root.right,lvl+1,M);
            return root;
        }else{
            BTNode temp = root.left;

            root.left = mirror(root.right,lvl+1,M);
            root.right = mirror(temp,lvl+1,M);
            return root;
        }
    }
}

public class Task5 {


    public static Integer subtractSummation( BTNode root ){
        Integer sumOfLeft = sum(root.left);
        Integer sumOfRight = sum(root.right);
        return sumOfLeft - sumOfRight;
    }

        public static int sum(BTNode root){
            if(root == null){
                return 0;
            }
            int left = sum(root.left);
            int right = sum(root.right);

            return (int)root.elem+left+right;
        }
}

    public static Integer levelSum( BTNode root ){
        return helper(root,0);
    }

    public static int helper(BTNode root,int lvl){
        if(root == null){
            return 0;
        }

        if(lvl % 2 == 0){
            return (-1*(int)root.elem + helper(root.left,lvl+1) + helper(root.right,lvl+1));
        }else{
            return ((int)root.elem + helper(root.left,lvl+1) + helper(root.right,lvl+1));
        }
    }
}

public class Task4 {


    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root == null){
            return 0;
        }

        int left = rangeSum(root.left,low,high);
        int right = rangeSum(root.right,low,high);

        int sum = left+right;

        if(root.elem>=low && root.elem<=high){
            sum+=root.elem;
        }
        return sum;
    }
}




public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
        if(root == null){
            return 0;
        }

        return helper(root.left,root.right);
    }

    public static int helper(BSTNode left,BSTNode right){
        if(left == null || right == null){
            return 0;
        }else{


            int sum = (int)left.elem+(int)right.elem;
            sum += helper(left.left,right.right);
            sum += helper(left.right,right.left);
            return sum;
        }
    }
}


public class Task6 {

    public static Boolean isBST( BSTNode root ){


        if(root == null || (root.left == null && root.right == null)){
            return true;
        }

        int leftMax = max(root.left);
        int rightMin = min(root.right);

        if(root.elem>=leftMax && root.elem<=rightMin){
            return true;
        }else{
            return false;
        }
    }

        public static int max(BSTNode root){
            if(root == null){
                return Integer.MIN_VALUE;
            }


            int left = max(root.left);
            int right = max(root.right);
            return Math.max(root.elem,Math.max(left,right));
        }

        public static int min(BSTNode root){
            if(root == null){
                return Integer.MAX_VALUE;
            }


            int left = min(root.left);
            int right = min(root.right);
            return Math.min(root.elem,Math.min(left,right));
        }
}
