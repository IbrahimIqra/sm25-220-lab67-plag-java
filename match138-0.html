public class Task6 {


    public static Integer levelSum(BTNode root) {
        return levelSumHelper(root, 0);
    }


    private static Integer levelSumHelper(BTNode node, int level) {
        if (node == null) {
            return 0;
        }


        int nodeValue = (Integer) node.elem;


        int leftSum = levelSumHelper(node.left, level + 1);
        int rightSum = levelSumHelper(node.right, level + 1);


        if (level % 2 == 0) {
            return nodeValue - leftSum - rightSum;
        } else {
            return -nodeValue + leftSum + rightSum;
        }
    }
}

public class Task4 {


    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        return rangeSumHelper(root, low, high);
    }




    private static Integer rangeSumHelper(BSTNode node, Integer low, Integer high) {
        if (node == null) {
            return 0;
        }


        int sum = 0;
        if (node.value >= low && node.value <= high) {
            sum += node.value;
        }


        if (node.value < low) {
            sum += rangeSumHelper(node.right, low, high);
        }
        else if (node.value > high) {
            sum += rangeSumHelper(node.left, low, high);
        }
        else {
            sum += rangeSumHelper(node.left, low, high);


            sum += rangeSumHelper(node.right, low, high);
        }


        return sum;
    }
}


public class Task5 {
 
    public static Integer mirrorSum(BSTNode root) {
        if (root == null) {
            return 0;
        }
        return mirrorSumHelper(root.left, root.right);
    }


    public static Integer mirrorSumHelper(BSTNode left, BSTNode right) {
        if (left == null && right == null) {
            return 0;
        }


        if (left == null || right == null) {
            return 0;
        }


        int sum = left.value + right.value;


        sum += mirrorSumHelper(left.left, right.right);
        sum += mirrorSumHelper(left.right, right.left);


        return sum;
    }
}


public class Task6 {


    public static Boolean isBST(BSTNode root) {
        return isBSTUtil(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }


    public static Boolean isBSTUtil(BSTNode node, int min, int max) {


        if (node == null) {
            return true;
        }


        if (node.value < min || node.value > max) {
            return false;
        }


        return isBSTUtil(node.left, min, node.value - 1) && isBSTUtil(node.right, node.value + 1, max);
    }
