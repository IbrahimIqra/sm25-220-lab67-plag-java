public class Task5 {
    
    public static Integer subtractSummation( BTNode root ){
        if (root == null) return null;
        
        int leftSum  = subtreeSum(root.left);
        int rightSum = subtreeSum(root.right);
        
        return Integer.valueOf(leftSum - rightSum);
    }
    
    private static int subtreeSum(BTNode node) {
        if (node == null) return 0;
        
        int val = (Integer) node.elem;
        return val + subtreeSum(node.left) + subtreeSum(node.right);
    }

}


//TASK06
public class Task6 {




    public static Integer levelSum(BTNode root){
        return levelSumHelper(root, 0); 
    }


    private static int levelSumHelper(BTNode node, int level){
        if (node == null) {
            return 0;
        }

        int val = (Integer) node.elem; 
        int leftSum = levelSumHelper(node.left, level + 1);
        int rightSum = levelSumHelper(node.right, level + 1);

        if (level % 2 == 1) {  
            return val + leftSum + rightSum;
        } else {             
            return -val + leftSum + rightSum;
        }
    }
    

}

public class Task4 {

 
    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) {
            return 0; 
        }

        int sum = 0;

        if (root.elem >= low && root.elem <= high) {
            sum += root.elem;
        }
        
        if (root.elem > low) {
            sum += rangeSum(root.left, low, high);
        }

        if (root.elem < high) {
            sum += rangeSum(root.right, low, high);
        }

        return sum;
    }
    

}


//TASK 05

public class Task5 {

   
    public static Integer mirrorSum(BSTNode root) {
        if (root == null) {
            return 0;
        }
       
        return mirrorSum(root.left, root.right);
    }

    
    public static Integer mirrorSum(BSTNode leftNode, BSTNode rightNode) {
        if (leftNode == null || rightNode == null) {
            return 0;
        }

        int sum = leftNode.elem + rightNode.elem;

        sum += mirrorSum(leftNode.left, rightNode.right);
        sum += mirrorSum(leftNode.right, rightNode.left);

        return sum;
    }
   

}


//Task 6
public class Task6 {

    public static Boolean isBST(BSTNode root) {
        return checkBST(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }

    private static Boolean checkBST(BSTNode node, int min, int max) {
        if (node == null) {
            return true; 
        }

        if (node.elem <= min || node.elem >= max) {
            return false;
        }

        
        return checkBST(node.left, min, node.elem) && checkBST(node.right, node.elem, max);
    }
    
}
