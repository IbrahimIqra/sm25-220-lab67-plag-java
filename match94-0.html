public static void swapChild( BTNode root, int lvl, int M ){
        if( root == null)
        {
          return ;
        }
        
        if(lvl<= M)
        {
          BTNode temp=root.left;
          root.left=root.right;
          root.right=temp;
        }
        
        swapChild(root.left,lvl+1,M);
        swapChild(root.right,lvl+1,M);
        
    }

public static Integer subtractSummation( BTNode root ){
      if(root == null)
      {
        return 0;
      }
      
      int leftSum = calculate(root.left);
      
      int rightSum = calculate(root.right);
      
      return leftSum-rightSum;
    }
    
    public static Integer calculate(BTNode node)
    {
      if(node == null)
      {
        return 0;
      }
      
      return calculate(node.left) + (Integer)node.elem + calculate(node.right);
    }

    public static Integer levelSum( BTNode root ){
      if(root == null)
      {
        return 0;
      }
      
      return calculate (root,0);
    }
    
    private static Integer calculate(BTNode node,int lvl)
    {
      if(node == null)
      {
        return 0;
      }
      
      if(lvl % 2==0)
      {
        return -(Integer)node.elem + calculate (node.left,lvl+1) + calculate (node.right,lvl+1);
      }
      else
      {
        return (Integer)node.elem + calculate (node.left,lvl+1) + calculate (node.right,lvl+1);
      }
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
      if(root==null)
      {
        return 0;
      }
      int sum=0;      
      if(root.elem >= low && root.elem <=high)
      {
        sum+=root.elem;
      }
      if(root.elem>low)
      {
        sum+=rangeSum(root.left,low,high);
      }
      if(root.elem<high)
      {
        sum+=rangeSum(root.right,low,high);
      }
      return sum;
    }

public static Integer mirrorSum( BSTNode root ){
      if(root == null)
      {
        return 0;
      }
      return helper(root.left,root.right);
    }
    
    private static Integer helper(BSTNode left,BSTNode right)
    {
      if(left==null || right == null)
      {
        return 0;
      }
      int sum = left.elem + right.elem;
      sum+=helper(left.left,right.right);
      sum+=helper(left.right,right.left);
      return sum;
    }

public static Boolean isBST( BSTNode root ){
      return helper(root,null,null);
    }
    
    private static Boolean helper(BSTNode node,Integer min,Integer max)
    {
      if(node==null)
      {
        return true;
      }
      
      if(min != null && node.elem <=min)
      {
        return false;
      }


      if (max != null && node.elem >= max)
      {
        return false;
      }   
      
      return helper(node.left, min, node.elem) && helper(node.right, node.elem, max);
    }
