public static void swapChild(BTNode root, int lvl, int M) {
    if (root == null || lvl >= M) return;
    
    BTNode temp = root.left;
    root.left = root.right;
    root.right = temp;

    swapChild(root.left, lvl + 1, M);
    swapChild(root.right, lvl + 1, M);
    }

private static int sumCalc(BTNode root){
    if (root == null) return 0;

    int leftSum = sumCalc(root.left);
    int rightSum = sumCalc(root.right);

    return (Integer) root.elem + leftSum + rightSum;

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if (root == null) return 0;

            int sum = 0;

            if (root.elem >= low && root.elem <= high) {


                sum += root.elem;
            }

            sum += rangeSum(root.left, low, high);
            sum += rangeSum(root.right, low, high);

            return sum;
        }

//Lab7 Task 5 A5


    public static int mirrorSum(BSTNode root) {
        if (root == null) return 0;
        return mirrorSumHelper(root.left, root.right);
    }

    private static int mirrorSumHelper(BSTNode leftSubTree, BSTNode rightSubTree) {
        if (leftSubTree == null || rightSubTree == null) return 0;
        
        int sum = leftSubTree.elem + rightSubTree.elem;
        sum += mirrorSumHelper(leftSubTree.left, rightSubTree.right);
        sum += mirrorSumHelper(leftSubTree.right, rightSubTree.left);

        return sum;
    }

    private static Boolean helpBST(BSTNode root, BSTNode min, BSTNode max){


        if (root == null)  return true;
        if (min != null && root.elem <= min.elem)  return false;
        if (max != null && root.elem >= max.elem)  return false;

        return helpBST(root.left, min, root) && helpBST(root.right, root, max);
    }
    public static Boolean isBST( BSTNode root ){
      
        return helpBST(root, null, null); // remove this line
