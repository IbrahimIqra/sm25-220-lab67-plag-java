public static void swapChild( BTNode root, int lvl, int M ){
         if(root == null){
           return;
         }

        if(lvl < M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }

public static Integer subtractSummation( BTNode root ){
        if(root == null){
            return 0;        
        }
        return sumleft(root.left) - sumright(root.right);    
    }

     private static Integer sumleft(BTNode left_root){
       if(left_root == null){
          return 0;
       }
         int sum = (Integer)left_root.elem;
         
           sum += sumleft(left_root.left);
           sum += sumleft(left_root.right);
        return sum;
     }

        private static Integer sumright(BTNode right_root){
       if(right_root == null){
          return 0;
       }
         int sum = (Integer)right_root.elem;
         
           sum += sumright(right_root.left);
           sum += sumright(right_root.right);
        return sum; 
    }

public static Integer levelSum( BTNode root ){
        if(root == null){
            return 0;
        }

            Integer output = helper(root, 0);
            return output;
        }
        private static Integer helper(BTNode root, int level){
            if(root == null){
            return 0;
            }

            int val;
            if(level % 2 == 0){
                val = -(Integer)root.elem;
            }
            else{
                val = (Integer)root.elem;
            }

            val += helper(root.left, level+1);
            val += helper(root.right, level+1);
            return val;
        }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root == null){
            return 0;
        }
        int sum = 0;
        if(root.elem >= low && root.elem <= high){
            sum = root.elem;
        }
        sum += rangeSum(root.left, low, high);
        sum += rangeSum(root.right, low, high);


        return sum; 
    }

public static Integer mirrorSum( BSTNode root ){
        if(root == null){
            return 0;
        }
        return mirrorsum_helper(root.left, root.right); 
    }
    public static Integer mirrorsum_helper(BSTNode left_child ,BSTNode right_child) {
        Integer sum = 0;
        if(left_child == null || right_child == null){
            return 0;
        }
        else{


            sum += (Integer)left_child.elem + (Integer)right_child.elem;
        }
        sum += mirrorsum_helper(left_child.left, right_child.right );
        sum += mirrorsum_helper(left_child.right, right_child.left);
        return sum;
    }
