public class Task4 {

    public static void swapChild(BTNode root, int lvl, int M) {
        childswapper(root, lvl, M);
    }


    private static void childswapper(BTNode root, int lvl, int M) {
        if (root == null) {
            return;
        }
        if (lvl < M) {
            BTNode t = root.right;
            root.left = root.right;
            root.right = t;
        }
        childswapper(root.left,lvl+1,M);
        childswapper(root.right,lvl+1,M);
    }
}

public class Task5 {
    public static Integer subtractSummation(BTNode root) {
        if (root == null) {
            return 0;
        }
        int lSum = sum(root.left);
        int rSum = sum(root.right);
        return lSum - rSum;
    }
    private static Integer sum(BTNode node) {
        if (node == null) {
            return 0;
        }
        int nodeValue = (Integer) node.elem;
        return nodeValue + sum(node.left) + sum(node.right);
    }
}

public class Task6 {

    public static Integer levelSum(BTNode root) {
        if (root == null) {
            return 0;
        }


        return lvldiff(root, 0);
    }
    private static Integer lvldiff(BTNode node, int level) {
        if (node == null) {
            return 0;
        }
        int Value = (Integer) node.elem;
        if (level % 2 == 0) {
            return -Value + lvldiff(node.left, level + 1)
                    + lvldiff(node.right, level + 1);
        } else {
            return Value + lvldiff(node.left, level + 1)
                    + lvldiff(node.right, level + 1);
        }
    }
}

public class Task4 {
    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) {
            return 0;
        }
        int Cvalue = (Integer) root.elem;
        int sum = 0;
        if (Cvalue >= low && Cvalue <= high) {
            sum += Cvalue;
        }
        if (Cvalue > low) {
            sum += rangeSum(root.left, low, high);
        }


        if (Cvalue < high) {
            sum += rangeSum(root.right, low, high);
        }
        return sum;
    }
}

public class Task5 {
    public static Integer mirrorSum(BSTNode root) {
        if (root == null)return 0;
        return Cmirrrorsum(root.left, root.right);
    }
    private static Integer Cmirrrorsum(BSTNode leftNode, BSTNode rightNode) {
        if (leftNode == null){
            if(rightNode == null)return 0;
        }
        Integer sum = (Integer)leftNode.elem + (Integer)rightNode.elem;
        sum = sum+Cmirrrorsum(leftNode.left, rightNode.right);
        sum = sum+Cmirrrorsum(leftNode.right, rightNode.left);  
        return sum;
    }
}

public class Task6 {
    public static Boolean isBST(BSTNode root) {
        if (root == null)return true;
        }


        return checker(root, null, null);
    }
    private static Boolean checker(BSTNode node, Integer min, Integer max) {
        if (node == null)return true;
        int Cvalue = (Integer) node.elem;
        if ((min != null && Cvalue <= min) || (max != null && Cvalue >= max)) return false;
        return checker(node.left, min, Cvalue) && 
               checker(node.right, Cvalue, max);
    }
