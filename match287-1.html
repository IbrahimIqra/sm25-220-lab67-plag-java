public static void swapChild( BTNode root, int lvl, int M ){
   if(root == null)return;


   if(lvl<M){
       BTNode temp = root.left ;
       root.left = root.right ;


       root.right = temp ;


       swapChild(root.left , lvl+1 , M);
       swapChild(root.right , lvl+1 , M);
   }




}

public static Integer subtractSummation( BTNode root ){
  if(root==null)return 0 ;


  int l = summation(root.left);
  int r = summation(root.right);


  return l-r ;
}
public static Integer summation( BTNode root ){
   if(root==null)return 0 ;


   return (Integer)root.elem + summation(root.left)+ summation(root.right);


}

public static Integer levelSum( BTNode root ){
   if(root==null)return 0 ;


   Integer s = summ(root , 0 );
   return s ;
}


public static Integer summ( BTNode root, int lvl ){
   if(root==null)return 0 ;




   int leftS = summ(root.left , lvl+1) ;
   int rightS = summ(root.right , lvl+1);


   if(lvl%2==0){
       return -(Integer)root.elem + leftS + rightS ;
   }
   else {
       return (Integer)root.elem + leftS + rightS ;
   }


}

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
  if(root==null)return 0 ;


  int sum=0 ;
  if(root.elem>=low && root.elem<=high){
      sum += root.elem;
  }
  if(low<root.elem){
      sum+=rangeSum(root.left, low , high );
  }
  if(high>root.elem){


      sum+=rangeSum(root.right , low , high);
  }
  return sum ;
}

public static Integer mirrorSum( BSTNode root ){
  if(root==null)return 0;


  int s =mirrorSumHelper(root.left , root.right);
  return s ;
}
public static Integer mirrorSumHelper(BSTNode r1, BSTNode r2 ){
  if(r1==null && r2==null)return 0 ;


  int sum=0 ;
  if(r1==null || r2==null)return sum ;//extra check


  sum+= r1.elem +r2.elem ;
  return sum+mirrorSumHelper(r1.left,r2.right)+mirrorSumHelper(r1.right,r2.left);


}
