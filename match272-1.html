  public static void swapChild( BTNode root, int lvl, int M ){
      if(root==null){
        return;
      }
      
      if(lvl<M){
        BTNode temp=root.left;
        root.left=root.right;
        root.right=temp;
      }
      
      swapChild(root.left,lvl+1,M);
      swapChild(root.right,lvl+1,M);
    }

public static Integer subtractSummation( BTNode root ){
      if(root==null){
        return 0;
      }
      
      int left=getSum(root.left);
      int right=getSum(root.right);
      
      return left-right;    
    }
    
    private static int getSum(BTNode node){
      if(node==null){
        return 0;
      }
      return (Integer)node.elem+getSum(node.left)+getSum(node.right);
    }

  public static Integer levelSum( BTNode root ){ 
        return(Integer)helper(root,0);        
    }
    private static int helper(BTNode root,int lvl){
      
      if(root==null){
        return 0;
      }
      int sum=0;
      
      if(lvl%2==0){
        sum-=(Integer)root.elem;
      }
      if(lvl%2==1){
        sum+=(Integer)root.elem;
      }
      sum+=helper(root.left,lvl+1);
      sum+=helper(root.right,lvl+1);
      
      return sum;
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
      int sum=0;
      if(root==null){
        return 0;
      }
      if(root.elem>=low && root.elem<=high){
        sum+=root.elem;
      }
      sum+=rangeSum(root.left,low,high);
      sum+=rangeSum(root.right,low,high);
      return (Integer)sum;
    }


//Task 5 (BST)

public static Integer mirrorSum( BSTNode root ){
      if(root==null){
        return 0;
      }
      int sum=helper(root.left,root.right);
      return (Integer)sum;  
    }


    private static int helper(BSTNode r1,BSTNode r2){
      
      if(r1==null || r2==null){
        return 0;
      }
      int sum=r1.elem+r2.elem;
      

      sum+=helper(r1.left,r2.right);
      sum+=helper(r1.right,r2.left);
      
      return sum;
}

  public static Boolean isBST( BSTNode root ){
  
           return helper(root, Integer.MIN_VALUE, Integer.MAX_VALUE);

         if (node == null) return true;

              if (node.elem <= min || node.elem >= max) {

    }

             return isBST(node.left, min, node.elem) && isBST(node.right, node.elem, max);
