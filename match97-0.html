   public static void swapChild( BTNode root, int lvl, int M ){


        if (root == null) {
            return;
        }

        if (lvl < M) {
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }

        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);

    }

    public static Integer subtractSummation( BTNode root ){
        if (root == null) {
            return 0;
        }

        int left = sum(root.left);
        int right = sum(root.right);

        return left - right;
    }

    private static Integer sum(BTNode root){
        if(root == null){
            return 0;
        }
        return sum(root.left) + sum(root.right) + (Integer) root.elem;
    }


//Task 6



    public static Integer levelSum( BTNode root ){
        return sum(root, 1);
    }

    private static int sum(BTNode root, int lvl) {
        if (root == null) {
            return 0;
        }

        int c = (Integer) root.elem;   

        if (lvl%2 != 0) {
            c = -c;
        }

        return c + sum(root.left, lvl+1) + sum(root.right, lvl+1);
    }

   public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null) {
            return 0;
        }

        int sum = 0;

        if(root.left.elem<=low && root.right.elem>=high){
            sum += root.elem;
        }

        if(root.elem > low){
            sum += rangeSum(root, low, high);
        }

        if(root.elem < high){
            sum += rangeSum(root, low, high);
        }
        return sum;
    }

   public static Integer mirrorSum( BSTNode root ){
        if(root == null || root.left == null || root.right == null){
            return 0;
        }
        return sum(root.left, root.right);
    }

    private static Integer sum(BSTNode n1, BSTNode n2) {
        if (n1 == null || n2 == null) {
            return 0;
        }

        int pairSum = n1.elem + n2.elem;   

        int outerSum = sum(n1.left, n2.right);

        int innerSum = sum(n1.right, n2.left);

        return pairSum + outerSum + innerSum;

    }

    public static Boolean isBST( BSTNode root ){


        return isBSTHelper(root, null, null);
    }

    private static boolean isBSTHelper(BSTNode node, Integer min, Integer max) {
        if (node == null) {
            return true;
        }

        if (min != null && node.elem <= min) {
            return false;
        }

        if (max != null && node.elem >= max) {
            return false;
        }

        return isBSTHelper(node.left, min, node.elem) && isBSTHelper(node.right, node.elem, max);
    }
