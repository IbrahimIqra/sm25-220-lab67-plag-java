public class Binarytree{
  public static class Node{
    Node left;
    Node right;
    char elem;
    public Node(char e){
      this.elem=e;
      this.left=null;
      this.right=null;
    }
  }
  public static void swap_child(Node r,int l,int M){
    if(r==null){
      return;
    }
    if(l>=M){
      Node temp=r.left;
      r.left=r.right;
      r.right=temp;
    }
    swap_child(r.left,l+1,M);
    swap_child(r.right,l+1,M);
  }
  public static void tree2(Node r,int a){
    if(r==null){
      return;
    }
    for(int i=0;i<a;i++){
      System.out.println(" ");
    }
    System.out.println(r.elem);
    tree2(r.left,a+1);
    tree2(r.right,a+1);
  }
  public static void main(String[] args){
    Node r=new Node('A');

    r.left.left=new Node('D');
    r.left.right=new Node('E');

    r.left.left.left=new Node('H');
    r.left.left.right=new Node('I');

public class Binarytree4{
  public static class Node{
    int a;
    Node left;
    Node right;
    public Node(int a){
      this.a=a;
      this.left=null;
      this.right=null;
    }
  }
  public static int sum(Node r,int b,int c){
    if(r==null){
      return 0;
    }
    if(r.a<b){
      return sum(r.right,b,c);
    }
    else if(r.a>c){
      return sum(r.left,b,c);
    }
    else{


      return r.a+sum(r.left,b,c)+sum(r.right,b,c);
    }
  }
  public static void main(String[] args){
    Node r1=new Node(6);
    r1.left=new Node(8);
    r1.right=new Node(10);
    r1.left.left=new Node(1);
    r1.left.right=new Node(4);


    r1.right.left=new Node(34);
    r1.right.right=new Node(11);
    System.out.println(sum(r1,6,12));
  }

public class Binarytree5{
  public static class Node{
    int a;
    Node left;
    Node right;
    public Node(int a){
      this.a=a;
      this.left=null;
      this.right=null;
    }
  }
  public int sum(Node r){
    if(r==null){
      return 0;
    }
    return sum1(r.left,r.right);
  }
  private int sum1(Node left,Node right){
    if(left==null&&right==null){
      return 0;
    }
    int s=left.a+right.a;
    s=s+sum1(left.left,right.right);
    s=s+sum1(left.right,right.left);
    return s;
  }
  public static void main(String[] args){


    Binarytree5 t=new Binarytree5();
     Node r1=new Node(3);
     r1.left=new Node(7);
     r1.right=new Node(4);
     r1.left.left=new Node(14);
     r1.left.right=new Node(17);


     r1.right.left=new Node(7);
     r1.right.right=new Node(23);
     System.out.println(t.sum(r1));
  }
}
