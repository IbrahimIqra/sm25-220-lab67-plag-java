public class Task4 {


    public static void swapChild(BTNode node, int currentLevel, int maxLevel) {
        if (node == null) {
            return;
        }
        if (currentLevel <= maxLevel) {
            BTNode temp = node.right;
            node.right = node.left;
            node.left = temp;
        }
        swapChild(node.right, currentLevel + 1, maxLevel);
        swapChild(node.left, currentLevel + 1, maxLevel);
    }
}


public class Task5 {
    private static int sumAll(BTNode node) {
        if (node == null) {


            return 0;
        }
        int value = (Integer) node.elem;
        int leftSide = sumAll(node.left);
        int rightSide = sumAll(node.right);
        // System.out.println("Node: " + value);
        return value + leftSide + rightSide;
    }
    public static Integer subtractSummation(BTNode root) {
        if (root == null) {
            return 0;
        }
        int leftSide = sumAll(root.left);
        int rightSide = sumAll(root.right);
        return leftSide - rightSide;
    }
}

public class Task6 {
    private static class NodeWithLevel {
        BTNode treeNode;
        int level;
        NodeWithLevel(BTNode treeNode, int level) {
            this.treeNode = treeNode;
            this.level = level;
        }
    }
    public static Integer levelSum(BTNode root) {
        if (root == null) {
            return 0;
        }
        NodeWithLevel[] queue = new NodeWithLevel[1000];
        int first = 0;
        int last = 0;
        queue[last++] = new NodeWithLevel(root, 0);


        int sumEvenLevels = 0;
        int sumOddLevels = 0;
        while (first < last) {
            NodeWithLevel current = queue[first];
            first++;
            BTNode node = current.treeNode;
            int level = current.level;
            int value = (Integer) node.elem;
            if (level % 2 == 0) {
                sumEvenLevels += value;
            } else {
                sumOddLevels += value;
            }
            if (node.left != null) {
                queue[last++] = new NodeWithLevel(node.left, level + 1);
            }
            if (node.right != null) {
                queue[last++] = new NodeWithLevel(node.right, level + 1);
            }
        }
        // System.out.println("Even: " + sumEvenLevels + ", Odd: " + sumOddLevels);
        return sumEvenLevels - sumOddLevels;
    }
}

public class Task4 {
    public static Integer rangeSum(BSTNode node, Integer lowValue, Integer highValue) {
        if (node == null) {
            return 0;
        }
        int nodeValue = node.elem;
        if (nodeValue < lowValue) {
            return rangeSum(node.right, lowValue, highValue);
        }
        if (nodeValue > highValue) {
            return rangeSum(node.left, lowValue, highValue);
        }
        int sumLeft = rangeSum(node.left, lowValue, highValue);
        int sumRight = rangeSum(node.right, lowValue, highValue);
        // System.out.println("Adding: " + nodeValue);
        return nodeValue + sumLeft + sumRight;
    }
}



public class Task5 {
    private static int sumMirrors(BSTNode leftChild, BSTNode rightChild) {
        if (leftChild == null || rightChild == null) {
            return 0;
        }
        int totalSum = leftChild.elem + rightChild.elem;
        int sum1 = sumMirrors(leftChild.right, rightChild.left);
        int sum2 = sumMirrors(leftChild.left, rightChild.right);
        // System.out.println("Pair: " + leftChild.elem + ", " + rightChild.elem);
        return totalSum + sum1 + sum2;
    }
    public static Integer mirrorSum(BSTNode root) {
        if (root == null) {
            return 0;
        }
        return sumMirrors(root.left, root.right);
    }
}

public class Task6 {


    private static boolean validate(BSTNode currentNode, int lowerBound, int upperBound) {
        if (currentNode == null) {
            return true;
        }
        int nodeValue = currentNode.elem;
        if (nodeValue <= lowerBound || nodeValue >= upperBound) {
            return false;
        }
        // System.out.println("Checking: " + nodeValue);
        boolean isLeftValid = validate(currentNode.left, lowerBound, nodeValue);
        boolean isRightValid = validate(currentNode.right, nodeValue, upperBound);
        return isLeftValid && isRightValid;
    }
    public static Boolean isBST(BSTNode root) {
        return validate(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }
}
