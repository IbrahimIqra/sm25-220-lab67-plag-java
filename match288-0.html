public static void swapChild(BTNode root,int lvl,int M){
        
        if (root==null){
            return;}
        if (lvl<=M){
            BTNode temp=root.left;
            root.left=root.right;
            root.right=temp;}        
        swapChild(root.left,lvl+1,M);
        swapChild(root.right,lvl+1,M);}}

    public static Integer subtractSummation(BTNode root){
        if(root==null){
            return 0;}
        int leftSum=calSum(root.left);
        int rightSum=calSum(root.right);
        return leftSum-rightSum;}
    public static int calSum(BTNode node){
        if(node==null){
            return 0;}
        int elemVal=0;
        if(node.elem instanceof Integer){
            elemVal=(Integer)node.elem;}
        return elemVal+calSum(node.left)+calSum(node.right);}}

    public static int levelSum(BTNode root){
        return CalDiff(root,0);}
    public static Integer CalDiff(BTNode node,int level){
        if(node==null){
            return 0;}
        int CurVal;
        if(node.elem instanceof Integer){
            CurVal=(Integer)node.elem;}
        else{
            CurVal=0;}
        if(level%2==0){
            CurVal=-CurVal;}
        int leftDiff=CalDiff(node.left,level+1);
        int rightDiff=CalDiff(node.right,level+1);
        return CurVal+leftDiff+rightDiff;}}

    public static Integer rangeSum(BSTNode root,Integer low,Integer high){
      if(root==null){
            return 0;}
       int CurSum=0;  
        if(root.elem>=low && root.elem<=high){


            CurSum=root.elem;}
        if(root.elem>low){
            CurSum+=rangeSum(root.left,low,high);}       
        if(root.elem<high){
            CurSum+=rangeSum(root.right,low,high);}
        return CurSum;}}

    public static Integer mirrorSum(BSTNode root){
        if(root==null) 
          return 0;
        return mirrorHelp(root.left,root.right);}
    public static int mirrorHelp(BSTNode left,BSTNode right){
        if(left==null || right==null)
          return 0;
        int sum=left.elem+right.elem;
        sum+=mirrorHelp(left.left,right.right);
        sum+=mirrorHelp(left.right,right.left);
        return sum;}}
