public static void swapChild(BTNode node, int currentLevel, int maxLevel) {
    if (node == null) {
        return;
    }

    if (currentLevel < maxLevel) {
        BTNode tempNode = node.left;
        node.left = node.right;


        node.right = tempNode;
    }

    swapChild(node.left, currentLevel + 1, maxLevel);
    swapChild(node.right, currentLevel + 1, maxLevel);
}

public class Task5 {

        public static Integer subtractSummation(BTNode root) {
        
        if (root == null) {
            return 0;
        }

        
        int leftSum = calculateSubtreeSum(root.left);
        int rightSum = calculateSubtreeSum(root.right);

        return leftSum - rightSum;
    }

    private static int calculateSubtreeSum(BTNode node) {
        
        if (node == null) {
            return 0;
        }

        return (Integer) node.elem
               + calculateSubtreeSum(node.left)


               + calculateSubtreeSum(node.right);
    }
}

public class Task6 {

    public static Integer levelSum(BTNode root) {
         return calculateAlternatingSum(root, 1);
    }

    private static int calculateAlternatingSum(BTNode node, int sign) {
        
        if (node == null) {
            return 0;
        }

        return (sign * (Integer) node.elem)
               + calculateAlternatingSum(node.left, -sign)
               + calculateAlternatingSum(node.right, -sign);
    }
}

public class Task4 {



    public static Integer rangeSum( BSTNode root, int low, int high ){
                
        if (root == null) {
            return 0;
        }

        int currentSum = 0;

        if (root.elem > low) {
            currentSum += rangeSum(root.left, low, high);
        }

        if (root.elem >= low && root.elem <= high) {
            currentSum += root.elem;
        }

        if (root.elem < high) {
            currentSum += rangeSum(root.right, low, high);
        }

        return currentSum;
    }
}


//Lab 7 Task 5
public class Task5 {

    public static Integer mirrorSum(BSTNode root) {
        
        if (root == null) {
            return 0;
        }
        return calculateSymmetricSum(root.left, root.right);
    }

    private static Integer calculateSymmetricSum(BSTNode node1, BSTNode node2) {
        if (node1 == null || node2 == null) {
            return 0;
        }

        return node1.elem + node2.elem
               + calculateSymmetricSum(node1.left, node2.right)
               + calculateSymmetricSum(node1.right, node2.left);
    }
}

public class Task6 {

    public static Boolean isBST(BSTNode root) {
        return isTreeBranchValid(root, null, null);
    }

    private static boolean isTreeBranchValid(BSTNode currentNode, int lowerLimit, int upperLimit) {


        if (currentNode == null) {
            return true;
        }

        if (lowerLimit != null && currentNode.elem <= lowerLimit) {
            return false;
        }

        if (upperLimit != null && currentNode.elem >= upperLimit) {
            return false;
        }

        boolean isLeftPathCorrect = isTreeBranchValid(currentNode.left, lowerLimit, currentNode.elem);
        boolean isRightPathCorrect = isTreeBranchValid(currentNode.right, currentNode.elem, upperLimit);

        return isLeftPathCorrect && isRightPathCorrect;
    }
}
