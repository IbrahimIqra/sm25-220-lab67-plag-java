    public static void swapChild(BTNode root,int lvl,int M){
        swapChildHelper(root,lvl,M);
    }
    private static BTNode swapChildHelper( BTNode root, int lvl, int M ){
        if(root==null){
            return null;
        }
        if(lvl>M){
            return root;
        }


        if(lvl==M){
            root.left=swapChildHelper(root.left,lvl+1,M);
            root.right=swapChildHelper(root.right,lvl+1,M);
            return root;
        }
        else {
            BTNode temp = root.left;
            root.left = swapChildHelper(root.right, lvl + 1, M);
            root.right = swapChildHelper(temp, lvl + 1, M);
            return root;
        }
    }

   private static int sum(BTNode root){
        if(root==null){
            return 0;
        }
        int left=sum(root.left);
        int right=sum(root.right);
        return (int)root.elem+left+right;
   }

public static Integer levelSum(BTNode root) {

        return helper(root,0);
    }
    private static int helper(BTNode root, int lvl) {
        if (root == null) {
            return 0;
        }
        if (lvl % 2 == 0) {
            return (-1*(int) root.elem) + helper(root.left, lvl + 1) + helper(root.right, lvl + 1);
        } else {
            return (int) root.elem+ helper(root.left,lvl + 1) + helper(root.right,lvl + 1);
        }
    }

public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) {
            return 0;
        }
        int left = rangeSum(root.left, low, high);
        int right = rangeSum(root.right, low, high);
        

        if (root.elem >= low && root.elem <= high) {
            root.elem+left+right;
        }
        return left+right;
    }

public static Integer mirrorSum(BSTNode root) {
        if (root == null) {
            return 0;
        }
        return mirrorSumHelper(root.left,root.right);
    }
    private static int mirrorSumHelper(BSTNode left, BSTNode right){
        if (left == null || right == null) {
            return 0;
        }


        int sum = left.elem + right.elem;
        sum += mirrorSumHelper(left.left, right.right);
        sum += mirrorSumHelper(left.right, right.left);
        return sum;
    }

public static Boolean isBST( BSTNode root ){
        if(root==null){
            return true;
        }


        if(root.left==null && root.right==null){
            return true;
        }
       int leftmax=leftMax(root.left);
       int rightmin=rightMin(root.right);

       if(leftmax<=root.elem && rightmin>=root.elem){
           return true;
       }
       else{
           return false;
       }
    }
    private static int leftMax(BSTNode root){
        if(root==null){
            return -999999;
        }


        int left=leftMax(root.left);
        int right=leftMax(root.right);
        return Math.max(root.elem,Math.max(left,right));
    }
    private static int rightMin(BSTNode root) {
        if (root == null) {
            return 999999;
        }


        int left = rightMin(root.left);
        int right = rightMin(root.right);
        return Math.min(root.elem, Math.min(left, right));
    }
