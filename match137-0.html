public static void swapChild( BTNode root, int lvl, int M ) {


        if (root == null) return;


        if (lvl < M) {
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }


        swapChild(root.left,  lvl+1, M);
        swapChild(root.right, lvl+1, M);
}

public static Integer subtractSummation( BTNode root ) {


        if (root == null) return 0;


        int leftSum  = sumOnlySubT(root.left);
        int rightSum = sumOnlySubT (root.right);


        return leftSum - rightSum;
 }


 private static int sumOnlySubT(BTNode node) {


        if (node == null) return 0;


        int sum = (Integer) node.elem;


        return (sum + sumOnlySubT(node.left) + sumOnlySubT(node.right));
 }

public static Integer levelSum( BTNode root ) {
        return dfsLevel(root, 0); 
    }


    private static int dfsLevel(BTNode node, int lvl) {
        if (node == null) return 0;


        int val = (Integer) node.elem;
        int left  = dfsLevel(node.left,  lvl+1);
        int right = dfsLevel(node.right, lvl+1);


        if (lvl % 2 == 1) { 
            return val+left+right;
        } 
        else {
            return - val+left+right;
        }
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ) {


        if (root==null || low==null || high==null) return 0;




        int sum = root.elem;


        if (sum < low) {
            return rangeSum(root.right, low, high);
        } else if (sum > high) {
            return rangeSum(root.left, low, high);
        } else {
            return sum + rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
        }
    }

public static Integer mirrorSum( BSTNode root ) {


        if (root == null) return 0;


        return pairSum(root.left, root.right);
    }


    private static int pairSum(BSTNode nodeA, BSTNode nodeB) {


        if (nodeA == null || nodeB == null) return 0;


        int sum = nodeA.elem + nodeB.elem;


        sum += pairSum(nodeA.left, nodeB.right);
        sum += pairSum(nodeA.right, nodeB.left);
        return sum;
    }

public static Boolean isBST( BSTNode root ) {


        return isBSTCheck(root, null, null);


 }


 private static boolean isBSTCheck(BSTNode node, Integer min, Integer max) {


        if (node == null) return true;


        int val = node.elem;


        if ((min!=null && val<=min) || (max!=null && val>=max)) return false;


        return isBSTCheck(node.left, min, val) && isBSTCheck(node.right, val, max);
