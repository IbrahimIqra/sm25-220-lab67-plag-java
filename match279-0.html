public class Task4 {
    public static void swapChild( BTNode root, int lvl, int M ){
        if (root == null )return;   
        if(lvl < M){   
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
            
            swapChild(root.left, lvl+1, M);
            swapChild(root.right, lvl+1, M);
        }
    }
    
}








public class Task5 {
    
     public static Integer subtractSummation( BTNode root ){
        if(root == null)return 0;
        
        Integer  sub = 0;
        Integer L = sum (root.left);
        Integer R = sum (root.right);
        sub = L-R;
        
        return sub;
    }
    private static Integer sum(BTNode node){
        if(node== null)return 0;
        
        Integer sum = 0;


        sum = (Integer)node.elem+sum(node.left)+sum(node.right);
        return sum;
    }
}

            sum -= (Integer)root.elem;
            
        }
     
        else{
            sum += (Integer)root.elem;


        }


        sum+=levelSum (root.left, lvl+1);


        sum+=levelSum (root.right, lvl+1);
         
   
        
        
        return sum;
    }

public class Task4 {


    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        
        if(root == null) return 0;
        int sum = 0;


        if(root.elem >= low && root.elem <= high) {
            sum+=root.elem;
        }
        sum += rangeSum(root.left,low,high);
        sum += rangeSum(root.right,low,high);
        
        return sum;
    }
}
