 public static void swapChild( BTNode root, int lvl, int M ){
       if(root==null){
        return;
       }
       if(lvl<M){
        BTNode temp=root.left;
        root.left=root.right;
        root.right=temp;
       }
       swapChild( root.left, lvl+1, M );
       swapChild( root.right, lvl+1, M );
    }

 public static Integer subtractSummation( BTNode root ){            
    return helperTree(root.left,0) - helperTree(root.right,0);
  }
 private static int helperTree(BTNode root,int sum){
    if(root==null){
        return 0;
    }
    int val=(int)root.elem;          
    sum=val+ helperTree(root.left,sum) + helperTree(root.right,sum);
    return sum;
 }

 private static int helper(BTNode root,int lvl,int sum){
    if(root==null){
       return sum;
    }
    int val=(int)root.elem;
    if(lvl%2==0){
        int evenSum=-val + helper(root.left,lvl+1,sum)+ helper(root.right,lvl+1,sum);
        return evenSum;
    }else{
        int oddSum=val + helper(root.left,lvl+1,sum)+ helper(root.right,lvl+1,sum);


        return oddSum;
    }
}

 public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
       if(root==null){
           return 0;
        }
    int sum=0;
    if(root.elem>=low && root.elem<=high){
     sum+=root.elem;
    }
    if(root.elem<high){
      sum+=rangeSum(root.right,low,high);
    }
    if(root.elem>low){


      sum+=rangeSum(root.left,low,high);
    }
   return sum;
}

 private static int helper(BSTNode left, BSTNode right){
    if(left==null || right==null){
        return 0;
     }
    int sum=left.elem+right.elem;
    sum+=helper(left.left,right.right);
    sum+=helper(left.right,right.left);

    return sum;
}
