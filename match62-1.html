public class Task4 {


    public static void swapChild( BTNode root, int lvl, int M ){
        if(root==null){
            return;
        }
        if(lvl<M){
            BTNode temp = root.left;
            root.left=root.right;
            root.right=temp;
        }
        swapChild(root.left,lvl+1,M);
        swapChild(root.right,lvl+1,M);
    }
}

        int right_sum = helper_subtractSummation(root.right);

    }

        if(sent_root==null){
            return 0;

        int value= (int)sent_root.elem;

        return sum;
    }
}

public class Task6 {
    public static Integer levelSum( BTNode root ){
        int even = helper_even(root,1);
        int odd = helper_odd(root,1);
        return even-odd;
    }


    private static Integer helper_even(BTNode sent_root, int level){
        if(sent_root==null) return 0;
        int even_sum=0;
        if(level%2==0){
            even_sum+=(int)sent_root.elem;
        }
        even_sum+=helper_even(sent_root.left,level+1);
        even_sum+=helper_even(sent_root.right,level+1);
        return even_sum;
    }

    private static Integer helper_odd(BTNode sent_root, int level){
        if(sent_root==null) return 0;
        int odd_sum=0;
        if(level%2!=0){


            odd_sum+=(int)sent_root.elem;
        }
        odd_sum+=helper_odd(sent_root.left,level+1);
        odd_sum+=helper_odd(sent_root.right,level+1);
        return odd_sum;
    }
}

public class Task4 {
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null){
            return null;
        }
        return rangeSum_helper(root,low,high);
    }


    private static Integer rangeSum_helper(BSTNode s_root, Integer s_low, Integer s_high){
        if(s_root==null){
            return 0;
        }
        int sum=0;
        if(s_root.elem>=s_low&&s_root.elem<=s_high){
            sum+=s_root.elem;
        }
        sum+=rangeSum_helper(s_root.left,s_low,s_high);
        sum+=rangeSum_helper(s_root.right,s_low,s_high);
        return sum;
    }
}

public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
        if(root==null){
            return null;
        }


        return mirrorSum_helper(root.left,root.right);
    }
    private static Integer mirrorSum_helper(BSTNode root_l, BSTNode root_r){
        if(root_l==null||root_r==null){
            return 0;
        }


        int sum = root_l.elem+root_r.elem;

        sum+=mirrorSum_helper(root_l.left,root_r.right);
        sum+=mirrorSum_helper(root_l.right,root_r.left);
        return sum;
    }

}

public class Task6 {
    public static Boolean isBST( BSTNode root ){
        if(root==null){
            return true;
        }

        return isBST_helper(root);
    }
    private static Boolean isBST_helper(BSTNode s_root) {
        if (s_root == null) {
            return true;
        }
        boolean check = true;

        if((s_root.left!=null&&s_root.left.elem>=s_root.elem)||(s_root.right!=null&&s_root.right.elem<=s_root.elem)){
            check=false;
        }
        
        boolean check_left = isBST_helper(s_root.left);
        boolean check_right = isBST_helper(s_root.right);

        if(!check_left||!check_right){
            check=false;
        }

        return check;
    }

}
