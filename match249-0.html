public static void swapChild(BTNode root, int lvl, int M) {
        if (root == null || lvl >= M) return;

        BTNode temp = root.left;
        root.left = root.right;
        root.right = temp;

        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }

public static Integer subtractSummation(BTNode root) {
        if (root == null) return 0;

        int leftSum = sumMeth(root.left);
        int rightSum = sumMeth(root.right);

        return leftSum - rightSum;
    }

    private static int sumMeth(BTNode root) {
        if (root == null) return 0;

        int sum = (int) root.elem;
        sum += sumMeth(root.left); 
        sum += sumMeth(root.right);

        return sum;
    }

public static Integer levelSum(BTNode root) {
        if (root == null) return 0;

        return helpies(root, 0);
    }

    private static int helpies(BTNode root, int level) {
        if (root == null) return 0;

        int total = (int) root.elem;

        if (level % 2 == 0) {
            total = -total;
        }
        total = total + helpies(root.left, level + 1);
        total = total + helpies(root.right, level + 1);

        return total;
    }

public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
    if (root == null) return 0;

    int sum = 0;
    
    if ((int)root.elem >= low && (int)root.elem <= high) {


        sum += (int)root.elem;
    }

    sum += rangeSum(root.left, low, high);
    sum += rangeSum(root.right, low, high);

    return sum;

public static Integer mirrorSum(BSTNode root) {
        if (root == null) return 0;

        return helpies(root.left, root.right);
    }

    private static int helpies(BSTNode lala, BSTNode rara) {
        if (lala == null || rara == null) return 0;

        int sum = lala.elem + rara.elem;

        sum += helpies(lala.left, rara.right);
        sum += helpies(lala.right, rara.left);

        return sum;
    }

public static Boolean isBST(BSTNode root) {
        return helpsie(root, null, null);
    }



    private static boolean helpsie(BSTNode root, BSTNode min, BSTNode max) {
        if (root == null) return true;

        if (min != null && root.elem <= min.elem) return false;

        if (max != null && root.elem >= max.elem) return false;

        return helpsie(root.left, min, root) && helpsie(root.right, root, max);
    }
