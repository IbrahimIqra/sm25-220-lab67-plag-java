    public static void swapChild( BTNode root, int lvl, int M ){
        if (root == null) {
            return;
        }
        


        if (lvl + 1 <= M) {
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        
        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }

    public static Integer subtractSummation( BTNode root ){
        if (root == null) {
            return 0;
        }
        
        int sumL = subtreeSum(root.left);
        int sumR = subtreeSum(root.right);
        
        return sumL - sumR;
    }

    private static Integer subtreeSum(BTNode node) {
        if (node == null) {
            return 0;
        }
        
        Integer value = (Integer) node.elem;
        
        return value + subtreeSum(node.left) + subtreeSum(node.right);

    }

    public static Integer levelSum(BTNode root) {
        if (root == null) {
            return 0;
        }

        int[] sums = new int[2];
        levelSumHelper(root, 0, sums);
        return sums[1] - sums[0];
    }
    
    private static void levelSumHelper(BTNode node, int level, int[] sums) {
        if (node == null) {
            return;
        }

        int value = (Integer) node.elem;

        if (level % 2 == 0) {
            sums[0] += value;
        } 
        else {
            sums[1] += value;
        }

        levelSumHelper(node.left, level + 1, sums);
        levelSumHelper(node.right, level + 1, sums);
    }

    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) {
            return 0;
        }
        
        if (root.elem >= low && root.elem <= high) {
            return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
        } 
        else if (root.elem < low) {
            return rangeSum(root.right, low, high);
        } 
        else {
            return rangeSum(root.left, low, high);
        }
    }

Task#05



    public static Integer mirrorSum(BSTNode root) {
        if (root == null) {
            return 0;
        }
        return mirrorSumHelper(root.left, root.right);
    }
    
    private static Integer mirrorSumHelper(BSTNode left, BSTNode right) {
        if (left == null || right == null) {
            return 0;
        }
        else {
            int sum = left.elem + right.elem;
        
            sum += mirrorSumHelper(left.left, right.right);
            sum += mirrorSumHelper(left.right, right.left);
        
            return sum;
        }
    }

    public static Boolean isBST(BSTNode root) {
        return isBSTHelper(root, null, null);
    }

    private static Boolean isBSTHelper(BSTNode node, Integer min, Integer max) {
        if (node == null) {
            return true;
        }
        
        if ((min != null && node.elem <= min) || (max != null && node.elem >= max)) {
          return false;
        }
    
        return isBSTHelper(node.left, min, node.elem) && isBSTHelper(node.right, node.elem, max);
    }
