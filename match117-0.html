public static void swapChild( BTNode root, int lvl, int M ){
        if(root == null){
            return;
        }
        if(lvl<M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left,lvl+1,M);
        swapChild(root.right,lvl+1,M);
    }


//Task 5

  public static Integer subtractSummation( BTNode root ){
        int left_SubTree = sum(root.left);
        int right_SubTree = sum(root.right);
        return left_SubTree - right_SubTree;
    }
    private static int sum(BTNode node){
        if(node==null){
            return 0;
        }
        else{


            return (Integer)node.elem+sum(node.left)+sum(node.right);
        }
    }


//Task 6

 public static Integer levelSum( BTNode root ){
        return helper(root,0);
    }
    private static int helper(BTNode node, int level){
        if(node==null){
            return 0;
        }
        int value = (Integer)node.elem;
        if(level%2==0){
            value = -value;
        }
        return value + helper(node.left,level+1) + helper(node.right,level+1);

    }

 public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null){
            return 0;
        }
        int value = (Integer)root.elem;
        int current = 0;
        if(value>=low&&value<=high){
            current = value;
        }


        return current+rangeSum(root.left,low,high)+rangeSum(root.right,low,high);
    }


//Task 5

 public static Integer mirrorSum( BSTNode root ){
        return helper (root.left,root.right);
    }
        private static Integer helper(BSTNode r1, BSTNode r2){
            if(r1==null||r2==null){
                return 0;
            }


            return r1.elem+r2.elem+helper(r1.left,r2.right)+helper(r1.right,r2.left);
        }


//Task 6

public static Boolean isBST( BSTNode root ){
        return helper(root,null,null);
    }
        private static boolean helper(BSTNode n, Integer min, Integer max){
            if(n==null){
                return true;
            }
            if((min!=null&&n.elem<=min)||(max!=null&&n.elem>=max)){


                return false;
            }
            return helper(n.left,min,n.elem)&&helper(n.right,n.elem,max);
        }
