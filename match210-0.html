 public static void swapChild( BTNode root, int lvl, int M ){
      if(root == null){
        return;
        }
        if(lvl < M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }

 private static int sum(BTNode root){
        if(root == null){
            return 0;
        }
        return (Integer)root.elem + sum(root.left) + sum(root.right);
    }
    public static Integer subtractSummation( BTNode root ){
        if(root == null){
            return 0;
        }
        int leftSum = sum(root.left);
        int rightSum = sum(root.right);
        return leftSum - rightSum;
    }

  private static int sum(BTNode root, int lvl){

            return 0;
        }

        if (lvl % 2 == 1){
            return -value + sum(root.left, lvl+1) + sum(root.right, lvl+1);

            return value + sum(root.left, lvl+1) + sum(root.right, lvl+1);
        }
    }

 public static Integer rangeSum(BSTNode root, Integer low, Integer high){
         if(root == null){
            return 0;
        }
        int sum = 0;
        if(root.elem >= low && root.elem <= high){
            sum += root.elem;
        }
        if(root.elem > low){
            sum += rangeSum(root.left, low, high);
        }
        if(root.elem < high){
            sum += rangeSum(root.right, low, high);
        }
        return sum;
    }



//TASK_05



private static int sum(BSTNode left, BSTNode right){
        if(left == null || right == null){
            return 0;
        }
        int sum = left.elem + right.elem;
        sum += sum(left.left, right.right);
        sum += sum(left.right, right.left);
        return sum;
    }
    public static Integer mirrorSum(BSTNode root){
        if(root == null){
            return 0; 
        }


        return sum(root.left, root.right);
    }

 private static boolean helper(BSTNode root, Integer min, Integer max){
    if(root == null){
        return true;
    }
    if(min != null && root.elem <= min){
        return false;
    }
    if(max != null && root.elem >= max){
        return false;
    }
    return helper(root.left, min, root.elem) && helper(root.right, root.elem, max);
}
public static Boolean isBST(BSTNode root){
    return helper(root, null, null);
}
