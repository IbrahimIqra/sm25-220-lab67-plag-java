public class Task4 {
    public static void swapChild( BTNode root, int lvl, int M ){
        if (root== null){
            return;
        }
        if (lvl>= M){
            return;

        }   
        BTNode temp= root.left;
        root.left= root.right;


        root.right=temp;
        swapChild( root.left, lvl+1, M );
        swapChild(root.right, lvl+1, M);

   }
}

public class Task5 { 
    public static Integer subtractSummation( BTNode root ){
        if (root == null) return 0;
        int leftSum = sum(root.left);
        int rightSum = sum(root.right);
       
        return (leftSum - rightSum);
    }
    private static int sum(BTNode node){
        if (node == null){
             return  0;
        }
        int value = (Integer) node.elem;


        return (int)value + sum(node.left) + sum(node.right);
    }
}

public class Task6 {  
    public static Integer levelSum( BTNode root ){     
        return healper(root, 1);
    }
    private static int healper(BTNode node, int level){
        if (node == null){
             return 0;
        }
        int val=(int)node.elem;
        if (level%2==0) {
            return val+ healper(node.left, level+1)+ healper(node.right, level+1);
        }
        else{
            return -val+ healper(node.left, level+1) + healper(node.right, level+1);
        }    
}
}

public class Task4 {
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null){
            return 0;
        }
        int val=(int)root.elem;
        if(val>=low && val<= high){
            return val + rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
        }
        if(val<low){
            return rangeSum(root.right, low, high);
        }
        else{
            return rangeSum(root.left, low, high);
        }     
    }
}

public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
          if (root == null) return 0;
        return mirrorSumHelper(root.left, root.right);
    }

    private static Integer mirrorSumHelper(BSTNode left_node, BSTNode right_node) {
        if (left_node == null || right_node == null) return 0;


        int total = (int)left_node.elem +(int) right_node.elem;
        total += mirrorSumHelper(left_node.left, right_node.right);
        total += mirrorSumHelper(left_node.right, right_node.left);
        return total;
    }
}

public class Task6 {   
    public static Boolean isBST( BSTNode root ){
        if (root == null) {
            return true;
        }
        if (root.left!=null && (int) root.left.elem>(int) root.elem) {
            return false;
        }
        if (root.right!= null&&(int) root.right.elem < (int) root.elem) {
            return false;
        }
        return isBST(root.left) && isBST(root.right);
    }
