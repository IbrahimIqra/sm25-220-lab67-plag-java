public class Task5 {
    public static Integer subtractSummation( BTNode root ){


        int l_sum=leftadd(root.left);  
        int r_sum=rightadd(root.right); 
        return l_sum-r_sum;      
    }
    private static int leftadd(BTNode n){
        if (n==null){
            return 0; 
        }
        int l=(Integer)n.elem+leftadd(n.left)+leftadd(n.right);
        return l;
    }
    private static int rightadd(BTNode n){
        if(n==null){
            return 0; 
        }
        int r=(Integer)n.elem+rightadd(n.left)+rightadd(n.right);


        return r;
    }

}

public class Task6 {

    public static Integer levelSum( BTNode root ){
      
        return sum_help(root,0);

    }
    private static Integer sum_help(BTNode n,int lvl){
        if(n==null){
            return 0;
        }
        int x;
        if(lvl==0){
            x=-(Integer)n.elem;
        }
        if(lvl%2!=0){
            x=(Integer)n.elem;
        }
        else{
            x=-(Integer)n.elem;
        }
        int l_sum=sum_help(n.left,lvl+1);
        int r_sum=sum_help(n.right,lvl+1);
        return x+l_sum+r_sum;
        }

}

public class Task4 {


    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null){
            return 0;
        }
        int sum=0;
        if(root.elem>=low && root.elem<=high){
            sum+=root.elem;
        }
        if(root.elem>low){
            sum+=rangeSum(root.left, low, high);
        }
        if(root.elem<high){


            sum+=rangeSum(root.right, low, high);
        }
        return sum; 
    }
}

//Task5
public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
        if(root==null || root.left==null || root.right==null){
            return 0;
        }
        return summation(root.left,root.right);
    }
    private static Integer summation(BSTNode n1,BSTNode n2){


        if(n1==null || n2==null){
            return 0;
        }
        int s=n1.elem+n2.elem;
        s+=summation(n1.left, n2.right);
        s+=summation(n1.right, n2.left);
        return s;
    }
        
}

public class Task6 {
    public static Boolean isBST( BSTNode root ){
        return chBoolean(root, Integer.MAX_VALUE,Integer.MIN_VALUE);
    }
    private static Boolean chBoolean(BSTNode r,int max ,int min){
        if(r==null){
            return true;
        }
        if(r.elem>=max && r.elem<=min){
            return false;
        }
        return chBoolean (r.right,max,r.elem) && chBoolean(r.left,min,r.elem);
       
    }

}
