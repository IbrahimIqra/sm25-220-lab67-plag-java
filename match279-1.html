public class Task4 {

    public static void swapChild( BTNode root, int lvl, int M ){
        if(root==null){
        return;
        }
        if(lvl<M){
        BTNode hold = root.left;
        root.left=root.right;
        root.right=hold;
        }
        swapChild(root.left,lvl+1,M);
        swapChild(root.right,lvl+1,M);
    }
    
}

public class Task6 {

    public static Integer levelSum( BTNode root ){
        return levelSum(root ,0);
    }
    public static Integer levelSum( BTNode root,int lvl ){
        if(root==null){
        return 0;
        }
        Integer current=sumhelper(root,lvl);
        Integer lefts=levelSum(root.left,lvl+1 );
        Integer rights=levelSum(root.right,lvl+1 );
        current=current+lefts+rights;
        return current;
       }
    public static Integer sumhelper( BTNode root,int lvl ){
        Integer sum= (Integer)root.elem;
        if(lvl%2==0){
        sum=-sum;
        return sum;
        }
        else{
        return sum;
        }
    }

}

public class Task4 {
    
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null){
            return 0;
        }
        Integer sum=0;


        if(root.elem>=low && root.elem<=high){
            sum+=root.elem;
        }
        sum+=rangeSum(root.left,low,high);
        sum+=rangeSum(root.right,low,high);
        return sum;
    }
}



public class Task5 {

    public static Integer mirrorSum( BSTNode root ){
        return mirrorSum(root.left,root.right);
        }


     public static Integer mirrorSum( BSTNode r1, BSTNode r2 ){
         if(r1==null || r2==null){
         return 0;
         }
         int sum = r1.elem+r2.elem;
         sum+=mirrorSum( r1.left,r2.right );
         sum+=mirrorSum( r1.right,r2.left );
         return sum;
        }
    

}

    public static Boolean isBST( BSTNode root ){
        int min= Integer.MIN_VALUE;
        int max=Integer.MAX_VALUE;
        return isBST(root,min,max);
        
    }
    public static Boolean isBST( BSTNode root , int min1, int max2 ){
        if(root==null){
        return true;
        }
        if(root.elem>min1 && root.elem<max2){
        return isBST(root.left , min1, root.elem ) && isBST(root.right, root.elem, max2) ;
        
        }
        else{
        return false;
        }
        
        
    }
