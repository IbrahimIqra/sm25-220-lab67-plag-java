public static void swapChild( BTNode root, int lvl, int M ){
        if(root==null){return;}
        if(lvl<M){
            BTNode temp=root.left;
            root.left=root.right;
            root.right=temp;
        }
        swapChild(root.left,lvl+1,M);
        swapChild(root.right,lvl+1,M);
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root == null) return 0;
        if(root.elem < low) return rangeSum(root.right, low, high); // choto hoile right e jabo
        if(root.elem > high) return rangeSum(root.left, low, high); // boro hole left e jabo


        return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high); // range e thakle sum
    }

public static Integer mirrorSum( BSTNode root ){
        if(root==null) return 0 ;
        return sum(root.left ,root.right);
    }
    public static Integer sum(BSTNode l , BSTNode r){
        if(l==null || r==null) return 0;
        return l.elem + r.elem + sum(l.left , r.right) + sum(l.right , r.left);
    }

public static Boolean isBST( BSTNode root ){
        if(root==null) return true;

        if(root.left!=null && root.left.elem>root.elem) return false;  // left boro hoilei false
        if(root.right!=null && root.right.elem<root.elem) return false;  // right choto holei false
        return isBST(root.left) && isBST(root.right);
    }
