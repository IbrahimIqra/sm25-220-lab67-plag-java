public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null) {return 0;}

        int sum=0;

        if(root.elem>=low && root.elem<=high){sum+=root.elem;
        }
        if(root.elem > low){
            sum+= rangeSum(root.left,low,high);
        }
        if(root.elem<high){
            sum+=rangeSum(root.right,low,high);
        }
        return sum ;
    }

public static Boolean isBST( BSTNode root ){
                return checkBST(root, null, null); }
    private static boolean checkBST(BSTNode node, Integer min, Integer max) {
        if (node == null) return true;

        
        if (min != null && node.elem <= min) return false;


        if (max != null && node.elem >= max) return false;

        
        return checkBST(node.left, min, node.elem) &&
               checkBST(node.right, node.elem, max);
    }

public static Integer subtractSummation( BTNode root ){
        if (root == null) return 0;

        int leftSum = getSum(root.left);
        int rightSum = getSum(root.right);

        return leftSum - rightSum;
    }

    private static int getSum(BTNode node) {
        if (node == null) return 0;

        return (Integer) node.elem + getSum(node.left) + getSum(node.right);
    }

public static Integer levelSum( BTNode root ){
        return findSum(root, 1);
    }

    private static int findSum(BTNode root, int level) {
        if (root == null) return 0;

        int value = (Integer) root.elem;
        int left = findSum(root.left, level + 1);
        int right = findSum(root.right, level + 1);

        if (level % 2 == 1) { 
            return value + left + right; 
        } else { 
            return -value + left + right; 
        }
    }
