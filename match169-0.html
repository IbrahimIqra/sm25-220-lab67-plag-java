class Node {
    int elem;
    Node left, right;
    Node(int e) { elem = e; }
}

public class SubtractSummation {
   


    static int sum(Node n) {
        if (n == null) return 0;
        return n.elem + sum(n.left) + sum(n.right);
    }

   
    static int subtract_summation(Node root) {
        if (root == null) return 0;
        int leftSum  = root.elem + sum(root.left); 
        int rightSum = sum(root.right);
        return leftSum - rightSum;
    }
}

class Node {
    int data;
    Node left, right;
    Node(int d) { data = d; }
}

public class DifferenceOfLevelSum {
    static int getDifference(Node root) {
        return helper(root, 0);
    }
    static int helper(Node node, int level) {
        if (node == null) return 0;
        if (level % 2 != 0) {
            return node.data + helper(node.left, level + 1) + helper(node.right, level + 1);
        } else {
            return -node.data + helper(node.left, level + 1) + helper(node.right, level + 1);
        }
    }



    public static void main(String[] args) {
        Node root = new Node(1);
        root.left = new Node(2);
        root.right = new Node(3);
        root.left.left = new Node(4);
        root.right.left = new Node(5);


        root.right.right = new Node(6);
        root.right.left.left = new Node(7);
        root.right.left.right = new Node(8);
        System.out.println(getDifference(root));
    }
}

class Node {
    int val;
    Node left, right;

    Node(int val) {
        this.val = val;
        left = right = null;
    }
}

public class BSTChecker {
    
    public static boolean isBST(Node root) {
        if (root == null) return true;

      
        if (root.left != null && root.left.val >= root.val) return false;
        if (root.right != null && root.right.val <= root.val) return false;



        return isBST(root.left) && isBST(root.right);
    }

    public static void main(String[] args) {
        // Sample 1 - Not a BST
        Node root1 = new Node(4);
        root1.left = new Node(9);
        root1.right = new Node(2);


        root1.left.left = new Node(3);
        root1.left.right = new Node(-5);
        root1.right.right = new Node(7);
        System.out.println(isBST(root1)); 



        // Sample 2 - Is a BST
        Node root2 = new Node(4);
        root2.left = new Node(2);
        root2.right = new Node(7);
        root2.left.left = new Node(-5);


        root2.left.right = new Node(3);
        root2.right.right = new Node(9);
        System.out.println(isBST(root2)); 
    }
}

class Node {
    int val;
    Node left, right;

    Node(int val) {
        this.val = val;
        left = right = null;
    }
}

public class BSTRangeSum {
    
    public static int rangeSum(Node root, int low, int high) {
        if (root == null) return 0;

       
        if (root.val <= low) {
           
            return rangeSum(root.left, low, high);
        }

        
        if (root.val >= high) {
           
            return rangeSum(root.right, low, high);
        }

        


        return rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
    }

    public static void main(String[] args) {
       
        Node root1 = new Node(8);
        root1.left = new Node(4);
        root1.right = new Node(12);
        root1.left.left = new Node(2);
        root1.left.right = new Node(6);


        root1.right.left = new Node(10);
        root1.right.right = new Node(14);
        System.out.println(rangeSum(root1, 5, 13)); 

        // Example 2
        Node root2 = new Node(10);
        root2.left = new Node(5);
        root2.right = new Node(15);
        root2.left.left = new Node(3);
        root2.left.right = new Node(7);
        root2.right.right = new Node(18);
        System.out.println(rangeSum(root2, 7, 15)); 
    }
}
