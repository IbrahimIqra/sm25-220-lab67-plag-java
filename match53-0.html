public class Task5 {
    public static Integer subtractSummation( BTNode root ){
      if (root == null){ 
        return 0;}
        int left  =  totalSum(root.left);
        int right =  totalSum(root.right);
        return left - right;
    }
    public static int totalSum(BTNode node){
        if (node == null) 
          return 0;
        int val = (Integer) node.elem;
return val+ totalSum(node.left) + totalSum(node.right);
    }
  
}


//Difference of Level Sum :
public class Task6 {
    public static Integer levelSum( BTNode root ){
    return sumHelper(root, 1); 
    }
public static int sumHelper(BTNode node, int level) {
        if (node == null)
            return 0;

        int val = (Integer) node.elem;
        int result;

        if (level % 2 == 1) {
            result = -val;
        } else {
            result = val;
        }

        int leftSum = sumHelper(node.left, level + 1);
        int rightSum = sumHelper(node.right, level + 1);

        return result + leftSum + rightSum;
    }
}

public class Task4 {
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
      if (root == null){ 
        return 0;}
      
 int result = 0;
if (root.elem >= low && root.elem <= high) {
            result += root.elem;
        }

        if(root.elem > low) {
            result += rangeSum(root.left, low, high);
        }
       if (root.elem < high) {
            result += rangeSum(root.right, low, high);
        }

        return result;
    }
}
     


//Mirror Sum of a BST 


public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
      
      if (root == null) {
        return 0; }
return sumHelper(root.left, root.right);
    }
public static int sumHelper(BSTNode left, BSTNode right) {
  
  if (left == null || right == null){ 
    return 0;}

       
        int result = left.elem + right.elem;
        result += sumHelper(left.left, right.right);
        result+= sumHelper(left.right, right.left);

        return result; 
    }
}

public class Task6 {
    public static Boolean isBST( BSTNode root ){
        return isBSTHelper(root, null, null);
    }

    public static boolean isBSTHelper(BSTNode node, Integer small, Integer large) {
      if (node == null){ 
        return true;}


 if ((small!= null && node.elem <= small) || (large != null && node.elem >= large)) {
            return false;
        }

        return 
          isBSTHelper(node.left, small, node.elem) &&
               isBSTHelper(node.right, node.elem, large);
    }
}
