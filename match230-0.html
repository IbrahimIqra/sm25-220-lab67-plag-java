public class Task4 {

   


    public static void swapChild( BTNode root, int lvl, int M ){
      
      if(root ==null ){
      return;
      }
      if(lvl>=M){
        return;
      }
      BTNode temp=root.left;
      root.left=root.right;
      root.right=temp;
      
      if(root.left!=null){swapChild(root.left,lvl +1,M);}
      
      if(root.right!=null) {swapChild(root.right,lvl +1,M);
      }
    }
   
}

        return sumhelper(root,0);

    private static int sumhelper(BTNode root,int lvl){
      if(root==null){
      return 0;
      }
      int val=(int)root.elem;
      if(lvl%2==0){
      return -val + sumhelper(root.left,lvl+1)+sumhelper(root.right,lvl+1);
      }
      return val + sumhelper(root.left,lvl+1)+sumhelper(root.right,lvl+1);
    }
   
}

      if(root==null){

      return val + sumhelp(root.left,low2,high2)+sumhelp(root.right,low2,high2);
      }

    public static Boolean isBST( BSTNode root ){
      return ishelper(root,null,null);
    }
 private static boolean ishelper(BSTNode temp, Integer minval,Integer maxval) {
     if (temp == null) {
        return true;
      }


     if (temp.elem <= minval || temp.elem >= maxval) {
         return false;
     }    
return ishelper(temp.left, minval, temp.elem) && ishelper(temp.right, temp.elem, maxval);
