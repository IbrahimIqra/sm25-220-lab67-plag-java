public class Task4 {
  public static void swapChild(BTNode root, int lvl, int M) {
    if (root == null){
      return;
    }
    
    if (lvl >= M){
      return;
    }
    
    BTNode t = root.left;
    root.left = root.right;
    root.right = t;
    
    swapChild(root.left,  lvl + 1, M);
    swapChild(root.right, lvl + 1, M);
  }
}

public class Task5 {


    public static Integer subtractSummation(BTNode root){
      
      if (root == null){
          return 0;
      }
        int leftSum = subtreeSum(root.left);
        int rightSum = subtreeSum(root.right);
        return leftSum - rightSum;
    }
    
    private static int subtreeSum(BTNode node){
        if (node == null) return 0;
        return (Integer) node.elem + subtreeSum(node.left) + subtreeSum(node.right);
    }
}

public class Task6 {
  public static Integer levelSum(BTNode root){
    return levelSumHelper(root, 0); // start from level 0 (even)
  }


  private static Integer levelSumHelper(BTNode node, int level){
    if(node == null){
      return 0;
    }
    
    int val = (Integer) node.elem; // typecast to Integer
    if(level % 2 == 0){
      return val * -1 + levelSumHelper(node.left, level+1) + levelSumHelper(node.right, level+1);
    }
    else{
      return val + levelSumHelper(node.left, level+1) + levelSumHelper(node.right, level+1);
    }
  }
}

public class Task4 {


  public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
    if (root == null){
      return 0;
    }
    
    int sum = 0;
    
    if (root.elem >= low && root.elem <= high){
      sum += root.elem;
    }
    
    if (root.elem > low){
      sum += rangeSum(root.left, low, high);
    }
    
    if (root.elem < high){
      sum += rangeSum(root.right, low, high);
    }
    return sum;
  }
}


TASK 05

public class Task5 {


  public static Integer mirrorSum( BSTNode root ){
    if (root == null){
      return 0;
    }
    return mirrorSumHelper(root.left, root.right);
  }
  private static int mirrorSumHelper(BSTNode left, BSTNode right) {
    if (left == null || right == null){
      return 0;
    }
    int sum = left.elem + right.elem;
    
    sum += mirrorSumHelper(left.left, right.right);
    sum += mirrorSumHelper(left.right, right.left);
    
    return sum;
  }
}

public class Task6 {
  public static Boolean isBST( BSTNode root ){
    return isBSTHelper(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
  }
  
  private static boolean isBSTHelper(BSTNode node, int min, int max) {
    if (node == null){
      return true;
    }
    
    if (node.elem <= min || node.elem >= max) {
      return false;
    }
    return isBSTHelper(node.left, min, node.elem) &&
      isBSTHelper(node.right, node.elem, max);
  }
}
