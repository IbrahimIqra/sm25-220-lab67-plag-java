 private static Integer sumHelper(BSTNode root1, BSTNode root2){
        if(root1==null || root2 == null){
           return 0;
        }

        int add=(Integer)root1.elem+(Integer)root2.elem;
        add+=sumHelper(root1.left, root2.right);
        add+=sumHelper(root1.right, root2.left);
