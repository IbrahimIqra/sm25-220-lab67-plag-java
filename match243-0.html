public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
if(root == null){
   return 0;
}
int sum=0;
if(low<=root.elem && root.elem<=high){
   sum+=root.elem;
}
if(low<=root.elem){
   sum+=rangeSum(root.left,low,high);
}
if(high>=root.elem){
   sum+=rangeSum(root.right,low,high);
}


return sum;
}

public static Integer mirrorSum( BSTNode root ){
if(root == null){
    return 0;
}
return helperfunc(root.left, root.right);
}
private static int helperfunc(BSTNode rootLeft, BSTNode rootRight){
if(rootLeft==null || rootRight==null){
    return 0;
}
int sum=rootLeft.elem + rootRight.elem;
sum+=helperfunc(rootLeft.left , rootRight.right);
sum+=helperfunc(rootLeft.right , rootRight.left);
return sum;
}
