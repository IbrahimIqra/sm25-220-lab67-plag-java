public class Task4 {
    public static void swapChild(BTNode root, int lvl, int M) {
        if (root == null) return;


        if (lvl <= M) {
            BTNode temp = root.left;
            root.left = root.right;


            root.right = temp;
        }


        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }
}

public class Task5 {
    public static Integer subtractSummation(BTNode root) {
        if (root == null) {
            return 0;
        }
        
        int ls = sum(root.left);
        int rs = sum(root.right);
        
        return ls - rs;
    }


    private static int sum(BTNode node) {
        if (node == null) {
            return 0;
        }
        return (Integer)node.elem + sum(node.left) + sum(node.right);
    }
}

public class Task6 {


    public static Integer levelSum(BTNode root) {
        int h = height(root);
        int max = Integer.MIN_VALUE;


        for (int level = 1; level <= h; level++) {
            int sum = sumAtLevel(root, level);
            if (sum > max) max = sum;
        }
        return max;
    }


    private static int height(BTNode node) {
        if (node == null) return 0;
        return 1 + Math.max(height(node.left), height(node.right));
    }


    private static int sumAtLevel(BTNode node, int level) {
        if (node == null) return 0;
        if (level == 1) return (Integer) node.elem;
        return sumAtLevel(node.left, level - 1) + sumAtLevel(node.right, level - 1);
    }
}

public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
    if (root == null) return 0;


    int sum = 0;
    if (root.data >= low && root.data <= high) {
        sum += root.data;
    }
    if (root.data > low) { // left subtree might have valid values
        sum += rangeSum(root.left, low, high);
    }
    if (root.data < high) { // right subtree might have valid values
        sum += rangeSum(root.right, low, high);
    }


    return sum;
}

public static Integer mirrorSum(BSTNode root) {


    if (root == null) return 0;
    return mirrorSumHelper(root.left, root.right);
}


private static int mirrorSumHelper(BSTNode left, BSTNode right) {
    if (left == null || right == null) return 0;


    // sum the mirrored nodes + recurse for children
    int sum = left.data + right.data;
    sum += mirrorSumHelper(left.left, right.right);
    sum += mirrorSumHelper(left.right, right.left);


    return sum;
}


//TASK6
public static Boolean isBST(BSTNode root) {


    return isBSTHelper(root, null, null);
}


private static boolean isBSTHelper(BSTNode node, Integer min, Integer max) {
    if (node == null) return true;


    if ((min != null && node.data <= min) || (max != null && node.data >= max)) {
        return false;
    }


    return isBSTHelper(node.left, min, node.data) && 
           isBSTHelper(node.right, node.data, max);
}
