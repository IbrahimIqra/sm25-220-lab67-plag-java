    public static void swapChild( BTNode root, int lvl, int M ){
      if(root==null){
         return;
      }

      if(lvl<M){
        BTNode temp=root.left;
        root.left=root.right;
        root.right=temp;
        
      }
      swapChild(root.left,lvl+1,M );
      swapChild(root.right,lvl+1,M );
    }

    public static Integer subtractSummation( BTNode root ){
      if(root==null){
        return 0;
      }
      int ltotal=calc(root.left);
      int rtotal=calc(root.right);
      return ltotal-rtotal;
    }
    private static int calc(BTNode val){
      if(val==null){
      return 0;
      }
      int ltotal=calc(val.left);
      int rtotal=calc(val.right);
      return (int)val.elem+ltotal+rtotal;
      
    }
Task03


    public static Integer levelSum( BTNode root ){
      return calc(root,0);
    }
    private static int calc(BTNode val,int lv){
      if(val==null){
        return 0;
      }
      int crntVal=(int)val.elem;
      if(lv%2==0){
        crntVal=-1*crntVal;
      }
      int leftNodes=calc(val.left,lv+1);
      int rightNodes=calc(val.right,lv+1);
      return crntVal+leftNodes+rightNodes;
    }

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
      if(root==null){
        return 0;
      }
      return calc(root,low,high);
    }
    private static Integer calc(BSTNode val,Integer l,Integer h){
      if(val==null){
        return 0;
      }
      if(val.elem<=h && val.elem>=l){
        return (Integer)val.elem+calc(val.left,l,h)+calc(val.right,l,h);
      }
      else{
        return calc(val.left,l,h)+calc(val.right,l,h);
      }
    }

    public static Integer mirrorSum( BSTNode root ){
      int sum;


      if(root==null){
        return 0;
      }
      
      return lctMrr(root.left,root.right);
    }
      
      
    private static Integer lctMrr(BSTNode lftVal,BSTNode rgtVal){
      if(lftVal==null || rgtVal==null){
      return 0;
      }


      Integer sum=(Integer)lftVal.elem+(Integer)rgtVal.elem;
      sum+=lctMrr(lftVal.left,rgtVal.right);
      sum+=lctMrr(lftVal.right,rgtVal.left);
      
      return sum;
    }

    public static Boolean isBST( BSTNode root ){
      return tree(root,null,null);     
    }
    private static Boolean tree(BSTNode val,Integer min,Integer max){
      if(val==null){
        return true;
      }
      if(max!=null && val.elem>=max || min!=null && val.elem<=min){


        return false;
      }
      return tree(val.left,min,val.elem) && tree(val.right,val.elem,max);
    }
