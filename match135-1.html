public class Task4 {

    
    public static void swapChild( BTNode root, int lvl, int M ){


      if(root==null){
        return ;
      }
      if(lvl<M){
        BTNode temp=root.right;
        root.right=root.left;
        root.left=temp;
      }
      if(lvl>=M){
        root.right=root.right;
        root.left=root.left;
      }
      
      
      else{
        


        swapChild( root.left,  lvl+1,  M );
        swapChild( root.right,  lvl+1,  M );
        
      
       
    }
    
}
}

public class Task4 {
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
      int sum=0;
      if(root==null){
        return 0; 
      }
        if(root.elem>=low && root.elem<=high){
          sum+=(int)root.elem;


        }
        if(root.elem>low){
        sum+=rangeSum( root.left,low,  high );
        }
        if(root.elem<high){
        
         sum+= rangeSum(root.right,  low,  high );
        }

          return sum;
        
   
}
}

public class Task5 {

    
    public static Integer mirrorSum( BSTNode root ){



        int result= helper_fun(root.left,root.right);
        return result;

    }
    public static Integer helper_fun( BSTNode root1,BSTNode root2 ){
      
      if(root1==null&&root2==null){
        return 0;
      }
      if(root1==null || root2==null){
        return 0;
      }

      return root1.elem+root2.elem+helper_fun(root1.left,root2.right)+helper_fun(root1.right,root2.left);
      
    }
}

public class Task6 {

    public static Boolean isBST( BSTNode root ){


      if(root==null){
        
        return true; 
    }
      if(root.left!=null && root.left.elem>=root.elem){
        return false;
      }
      if(root.right!=null && root.right.elem<=root.elem){
        return false;
      }
      return isBST(root.left)&&isBST(root.right);
    }
    
    


}
