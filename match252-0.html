       public static void swapChild( BTNode root, int lvl, int M ){
        //TO DO
        if (root== null){
            return;
        }
        if (lvl < M){
            BTNode temp = root.right;
            root.right = root.left;
            root.left = temp;
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }

    //============================================================================
}

public class Task4 {

    //===================================TASK#4======================
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        //TO DO
        if (root== null){
            return 0 ;
        }
        int val = (Integer) root.elem;
        int sum = 0;
        if (val >= low && val <= high){
           sum += val;
        }
        sum += rangeSum(root.left, low, high);


        sum += rangeSum(root.right, low, high);
        return sum;
    }
  
    //===============================================================


}

       public static Integer mirrorSum( BSTNode root ){
        //TO DO
        if (root==null){
            return 0;
        }
        return help(root.left,root.right);
    }
    public static Integer help( BSTNode root1, BSTNode root2 ){
        if (root1==null|| root2==null ){
            return 0;
        }
        int sum = root1.elem +root2.elem;


        sum +=help(root1.left,root2.right);
        sum +=help(root1.right,root2.left);
        return sum;
    }
    //===============================================================


}





public class Task6 {

    //===================================TASK#6======================
     public static Boolean isBST( BSTNode root ){
        //TO DO
        if (root==null){
            return true;
        }
        BSTNode val1=root.left;
        BSTNode val2=root.right;


        return help(root,val1,val2);
    }
    public static Boolean help( BSTNode root , BSTNode a , BSTNode b){

        if (root==null){
            return true;
        }
        int val = (Integer)root.elem;
        int min = (Integer)a.elem;
        int max = (Integer)b.elem;
        if(val<=min || val>=max){
            return false;
        }
        return help(root.left,root.left.left,root.left.right) && help(root.right,root.right.left,root.right.right);
    }
    //===============================================================


}
