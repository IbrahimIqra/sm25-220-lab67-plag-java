  public static void swapChild( BTNode root, int lvl, int M ){
    if(root==null){
      return ;
    }
    if(lvl<M){
      BTNode temp=root.right;
      root.right=root.left;
      root.left=temp;
    }
    swapChild( root.left,lvl+1,M );
    swapChild( root.right,lvl+1,M );
    
  }

  public static Integer subtractSummation( BTNode root ){
    if(root==null){
      return 0;
    }
    int sum1=sumHelper(root.left) ;
    int sum2=sumHelper(root.right);
    
    return sum1-sum2;
  }
  
  private static Integer sumHelper(BTNode root){
    if(root==null){
      return 0;
    }
    int sum=0;
    sum=(Integer)root.elem;
    sum+=(Integer)sumHelper(root.left);
    sum+=(Integer)sumHelper(root.right);
    return sum;
  }

  public static Integer levelSum( BTNode root ){
    return  sumHelper(root,0);
  }


  private static Integer sumHelper(BTNode root,int lvl){
    if(root==null){
      return 0;
    }
    int sum=0;
    if(lvl%2==0){
      sum-=(Integer)root.elem;
    }
    else{


      sum+=(Integer)root.elem;
    }
    sum+=sumHelper(root.left,lvl+1);
    sum+=sumHelper(root.right,lvl+1);
    return sum;
  }

  public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
    if(root == null){
      return 0;
    }
     Integer sum=0;


    if(root.elem>=low && root.elem<=high){
      sum+=root.elem;
    }
    sum+=rangeSum(root.left,low,high);
    sum+=rangeSum(root.right,low,high);
    return sum;
  }




//Task 05

  public static Integer mirrorSum( BSTNode root ){
    if(root==null){
      return 0;
    }


    return helperSum(root.right,root.left);
    
  }
  private static Integer helperSum(BSTNode left, BSTNode right){
    if(right ==null|| left== null){
      return 0;
    }
    int sum=0;


    sum+= left.elem+right.elem;
    
    sum+=helperSum(left.right,right.left);
    sum+=helperSum(right.right,left.left);
    return sum;
  }

  public static Boolean isBST( BSTNode root ){
    if(root==null){
      return true;
    }
    return helperM(root);
  }
  private static Boolean helperM(BSTNode root){
    if(root == null){
      return true;
    }
    boolean flag=true;
    if(root.left!=null && root.left.elem>root.elem){
      flag=false;
    }
    else if( root.right!=null && root.right.elem<root.elem){
      flag=true;
    }
    
    if(helperM(root.left) && helperM(root.right)){
      return flag;
    }
    return false ;
  }
