public class Task4 {

    public static void swapChild( BTNode root, int lvl, int M ){
        if(root== null) {
            return;
        }
        if (lvl <= M-1){
            BTNode t1 = root.left;
            root.left = root.right;


            root.right = t1;

        }
        swapChild(  root.left,lvl+1, M );
        swapChild(  root.right,lvl+1, M );
    }
    //============================================================================

}

public class Task5 {


    public static Integer subtractSummation( BTNode root ) {

        if (root == null) {
            return 0;
        }

        int leftSum = sumHelp(root.left);
        int rightSum = sumHelp(root.right);

        return leftSum - rightSum;
    }

        private static int sumHelp(BTNode nd ){

            if( nd == null){
                return 0;
            }

            int val = (int)nd.elem;
            int left_val = sumHelp(nd.left);
            int right_val = sumHelp(nd.right);

            return val+ left_val+ right_val;
        }



    }

public class Task6 {

    
    
    public static Integer levelSum( BTNode root ){
        if(root ==null){
            return 0;
        }
        int oddone = sumHelper(root, 0, true);
        int eventone = sumHelper(root, 0, false);

        return oddone - eventone;


    }

    private static int sumHelper(BTNode nd, int lvl, boolean flag) {
        if(nd==null){
            return 0;
        }


        int currVal = 0;
        boolean isOddflag = (lvl % 2 !=0 );

        if (flag && isOddflag) {
            currVal = (int) nd.elem;
        }
        else if (!flag && !isOddflag) {
            currVal = (int) nd.elem;
        }


        int leftSum= sumHelper(nd.left, lvl + 1,  flag);
        int rightSum= sumHelper(nd.right, lvl + 1,  flag);



        return currVal+ leftSum+ rightSum;

    }

    //============================================================================

}

public class Task4 {

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root == null){
            return 0;
        }

        else if(root.elem> high){
            return rangeSum( root.left,  low,  high );
        }

        else if(root.elem< low){
            return rangeSum( root.right,  low,  high );
        }

        return root.elem + rangeSum( root.left,  low,  high )+ rangeSum( root.right,  low,  high );


    }

}

public class Task5 {

    public static Integer mirrorSum( BSTNode root ) {
        if (root == null) {
            return 0;
        }
        return sumHelper(root.left, root.right);
    }

        private static int sumHelper(BSTNode ndleft, BSTNode ndright){
            if(ndleft == null || ndright== null){
                return 0;
            }
            int currpair = ndleft.elem + ndright.elem ;

            int outer = sumHelper(ndleft.left , ndright.right);
            int inner = sumHelper(ndleft.right , ndright.left);

            return currpair + outer+ inner;

        }
    }

public class Task6 {

    public static Boolean isBST( BSTNode root ) {
        return calcHelper(root, null, null);

    }

    private static Boolean calcHelper(BSTNode nd, Integer mn, Integer mx) {
        if (nd == null) {
            return true;


        }
         if ((mn != null && nd.elem <= mn) || (mx != null && nd.elem >= mx)) {
            return false;
        }

        return calcHelper(nd.left, mn, nd.elem) && calcHelper(nd.right, nd.elem, mx);
    }
