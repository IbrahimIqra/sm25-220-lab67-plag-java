public static Integer levelSum( BTNode root){
        if (root == null) {
            return 0;
        }
        
        int oddSum = calculateOddLevelSum(root, 1);
        int evenSum = calculateEvenLevelSum(root, 1);
        return evenSum - oddSum;
    }
private static int calculateOddLevelSum(BTNode node, int level){
        if (node == null) {
            return 0;
        }
         int sum = 0;
        if (level % 2 == 1) {
            sum += (Integer)node.elem;
        }
        sum += calculateOddLevelSum(node.left, level + 1);
        sum += calculateOddLevelSum(node.right, level + 1);
        
        return sum;
    }
      private static int calculateEvenLevelSum(BTNode node, int level){
        if (node == null) {
            return 0;
        }
         int sum = 0;
        if (level % 2 == 0) {
            sum += (Integer)node.elem;
        }
         sum += calculateEvenLevelSum(node.left, level + 1);
        sum += calculateEvenLevelSum(node.right, level + 1);
        
        return sum;
    }
}
