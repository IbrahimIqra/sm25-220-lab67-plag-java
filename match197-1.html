public static void swapChild( BTNode root, int lvl, int M ){
        if (root == null) {
            return;
        }
        if (lvl < M) {
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }


//task 05
 public static Integer subtractSummation( BTNode root ){
        return sum(root.left)-sum(root.right);
    }
    private static int sum(BTNode sroot){
        if (sroot == null) {
            return 0;
        }
        int sum = (int)sroot.elem;


        return sum + sum(sroot.left) + sum(sroot.right);
    }

public static Integer levelSum( BTNode root ){
        return helper(root, 0);
    }
    private static int helper(BTNode troot, int level){
        if (troot == null) {
            return 0;
        }
        int sum = (int)troot.elem;
        if (level%2==0) {


            sum = -sum;
        }
        return sum + helper(troot.left, level+1) + helper(troot.right, level+1);
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if (root == null) {
            return 0;
        }

        if (root.elem < low) {
            return rangeSum(root.right, low, high);
        } 
        else if (root.elem > high) {
            return rangeSum(root.left, low, high);
        } 
        else {
            return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
        }
    }


//task 05
public static Integer mirrorSum( BSTNode root ){
        return helper(root.left, root.right);
    }


    private static int helper(BSTNode left, BSTNode right){
        if (left == null || right == null) {
            return 0;
        }
        int sum = left.elem + right.elem;
        return sum + helper(left.left, right.right) + helper(left.right, right.left);
    }


//task 06


public static Boolean isBST( BSTNode root ){
        return isHelper(root, null, null);
    }
    private static boolean isHelper(BSTNode node, Integer low, Integer high){
        if (node == null) {
            return true;
        }
        if (low != null && node.elem <= low) {
            return false;
        }
        if (high != null && node.elem >= high) {
            return false;
        }

        return isHelper(node.left, low, node.elem) && isHelper(node.right, node.elem, high);
    }
