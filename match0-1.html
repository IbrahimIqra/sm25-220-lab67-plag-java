    public static void swapChild(BTNode root, int lvl, int M) {
        if (root == null) {
            return;
        }
        if (lvl <= M) {
            BTNode t = root.left;
            root.left = root.right;
            root.right = t;
        }
        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }
}

    private static class LevelNode {
        BTNode n;
        int l;
        LevelNode(BTNode n, int l) {
            this.n = n;
            this.l = l;
        }
    }
    public static Integer levelSum(BTNode root) {
        if (root == null) {
            return 0;
        }
        LevelNode[] q = new LevelNode[1000];
        int start = 0;
        int end = 0;
        q[end] = new LevelNode(root, 0);
        end++;


        int even = 0;
        int odd = 0;
        while (start < end) {
            LevelNode curr = q[start];
            start++;
            BTNode node = curr.n;
            int level = curr.l;
            if (level % 2 == 0) {
                even = even + (Integer) node.elem;
            } else {
                odd = odd + (Integer) node.elem;
            }
            if (node.left != null) {
                q[end] = new LevelNode(node.left, level + 1);
                end++;
            }
            if (node.right != null) {
                q[end] = new LevelNode(node.right, level + 1);
                end++;
            }
        }
        return even - odd;
    }
}

public class Task4 {
    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) {
            return 0;
        }
        int x = root.elem;
        if (x < low) {
            return rangeSum(root.right, low, high);
        }
        if (x > high) {
            return rangeSum(root.left, low, high);
        }
        int left = rangeSum(root.left, low, high);
        int right = rangeSum(root.right, low, high);
        return x + left + right;
    }
}

public class Task5 {
    private static int mirror(BSTNode l, BSTNode r) {
        if (l == null || r == null) {
            return 0;
        }
        int s = l.elem + r.elem;
        int a = mirror(l.left, r.right);
        int b = mirror(l.right, r.left);
        return s + a + b;
    }
    public static Integer mirrorSum(BSTNode root) {
        if (root == null) {
            return 0;
        }
        return mirror(root.left, root.right);
    }
}

public class Task6 {


    private static boolean check(BSTNode n, int min, int max) {
        if (n == null) {
            return true;
        }
        int x = n.elem;
        if (x <= min || x >= max) {
            return false;
        }
        boolean leftOk = check(n.left, min, x);
        boolean rightOk = check(n.right, x, max);
        return leftOk && rightOk;
    }
    public static Boolean isBST(BSTNode root) {
        return check(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }
}
