public class Task4 {


    public static void swapChild(BTNode root, int level, int M){
        if(root==null) return;
        if(level<M){
            BTNode temp=root.left;
            root.left=root.right;
            root.right=temp;
        }
        swapChild(root.left,level+1,M);
        swapChild(root.right,level+1,M);
    }
}

        if(root==null){
            return 0;

    }

}

        return lvlsum1(root, 0)-lvlsum2(root,0);
    }

        if(root==null){
            return 0;
        }
        if(lvl%2!=0){
            return (Integer)root.elem+lvlsum1(root.left,lvl+1)+lvlsum1(root.right,lvl+1);
        } else {
            return lvlsum1(root.left, lvl+1) + lvlsum1(root.right, lvl+1);
        }
    }
    public static Integer lvlsum2(BTNode root, int lvl){
        if(root==null){
            return 0;
        }

            return (Integer)root.elem+lvlsum2(root.left,lvl+1)+lvlsum2(root.right,lvl+1);
        }
        else{
            return lvlsum2(root.left, lvl+1) + lvlsum2(root.right, lvl+1);
        }

public class Task4 {
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root == null){
            return 0;
        }
        int sum = 0;
        if(root.elem > low){
            sum += rangeSum(root.left, low, high);
        }
        if(root.elem < high){
            sum += rangeSum(root.right, low, high);
        }
        if(root.elem >= low && root.elem <= high){
            sum += root.elem;
        }
        return sum;
    }
}

public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
        if(root==null) return 0;
        return mrrsum(root.left, root.right);
    }
    public static Integer mrrsum(BSTNode root1, BSTNode root2){
        if(root1==null && root2==null){
            return 0;
        }
        


        if(root1==null || root2==null){
            return 0;
        }
        int sum=root1.elem+root2.elem;
        sum+=mrrsum(root1.left,root2.right);
        sum+=mrrsum(root1.right,root2.left);
        return sum;
    }
}

public class Task6 {
    public static Boolean isBST( BSTNode root ){


        return helper(root, null, null);
    }
    public static Boolean helper(BSTNode root, Integer min, Integer max){
        if(root==null) return true;
        if((min!=null && root.elem<=min)||(max!=null && root.elem>=max)) return false;
        return helper(root.left, min, root.elem) && helper(root.right, root.elem, max);
    }}
