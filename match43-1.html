    //===================================TASK#4===================================
    // This method takes only 3 parameters
    // 1st one is root of the given tree
    // 2nd one is there to help you count the levels
    // 3rd one is the level TILL whic you need to swap childs
    // All the changes will happen in-place
    // This method doesn't return anything
    // You can use extra helper private static methods as per need


    public static void swapChild( BTNode root, int lvl, int M ){
        //TO DO
      if(root==null)
        return;
      if(lvl>=M){
        BTNode temp = root.left;
        root.left=root.right;
        root.right=temp;
      }
      swapChild(root.left, lvl+1, M);
      swapChild(root.right, lvl+1, M);
    }
    //============================================================================
}

        //TO DO
       if(root==null)


         return null;
       int ls=sumMethod(root.left);
       int rs=sumMethod(root.right);
       return ls-rs;
    }
    private static int sumMethod(BTNode node){
      if(node==null)
        return 0;
      int l=sumMethod(node.left);

    // You can use extra helper private static methods with extra extra params as per need
    public static Integer levelSum( BTNode root ){
        //TO DO
        if(root==null)
          return 0;
        int a = helperSum(root, 1);
        return a;
    }


    private static int helperSum(BTNode node, int lvl){
      if(node==null)
        return 0;
      int val=(Integer)node.elem;
      if(lvl%2==1){
        return val+helperSum(node.left, lvl+1);
      }
      else{
        return -val+helperSum(node.left, lvl+1);
      }
      
    }
    //============================================================================

}

public class Task4 {

    //===================================TASK#4======================
    // This method takes only three parameters
    // first one is the root of the given tree
    // second one the low range and third one is high range
    // You can use extra helper private static methods as per need
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        //TO DO
        if(root==null)
          return 0;
        int sum=0;
        if(root.elem>=low && root.elem<=high){
          sum+=root.elem;
        }
        if(root.elem>low){
          sum+=rangeSum(root.left, low, high);
        }
        if(root.elem<high){
          sum+= rangeSum(root.right, low, high);
        }
        return sum;
    }
    //===============================================================


}

    public static Integer mirrorSum( BSTNode root ){
        //TO DO
      if(root==null){
        return 0;
      }
      return sumHelper(root.left, root.right);
    }
    private static int sumHelper(BSTNode l, BSTNode r){
      if(l==null || r==null){
        return 0;
      }
      int sum = 0;


      sum+=l.elem+r.elem;
      sum+=sumHelper(l.left, r.right);
      sum+=sumHelper(l.right, r.left);
      return sum;
    }
    //===============================================================


}

    //===================================TASK#6======================
    // This method takes only one parameter
    // it is root of the given tree
    // You can use extra helper private static methods as per need
    public static Boolean isBST( BSTNode root ){
        //TO DO
       return helperFunc(root, null, null);
    }
    private static boolean helperFunc(BSTNode node, Integer min, Integer max){
      if(node==null)
        return true;
      if((min!=null && node.elem<=min) || (max!=null && node.elem>=max)){
        return false;
      }
      Boolean a= helperFunc(node.left, min, node.elem);
      Boolean b= helperFunc(node.right, node.elem, max);
      return a&&b;
    }
    //===============================================================


}
