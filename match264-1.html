    private static Integer subtreesum(BSTNode lefttree, BSTNode righttree, int sum){
        if (lefttree==null || righttree==null){
            return 0;
        }
        else{
            sum=lefttree.elem+righttree.elem;
            sum=sum+subtreesum(lefttree.left, righttree.right, sum);
            sum=sum+subtreesum(lefttree.right, righttree.left, sum);
            return sum;
        }
    }

 public static Boolean isBST( BSTNode root ){
        //TO DO
        


       return isBSTHelper(root, 0, 0);

     
    }
    private static Boolean isBSTHelper(BSTNode node, Integer min, Integer max) {
    if (node == null) {
        return true;
    }
    
    if (min != null && node.elem <= min) {
        return false;
    }
   
    if (max != null && node.elem >= max) {
        return false;
    }
    
    Boolean leftOK = isBSTHelper(node.left, min, node.elem);
    Boolean rightOK = isBSTHelper(node.right, node.elem, max);

    return leftOK && rightOK;
}
