public static void swapChild( BTNode root, int lvl, int M ){
       if(root == null || lvl == M){
        return;
       }

       swapChild(root.left, lvl + 1, M);
       swapChild(root.right, lvl + 1, M);
       BTNode temp = root.left;
       root.left = root.right;


       root.right = temp; 
    }

//Task5
public static Integer subtractSummation( BTNode root ){
        if(root == null){
            return 0;
        }
        int left = helper(root.left);
        int right = helper(root.right);

        return left - right;
    }
    public static int helper(BTNode root){
        if (root == null){
            return 0;
        }
        return (int)root.elem + helper(root.left) + helper(root.right);
    }

//Task6
public static Integer levelSum( BTNode root ){
       return helper(root, 0);
    }

    public static int helper(BTNode root, int level){
        if (root == null){
            return 0;
        }

        int key = (int) root.elem;

        if(level % 2 == 0){
            return -key + helper(root.left, level + 1) + helper(root.right, level + 1);
        }
        else{
            return key + helper(root.left, level + 1) + helper(root.right, level + 1);
        }
    }

//Task7
public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        return rangehelper(root, low, high, 0);
    }
    private static Integer rangehelper(BSTNode node, Integer low, Integer high, Integer sum) {
        if (node == null) return sum;
        
        if (node.elem < low) {
            return rangehelper(node.right, low, high, sum);
        }
        
        if (node.elem > high) {
            return rangehelper(node.left, low, high, sum);
        }
        
        Integer current = sum + node.elem;
        current = rangehelper(node.left, low, high, current);


        return rangehelper(node.right, low, high, current);
    }

public static Integer mirrorSum( BSTNode root ){
        if(root == null){
            return 0;
        }

        return helper(root.left, root.right);
    }

    public static int helper(BSTNode left, BSTNode right){
        if(left == null || right == null){
            return 0;
        }

        int sum = left.elem + right.elem;

        sum += helper(left.left, right.right);
        sum += helper(left.right, right.left);

        return sum;
    }

//Task9
public static Boolean isBST( BSTNode root ){
        return helper(root, null, null);
    }

    public static boolean helper(BSTNode root, Integer max, Integer min){
        if(root ==null){
            return true;
        }

        boolean v = true;

        if (max != null){
            v = root.elem < max;
        }

         if (min != null){
            v = v && root.elem > min;
        }

        if(!v){
            return false;
        }

        return helper(root.left, root.elem, min) && helper(root.right, max, root.elem);
    }
