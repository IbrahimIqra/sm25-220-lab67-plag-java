public class Task4 {
    public static void swapChild(BTNode root, int lvl, int M) {
        if (root == null) {
            return;
        }

        int new_lvl = lvl + 1;



        if (new_lvl <= M) {
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }

        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }
}

public class Task5 {


    public static Integer subtractSummation( BTNode root ){
        if (root == null) {
            return 0;
        }

        int leftSum = calculateSubtreeSum(root.left);
        int rightSum = calculateSubtreeSum(root.right);

        return leftSum - rightSum;
    }

    private static int calculateSubtreeSum(BTNode node) {
        if (node == null) {
            return 0;
        }

        int currentValue = (Integer) node.elem;
        int leftSum = calculateSubtreeSum(node.left);
        int rightSum = calculateSubtreeSum(node.right);

        return currentValue + leftSum + rightSum;
    }
}

public class Task6 {


    public static Integer levelSum( BTNode root ){
        if (root == null) {
            return 0;
        }

        int[] levelSum = new int[2];
        calSum(root, 0, levelSum);

        return levelSum[1] - levelSum[0];
    }

    private static void calSum(BTNode node, int lvl, int[] levelSum) {
        if (node == null) {
            return;
        }

        int currVal = (Integer) node.elem;

        if (lvl % 2 == 0) {
            levelSum[0] += currVal;
        } else {
            levelSum[1] += currVal;
        }

        calSum(node.left, lvl + 1, levelSum);
        calSum(node.right, lvl + 1, levelSum);

    }
}

public class Task4 {
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if (root == null) {
            return 0;
        }
        if (root.elem < low) {
            return rangeSum(root.right, low, high);
        }
        if (root.elem > high) {
            return rangeSum(root.left, low, high);
        }
        return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
    }
}


Task_05:

public class Task5 {
    


    public static Integer mirrorSum( BSTNode root ){
        if (root == null) {
            return 0;
        }
        return helper(root.left, root.right);
    }

    private static Integer helper(BSTNode a, BSTNode b) {
        if (a == null || b == null) {
            return 0;
        }
        return a.elem + b.elem + helper(a.left, b.right) + helper(a.right, b.left);
    }
}

public class Task6 {


    public static Boolean isBST( BSTNode root ){
        return isBSTUtil(root, null, null);
    }
    private static Boolean isBSTUtil(BSTNode node, Integer min, Integer max) {
        if (node == null) {
            return true;
        }
        if ((min != null && node.elem <= min) || (max != null && node.elem >= max)) {
            return false;
        }
        return isBSTUtil(node.left, min, node.elem) && isBSTUtil(node.right, node.elem, max);
    }
}
