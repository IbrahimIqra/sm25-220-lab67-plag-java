public static void swapChild( BTNode root, int lvl, int M ){
        if(root==null || lvl>=M)
        {
            return;
        }

        
            BTNode tempNode = root.left;
            root.left = root.right;


            root.right = tempNode;
        

        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }
}

public static Integer subtractSummation( BTNode root ){
        if(root == null)
        {
            return -1;


        }
        int leftSum = totalSum(root.left);
        int rightSum = totalSum(root.right);
        return leftSum - rightSum;    }

    private static int totalSum(BTNode temp)
    {
        if(temp == null)
        {
            return 0;
        }
        return (int)temp.elem + totalSum(temp.left)+ totalSum(temp.right);
    }



// Task 6


public static Integer levelSum( BTNode root ){
        return tempMethod(root, 0);
    }

    private static int tempMethod(BTNode temp, int lvl){
        if(temp == null)
        {
            return 0;
        }

        Integer val = (Integer)temp.elem;

        if(lvl%2 == 0)
        {
            return -val + tempMethod(temp.left,lvl +1) + tempMethod(temp.right, lvl+1);
        }
        else
        {
             return val + tempMethod(temp.left,lvl +1) + tempMethod(temp.right, lvl+1);
        }
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if (root == null)
        {
            return 0;
        }
        Integer sum = 0;


        if (root.elem>=low && root.elem <=high)
        {
            sum+=root.elem;
        }

        if(root.elem>low)
        {
            sum+=rangeSum(root.left, low, high);
        }

        if(root.elem<high)
        {
            sum+=rangeSum(root.right, low, high);
        }

        return sum;
    }

public static Integer mirrorSum( BSTNode root ){
        if(root != null)
        {
            return check(root.left, root.right);
        }
        return 0;
    }
    private static Integer check(BSTNode leftnode, BSTNode rightnode)
        {
            if(leftnode == null || rightnode == null)
            {
                return 0;
            }


               Integer sum = leftnode.elem+rightnode.elem;

                sum+=check(leftnode.left, rightnode.right);
                sum+=check(leftnode.right, rightnode.left);
                return sum;
        }
