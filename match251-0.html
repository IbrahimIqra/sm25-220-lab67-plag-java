  public static void swapChild(BTNode root, int lvl, int M) {


    if (root == null || lvl >= M) {
      return;
    }
    if (lvl < M) {
      BTNode temp = root.left;
      root.left = root.right;
      root.right = temp;
    }
    swapChild(root.left, lvl + 1, M);
    swapChild(root.right, lvl + 1, M);
  }

  public static Integer subtractSummation( BTNode root ){
    if (root == null) {
      return 0;
    }


    int ls = sumHelper(root.left);
    int rs = sumHelper(root.right);
    return ls - rs;
  }
  private static int sumHelper(BTNode node) {
    if (node == null) {
      return 0;
    }
    return (Integer) node.elem + sumHelper(node.left) + sumHelper(node.right);
  }
}

  public static Integer levelSum(BTNode root) {
    return levelSumHelper(root, 1);
  }
  private static int levelSumHelper(BTNode node, int level) {
    if (node == null) {
      return 0;
    }
    int value = (Integer) node.elem;
    if (level % 2 == 0) {
      value = +value;
    }
    else {
      value = -value;
    }  
    return value + levelSumHelper(node.left, level + 1) + levelSumHelper(node.right, level + 1);
  }
}

  public static int rangeSum(BSTNode root, int low, int high) {


    if (root == null) {
      return 0;
    }
    if (root.elem < low) {
      return rangeSum(root.right, low, high);
    }
    if (root.elem > high) {
      return rangeSum(root.left, low, high);
    }
    return root.elem + rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
  }

  public static Integer mirrorSum (BSTNode root) {
    if (root == null) {
      return 0;
    }
    return helper(root.left, root.right);
  }
  public static int helper (BSTNode left, BSTNode right) {
    if (left == null || right == null) {
      return 0;
    }
    return left.elem + right.elem + helper(left.left, right.right) + helper(left.right, right.left);
  }
}

  public static Boolean isBST(BSTNode root) {
    if (root == null) {
      return true;
    }
    if (root.left != null && root.left.elem >= root.elem) {
      return false;
    }
    if (root.right != null && root.right.elem <= root.elem) {
      return false;
    }
    return isBST(root.left) && isBST(root.right);
  }
