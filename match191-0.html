public class Task4 {


    public static void swapChild( BTNode root, int lvl, int M ){
        if (root == null) {
            return;
        }
        if (lvl < M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
            
            
            swapChild(root.right, lvl+1,M);
            swapChild(root.left, lvl+1, M);
        }
    }
}

public class Task5 {
    public static Integer subtractSummation( BTNode root ){
       if (root == null) {


            return null; 
        }


        int leftSum = sum(root.left);
        int rightSum = sum(root.right);


        return leftSum - rightSum;
    }
     private static int sum(BTNode node) {
        if (node == null) {
            return 0;
        }
        return (int) node.elem + sum(node.left) + sum(node.right);
    }


}

public class Task6 {
     public static Integer levelSum(BTNode root) {
        if (root == null) {
            return null;
        }


        return sum(root, 0);
    }


    private static int sum(BTNode node, int lvl) {
        if (node == null) {
            return 0;
        }


        int val = (int) node.elem; 


        if (lvl % 2 == 0) {
            
            return -val + sum(node.left, lvl + 1) + sum(node.right, lvl + 1);
        } else {
            
            return val + sum(node.left, lvl + 1) + sum(node.right, lvl + 1);
        }
    }
}

public class Task4 {
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root == null){
            return 0;
        }
        
        int sum = 0;
        if(root.elem > low){
            sum += rangeSum(root.left,low, high);
            
        }
        if(root.elem >= low  && root.elem <= high){
            sum += root.elem;
        }
        if(root.elem < high ){
            sum += rangeSum(root.right,low, high);
        }
        
        return sum;
        }
}

public class Task5 {


    public static int mirrorSum (BSTnode root) {
        if (root == null) {
            return 0;
        }
        return helper(root.left, root.right);
    }
    


    private int helper(BSTnode left, BSTnode right) {
        if (left == null || right == null) {
            return 0;
        }
        int sum = left.elem + right.elem;
        sum += helper(left.left, right.right);
        sum += helper(left.right, right.left);
        return sum;
    }


}

public class Task6 {
    public static boolean isBST(BSTNode root) {
        return helper(root, null, null);
    }
    private static boolean helper(BSTNode node, Integer min, Integer max) {
        if (node == null) return true;
        if (min != null && node.elem <= min) return false;


        if (max != null && node.elem >= max) return false;


        return helper(node.left, min, node.elem) &&
               helper(node.right, node.elem, max);
    }
}
