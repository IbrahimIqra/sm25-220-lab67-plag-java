public static void swapChild(BTNode root, int lvl, int M){
        if(root==null) return;
        if(lvl < M){
            BTNode temporary = root.left;
            root.left = root.right;


            root.right = temporary;
            swapChild(root.left, lvl+1, M);
            swapChild(root.right, lvl+1, M);
        }    
    }

//Task 5
public static Integer subtractSummation( BTNode root ){
        Integer first = summation(root.left);
        Integer second = summation(root.right);
        return first-second;
    }
    public static Integer summation(BTNode root){
        if(root==null) return 0;
        Integer sum = (Integer)root.elem;
        sum += summation(root.left) + summation(root.right);


        return sum;
    }

//Task 6
public static Integer levelSum( BTNode root ){
        return calculateAnswer(root, 0);
    }
    public static Integer calculateAnswer( BTNode root, int lvl ){
        if(root==null) return 0;
        Integer sum = 0;
        if(lvl % 2 == 0)
            sum = -(Integer)root.elem;
        else


            sum = (Integer)root.elem;

        sum += calculateAnswer(root.left, lvl+1) + calculateAnswer(root.right, lvl+1);
        return sum;
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null) return 0;
        Integer sum = 0;
        if(root.elem <= high && root.elem >= low){


            sum = root.elem;
        }
        sum += rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
        return sum;
    }

//Task 5
public static Integer mirrorSum( BSTNode root ){
        if(root==null) return 0;


        return sumHelper(root.left, root.right);
    }
    public static Integer sumHelper(BSTNode root1, BSTNode root2){
        if(root1==null || root2==null) return 0;
        
        Integer sum = root1.elem + root2.elem;


        sum += sumHelper(root1.left, root2.right) + sumHelper(root1.right, root2.left);
        return sum;
    }

//Task 6
public static Boolean isBST( BSTNode root ){


        return helperBST(root, null, null);            
    }
    public static Boolean helperBST(BSTNode root, Integer min, Integer max){
        if(root==null) return true;
        if(max!=null && root.elem >= max) 
            return false;
        if(min!=null && root.elem <= min) 
            return false;

        Boolean checkSmaller = helperBST(root.left, min, root.elem);
        Boolean checkGreater = helperBST(root.right, root.elem, max);
        return checkSmaller && checkGreater;
    }
