public static void swapChild(BTNode root, int lvl, int M) {
    if (root == null) {return;}
    if (lvl >= M) {
        return;
    }

    BTNode tempNode =  root.left;
 root.left  = root.right;
     root.right =  tempNode;
     int incr = lvl+1;
     swapChild(root.left, incr, M);
 swapChild(root.right,  incr,  M);
    }

}

private static Integer subtreeSum(BTNode root) {
        if (root == null) {return 0;  }


    return (Integer) root.elem + subtreeSum(root.left) + subtreeSum(root.right);
    }

  public static Integer subtractSummation(BTNode root) { 
        if (root == null) { return 0; }
        int leftSum = subtreeSum(root.left);
       int rightSum = subtreeSum(root.right);
        return leftSum - rightSum;
    } }

public static Integer levelSum(BTNode root) {


        return utility(root, 0); }

 private static int utility(BTNode root, int level) {
        if (root == null) {return 0;}
     int value = (Integer) root.elem;
     int x;
      if (level % 2 == 0) { x = -value;} 
      else { x = value;}


  return x + utility(root.left, level + 1)  + utility(root.right, level + 1);
    }
}

public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) {return 0;}

        if (root.elem < low) {
            return rangeSum(root.right, low, high); } 
       else if (root.elem > high) {
            return rangeSum(root.left, low, high); } 
        else {


        return root.elem  + rangeSum(root.left, low, high) + rangeSum(root.right, low, high); }
    }
}

public static Integer mirrorSum(BSTNode root) {
        if (root == null) {return 0;}
        return mirrorSumHelper(root.left, root.right);}

        public static int mirrorSumHelper(BSTNode left, BSTNode right) {
        if (left == null || right == null) {return 0;}
     int sum = left.elem + right.elem;
      sum += mirrorSumHelper(left.left, right.right);
        sum += mirrorSumHelper(left.right, right.left);

     return sum;
    }
}

public static Boolean isBST(BSTNode root) {
    return isBSTHelper(root, -9999999, 9999999);}



private static Boolean isBSTHelper(BSTNode node, int min, int max) {
    
    if (node == null) {   return true; }
 if (node.elem <= min || node.elem >= max) {
        return false;}
 return isBSTHelper(node.left, min, node.elem) && isBSTHelper(node.right, node.elem, max);
}
