    public static Integer subtractSummation(BTNode root) {
        if (root == null) return 0;
        int leftSum = calcsum(root.left); //sum of left subtree
        int rightSum = calcsum(root.right); //sum of right subtree
        return leftSum - rightSum;
    }

    //Helper function to calculate sum of subtree
    private static int calcsum(BTNode node)
    {
        if (node == null) return 0;
        return (Integer) node.elem+calcsum(node.left)+calcsum(node.right);
    }

    public static Integer levelSum(BTNode root) {
        return helper(root, 1); // root is at level 1 (odd)
    }
    private static int helper(BTNode root, int level) 
    {
     if (root == null) return 0;
     int value = (Integer)root.elem;
      if (level % 2 == 1) 
        {
         return -value + helper(root.left, level + 1) + helper(root.right, level + 1);//odd
        }
         else 
        {
        return value + helper(root.left, level + 1) + helper(root.right, level + 1); //even
        }
    }

        // Helper function to sum mirror pairs
        private static Integer mirrorSumHelper(BSTNode left, BSTNode right) {
            if (left == null || right == null) {
                return 0;
            }
            // Add the values of mirror nodes
            int sum = left.elem + right.elem;
            sum += mirrorSumHelper(left.left, right.right);
            sum += mirrorSumHelper(left.right, right.left);
            return sum;
