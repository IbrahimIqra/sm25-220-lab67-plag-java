public static void swapChild(BTNode root, int lvl, int M) {
        if (root == null) {
            return;
        }

        if (lvl < M) {
                BTNode current = root.left;
                root.left = root.right;
                root.right = current;
            }

                swapChild(root.left, lvl + 1, M);
                swapChild(root.right, lvl + 1, M);
            }

public static Integer subtractSummation(BTNode root) {
        BTNode left = root.left;
        BTNode right = root.right;
        return sum_method(left) - sum_method(right);
        
    }

    public static int sum_method(BTNode root) {
        if (root == null) {
            return 0;
        }

        int sum = 0;

        sum += (int) root.elem;

        sum += sum_method(root.right);
        sum += sum_method(root.left);
        
        return sum;
        
    }


//Task 06



public static Integer levelSum( BTNode root ){
       return sum_method(root, 0);
       
    }

    private static int sum_method(BTNode root, int level) {
        if (root == null) {
            return 0;
        }

        int sum = 0;

        if (level % 2 == 0) {
            sum -= (int) root.elem;
        }

        else {
            sum += (int) root.elem;
        }

        sum += sum_method(root.left, level + 1);
        sum += sum_method(root.right, level + 1);



        return sum;
        
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if (root == null) {
            return 0;
        }

        int sum = 0;

        if (root.elem >= low && root.elem <= high) {
            sum += root.elem;
        }

        if (root.elem > low) {
            sum += rangeSum(root.left, low, high);
        }

        if (root.elem < high) {
            sum += rangeSum(root.right, low, high);
        }

        return sum;
        
    }


//Task 05

public static Integer mirrorSum(BSTNode root) {
        if (root == null) {
            return 0;
        }

        return sum_method(root.left, root.right);
        
    }

    private static int sum_method(BSTNode left, BSTNode right) {
        if (left == null || right == null) {
            return 0;
        }

        int sum = left.elem + right.elem;

        sum += sum_method(left.left, right.right);
        sum += sum_method(left.right, right.left);

        return sum;
        
    }

public static Boolean isBST( BSTNode root ){
        return BST_checker(root, null, null);
        
        }

        public static boolean BST_checker(BSTNode root, Integer min, Integer max) {
            if (root == null) {
                return true;
            }

            if ((min != null && root.elem <= min) || (max != null && root.elem >= max)) {
                return false;
            }

            return BST_checker(root.left, min, root.elem) && BST_checker(root.right, root.elem, max);
        }
