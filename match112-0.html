public class Task4 {
    public static void swapChild( BTNode root, int lvl, int M ){


      if (root==null || lvl>M){
        return;
      }
      if(lvl<M){
        BTNode t= root.left;
        root.left= root.right;
        root.right= t;
      }
      swapChild(root.left, lvl+1, M);
      swapChild(root.right, lvl+1, M);   
    }
}

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){


      if(root==null){
        return 0;
      }

      return rangeSum(root.right, low, high);

      return rangeSum(root.left, low, high);
    }
    return root.elem + rangeSum(root.left, low, high)+ rangeSum(root.right, low, high);
    }

public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
      if(root==null){
        return 0;
      }
        return mirrorHelper(root.left, root.right);
    }
   
        private static int mirrorHelper(BSTNode left, BSTNode right){
        if(left==null || right==null){
          return 0;
        }
        return( left.elem + right.elem) + mirrorHelper(left.left, right.right) + mirrorHelper(left.right, right.left);  
    }
}

public class Task6 {


    public static Boolean isBST( BSTNode root ){
      if(root==null){
        return true;
      }
      if(root.left!=null && root.left.elem>=root.elem){
        return false;
      }
      if(root.right!=null && root.right.elem<=root.elem){
        return false;
      }
      return isBST(root.left) && isBST(root.right);
    }
}
