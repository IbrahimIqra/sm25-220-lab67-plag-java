    public static Integer helper_rangeSum(BSTNode root, Integer low, Integer high, Integer Sum){
        if(root==null){
            return Sum;
        } if(low<=root.elem && root.elem<=high){
            Sum+=root.elem;
        }
        Sum = helper_rangeSum(root.left, low, high, Sum);


        Sum = helper_rangeSum(root.right, low, high, Sum);
        return Sum;
    }

    public static Integer mirrorSum( BSTNode root ){
        //TO DO
        if(root==null) {
            return 0;
        }
        return sum_func(root.left, root.right);
    }
    public static Integer sum_func(BSTNode left, BSTNode right) {
        if (left == null || right == null){
            return 0;
        }
        int sum_data = left.elem + right.elem;
        sum_data += sum_func(left.left, right.right);
        sum_data += sum_func(left.right, right.left);
        return sum_data;
    }
