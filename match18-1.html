public static void swapChild(BTNode root, int lvl, int M) {
     if (root == null){
         return;
     }
        if (lvl >= M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }

  public static Integer subtractSummation(BTNode root){
         if (root == null){
         return 0;
     }
        int lSum = sumTree(root.left);
        int rSum = sumTree(root.right);
        return lSum - rSum;
    }
    private static int sumTree(BTNode node){
         if (node == null){
         return 0;
     }
         int val =0;
         val = (Integer) node.elem;
        return value + sumTree(node.left) + sumTree(node.right);
    }
}

public static Integer levelSum(BTNode root) {
    return helper(root, 1);
}
private static int helper(BTNode node, int lvl) {
    if (node == null){
        return 0;
    } 
    int val = (Integer) node.elem;
    if (lvl % 2 == 1){
        return val + helper(node.left, lvl + 1) + helper(node.right, lvl + 1);
    }
    else{
        return -val + helper(node.left, lvl + 1) + helper(node.right, lvl + 1);
    }
}

public static Integer rangeSum(BSTNode root, Integer l, Integer h) {
        if (root == null){
            return 0;
        }
        int sum = 0;
        if (root.elem >= l && root.elem <= h){
            sum += root.elem;
        }
        if (root.elem > l){
            sum += rangeSum(root.left, l, h);
        }
        if (root.elem < h){
            sum += rangeSum(root.right, l, h);
        }
        return sum;  
    }
}

public static Integer mirrorSum(BSTNode root) {
    if (root == null){ 
        return 0;
    }
    return mirrorHelper(root.left, root.right);
}
private static int mirrorHelper(BSTNode left, BSTNode right) {
    if (left == null || right == null){ 
        return 0;
    }
    int sum = left.elem + right.elem;
    sum += mirrorHelper(left.left, right.right); 
    sum += mirrorHelper(left.right, right.left);  
    return sum;
}

public class Task6 {
    public static Boolean isBST(BSTNode root){
        return checkBST(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }
    private static boolean checkBST(BSTNode node, int min, int max) {
        if (node == null){
            return true;
        }
        if ( node.elem >= max || node.elem <= min){ 
            return false;
        }
        return checkBST(node.left, min, node.elem) && 
               checkBST(node.right, node.elem, max);
    }
}
