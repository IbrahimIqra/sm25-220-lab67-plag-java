    public static void swapChild( BTNode root, int lvl, int M ){


        if(root == null){
            return;
        }
        if(lvl<M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
            swapChild(root.left, lvl + 1, M);
            swapChild(root.right, lvl + 1, M);
        }
        else{
            return;
        }
    }

    public static Integer subtractSummation( BTNode root ){

        }

    public static int sum(BTNode node){
        if(node == null){
            return 0;
        }
        int n = (Integer)node.elem;
        int total = n + sum(node.left)+sum(node.right);

    public static int levelDifference(BTNode node, int level){


        if(node == null){
            return 0;
        }
        int val = (Integer)node.elem;
        int n = 0;


        if(level%2 == 1){
            n = val;
        }
        else{
            n = -val;
        }
        int left = levelDifference(node.left, level+1);
        int right = levelDifference(node.right, level+1);
        int total = n + left + right;
        return total;
    }

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){


        if(root == null||low == null||low == null||high == null||low>high){
            return 0;
        }


        int total = Sum(root,low,high);
        return total;
    }
    public static int Sum(BSTNode node, int low, int high){
        if(node == null){
            return 0;
        }
        int val = node.elem;
        int sum = 0;
        if(val>=low && val <= high){
            sum = sum + val;
        }
        if(val>low){
            sum = sum + Sum(node.left,low,high);
        }
        if(val<high){
            sum = sum + Sum(node.right,low,high);
        }
       
        return sum;
    }

    public static Integer mirrorSum( BSTNode root ){

            return null;
        }
        int total = Sum(root.left, root.right);
        return total;
    }
    public static int Sum(BSTNode left, BSTNode right){
        if(left == null||right == null){
            return 0;
        }
        int current = (int)left.elem + (int)right.elem;


        int out = Sum(left.left, right.right);
        int in = Sum(left.right, right.left);
        int sum = current + out + in;
        return sum;
    }

    public static Boolean isBST( BSTNode root ){


        if(root == null){
            return true;
        }
        return BST(root,null,null);
    }


    public static boolean BST(BSTNode node, Integer min, Integer max) {


        if(node == null){
            return true;
        }
        int val = node.elem;
        if(min != null && val <= min){
            return false;
        }
        if(max != null && val >= max){
            return false;
        }
        boolean left = BST(node.left,min,val);
        boolean right = BST(node.right,val,max);
       
        if(left == true && right ==true){
            return true;
        }
        else{
            return false;
        }
    }
