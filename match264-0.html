public class Task4 {

    


    public static void swapChild( BTNode root, int lvl, int M ){
        if (lvl == M && root != null){
           return;
        }
        else if (root == null){
            return;
        }
        else{
            swapChild(root.left, lvl + 1, M);
            swapChild(root.right, lvl + 1, M);
            BTNode t = root.left;
            root.left = root.right;
            root.right = t;
        }
    }
    
}

    public static Integer sum(BTNode rt, int a){
        if (rt == null){
            return 0;
        }
        else if (a % 2 != 0){
            return (int)rt.elem + sum(rt.left, a + 1) + sum (rt.right, a + 1);
        }
        else{
            return -(int)rt.elem + sum(rt.left, a + 1) + sum (rt.right, a + 1);
        }
    }

public class Task4 {

   
    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
       return rSHelper(root, low, high);
    }

    public static Integer rSHelper(BSTNode n, Integer l, Integer h){
        if (n == null){
            return 0;
        }

        int sum = 0;
         
        if (n.elem >= l && n.elem <= h){
            sum = sum + n.elem;
        }
        if (n.elem > l){
            sum = sum + rSHelper(n.left, l, h);
        }
         if (n.elem < h){
            sum = sum + rSHelper(n.right, l, h);
        }
        return sum;
    }

    
      


}

public class Task5 {

    public static Integer mirrorSum(BSTNode root) {
        return mirrorHelper(root.left, root.right);
    }

    private static Integer mirrorHelper(BSTNode leftN, BSTNode rightN) {
        if (leftN == null && rightN == null) {
            return 0;
        }
        
        if (leftN == null || rightN == null) {
            return 0;
        }

        int sum = leftN.elem + rightN.elem;

        sum += mirrorHelper(leftN.left, rightN.right);
        sum += mirrorHelper(leftN.right, rightN.left);

        return sum; 
    }
  

}

public class Task6 {

   
    public static Boolean isBST(BSTNode root) {


        return BSTHelper(root, null, null);
    }

    
    private static Boolean BSTHelper(BSTNode n, Integer min, Integer max) {
        
        if (n == null) {
            return true;
        }

       
        if ((min != null && n.elem <= min) || (max != null && n.elem >= max)) {
            return false;
        }


        return BSTHelper(n.left, min, n.elem) && BSTHelper(n.right, n.elem, max);
    }
    

}
