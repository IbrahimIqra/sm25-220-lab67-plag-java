 public static void swapChild( BTNode root, int lvl, int M ){
        
        if (root == null) {
            return;
        }
        
        
        if (lvl < M) {
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        
        
        swapChild(root.left, lvl + 1, M);
        swapChild(root.right, lvl + 1, M);
    }

    public static Integer subtractSummation( BTNode root ){
         if (root == null) {
            return 0;
        }
        
        int leftSum = 0;
        int rightSum = 0;
        
        
        if (root.left != null) {
            leftSum = (Integer) root.left.elem + help(root.left.left) + help(root.left.right);
        }
        
        // Calculate right subtree sum
        if (root.right != null) {
            rightSum = (Integer) root.right.elem + help(root.right.left) + help(root.right.right);
        }
        
        return leftSum - rightSum;
    }
    
    
    public static int help(BTNode node) {
        if (node == null) return 0;
        return (Integer) node.elem + help(node.left) + help(node.right);
    }

    public static Integer levelSum( BTNode root ){
        
       return help(root, 0);
    }

    private static int help(BTNode root, int level) {
        if (root == null) {
            return 0;
        }

        
        int nodeValue = (Integer) root.elem;
        
        
        int curSum;
        if (level % 2 == 0) {
            curSum = -nodeValue;  
        } else {
            curSum = nodeValue;   
        }

        
        int leftSum = help(root.left, level + 1);
        int rightSum = help(root.right, level + 1);



        return curSum + leftSum + rightSum;
    }

     public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if (root == null) {
            return 0;
        }
        
        int sum = 0;
        
        
        if (root.elem >= low && root.elem <= high) {
            sum += root.elem;
        }
        
        
        if (root.elem > low) {
            sum += rangeSum(root.left, low, high);
        }
        
        
        if (root.elem < high) {
            sum += rangeSum(root.right, low, high);
        }
        
        return sum;
    }

public static Integer mirrorSum( BSTNode root ){
        if (root == null) {
            return 0;
        }
        return help(root.left, root.right);
    }
    
    private static int help(BSTNode leftNode, BSTNode rightNode) {
        if (leftNode == null || rightNode == null) {
            return 0;
        }
        
        
        int sum = leftNode.elem + rightNode.elem;
        
        
        sum += help(leftNode.left, rightNode.right);
        sum += help(leftNode.right, rightNode.left);
        
        return sum;
    }

public static Boolean isBST( BSTNode root ){
        if (root == null) 
        return true;
        
        
        if (root.left != null && root.left.elem >= root.elem) 
        return false;
        if (root.right != null && root.right.elem <= root.elem) 
        return false;
        
        
        return isBST(root.left) && isBST(root.right);
    }
