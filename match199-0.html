public static void swapChild( BTNode root, int lvl, int M ){
        if(root==null){
            return;
        }
        if(lvl<M){
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);

    }

public static Integer subtractSummation( BTNode root ){
        return sum(root.left)-sum(root.right);
    }
    public static int sum(BTNode node){
        if(node==null){
            return 0;
        }
        return (int)node.elem+ sum(node.left)+sum(node.right);
    }

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root.elem>= low && root.elem<=high){
        return root.elem+sumOfRange(root.left, low, high)+sumOfRange(root.right, low, high);
        }
        else{
            return sumOfRange(root.left, low, high)+sumOfRange(root.right, low, high);
        }
    }
    private static Integer sumOfRange(BSTNode root, Integer low,Integer high){
        if(root==null){
            return 0;
        }
        if(root.elem>= low && root.elem<=high){
            return root.elem+sumOfRange(root.left, low, high)+sumOfRange(root.right, low, high);
        }
        if(root.elem<low){
            return sumOfRange(root.right, low, high);
        }


        return sumOfRange(root.left, low, high);
    }

public static Integer mirrorSum( BSTNode root ){
        if(root==null){
            return 0;
        }
        return sumhelper(root.left, root.right);

    }
    private static Integer sumhelper(BSTNode left, BSTNode right){
        if(left==null || right==null){
            return 0;
        }
        
        return left.elem+right.elem+sumhelper(left.left, right.right)+sumhelper(left.right, right.left);

    }

public static Boolean isBST( BSTNode root ){

        if(root==null){
            return true;
        }
        int left = root.left.elem;
        int right = root.right.elem;
        if(left>root.elem){
            return false;
        }
        if(right<root.elem){
            return false;
        }
        return isBST(root.left)&&isBST(root.right);
    }
