public static Integer mirrorSum(BSTNode root) {
//TO DO
       
    if(root == null){
      return 0;
    }
    return mirrorSum_helper_function(root.left, root.right);
}
private static Integer mirrorSum_helper_function(BSTNode left_node, BSTNode right_node){
     if(left_node == null || right_node == null){
       return 0;
     }

       int addition = left_node.elem + right_node.elem;
             addition = addition + mirrorSum_helper_function(left_node.left, right_node.right);
            addition = addition + mirrorSum_helper_function(left_node.right, right_node.left);
        return addition;
        }
}

public static Boolean isBST( BSTNode root ){
        //TO DO
        return  ifBST(root);
       
    }

    private static Boolean ifBST(BSTNode root){

        if(root==null) return true;
        int c=0;

        if(root.left !=null ){
            c++;
        }
        if(root.right !=null){
            c++;
        }
         return c<=2;
    }
