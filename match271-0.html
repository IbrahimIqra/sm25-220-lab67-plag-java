public static void swapChild( BTNode root, int lvl, int M ){
        if(root == null){
            return;
        }
        if(lvl < M){
            BTNode temp = root.left;
            root.left = root.right;


            root.right = temp;
            swapChild(root.left, lvl+1, M);
            swapChild(root.right, lvl+1, M);
        }
        
    }

public static Integer subtractSummation( BTNode root ){
       if (root == null) {
        return 0;
    }

    int leftSum = helper(root.left);
    int rightSum = helper(root.right);

    return leftSum - rightSum;
}

    public static int helper(BTNode node) {
    if (node == null) {
        return 0;
    }
    return (int)node.elem + helper(node.left) + helper(node.right);
}

    public static int helper(BTNode node, int level) {
        if (node == null) {
            return 0;
        }

        int val = (int) node.elem;

         if (level % 2 == 1) {
            return -val + helper(node.left, level + 1) + helper(node.right, level + 1);
        } 
        else {
            return val + helper(node.left, level + 1) + helper(node.right, level + 1);
        }
    }

 public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
       if(root == null){
        return 0;
       }

       int sum = 0;

       if((int)root.elem >= low && (int)root.elem <= high){
            sum+= (int)root.elem;
       }

        if ((int)root.elem > low) {
        sum += rangeSum(root.left, low, high);
        }

    
        if ((int)root.elem < high) {


            sum += rangeSum(root.right, low, high);
        }

        return sum;

    }

public static Integer mirrorSum( BSTNode root ){
        if (root == null) {
            return 0;
        }
        return helper(root.left, root.right);
    }

    public static int helper(BSTNode left, BSTNode right) {
    if (left == null || right == null) {
        return 0;
    }
    int sum = left.elem + right.elem;

    sum += helper(left.left, right.right);
    sum += helper(left.right, right.left);

    return sum;
    }

public static Boolean isBST( BSTNode root ){
        return helper(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }
    public static boolean helper(BSTNode node, int min, int max) {
    if (node == null) return true;

    
    if (node.elem <= min || node.elem >= max) {
        return false;
    }

    return helper(node.left, min, node.elem) &&
           helper(node.right, node.elem, max);
}
