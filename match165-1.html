  public static void swapChild(BTNode root, int lvl, int M) {
        if (root == null) {return;}

        if (lvl < M) {

        BTNode tempo = root.left;
        root.left = root.right;
        root.right = tempo;
        }
        
        swapChild(root.left, lvl+1, M);

        swapChild(root.right, lvl+1, M);
        


    }

  public static Integer subtractSummation(BTNode root) {
        if (root == null) {return 0;} 
        int lefts = sum(root.left);   
        int rights = sum(root.right); 
        return lefts - rights;      
    }

    private static int sum(BTNode root) {
        if (root == null) {return 0;}
        return (int) root.elem + sum(root.left) + sum(root.right);
    }

 public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) return 0;

        int val = (int) root.elem;

        if (val >= low && val <= high) {
            return val + rangeSum(root.left, low, high) + rangeSum(root.right, low, high);
        } 
        else if (val < low) {
            return rangeSum(root.right, low, high);
        } 

        else {


            return rangeSum(root.left, low, high);
        }
    }

    public static Integer mirrorSum(BSTNode root) {
        if (root == null){
            return 0;
        }
        return mirrors(root.left, root.right);
    }

    private static int mirrors(BSTNode left, BSTNode right) {

        if (left == null || right == null){
        return 0;
        } 

        int sum = left.elem + right.elem;

        sum = sum + mirrors(left.left, right.right);
        sum = sum + mirrors(left.right, right.left);

        return sum;
}

public static boolean isBST(BSTNode root) {
return checker(root, null, null);
}



private static boolean checker(BSTNode curr, Integer low, Integer high) {

if (curr == null) {
return true;
}

int val = curr.elem;

if (low != null) {
if (val <= low) {
return false;
}
}



if (high!= null) {
if (val >= high) {
return false;
}
}



boolean lefty = checker(curr.left, low, val);
boolean righty = checker(curr.right, val, high);
return lefty && righty;

}
