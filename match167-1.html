    private static void calculateLevelSum(BTNode node, int lvl, int[] sum){
      if (node==null){
        return;
      }
      if (lvl%2==0){
        sum[0]+=(int) node.elem;
      }
      else{
        sum[1]+=(int) node.elem;
      }
      calculateLevelSum(node.left, lvl+1, sum);
      calculateLevelSum(node.right, lvl+1, sum);
                     
    }

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        //TO DO
      if (root==null){
        return 0;
      }
      int sum = 0;
      if (root.elem>=low && root.elem <=high){
        sum+=root.elem;
      }
      if (root.elem>low){
        sum+=rangeSum(root.left, low, high);
      }
      if (root.elem<high){
        sum+=rangeSum(root.right, low, high);
      }
      return sum;
    }

    public static Integer mirrorSum( BSTNode root ){
        //TO DO
      if (root==null){
        return 0;
      }
      return helperFunc(root.left, root.right);
    }
    private static int helperFunc(BSTNode left,BSTNode right){
      if (left==null || right==null){
        return 0;
      }
      int sum = left.elem+right.elem;
      sum+=helperFunc(left.left, right.right);
      sum+=helperFunc(left.right,right.left);


      return sum;
    }

  public static Boolean isBST( BSTNode root ){
    return ifBST(root, null, null);
  }
  private static boolean ifBST(BSTNode node, Integer min, Integer max){
    if (node == null){
      return true;
    }
    if (min!=null && node.elem<=min){
      return false;
    }
    if (max!=null && node.elem>=max){
      return false;
    }
    return ifBST(node.left, min, node.elem)&& ifBST(node.right, node.elem, max);
  }
