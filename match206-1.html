public static void swapChild(BTNode root, int lvl, int M) {

        if (root == null) {
            return;
        }

        if (lvl >= M) {
            BTNode temp;
             temp = root.left;
            root.left = root.right;
            root.right = temp;
        }
        swapChild(root.left,lvl+1,M);
        swapChild(root.right,lvl+1,M);
    }

 public static Integer subtractSummation(BTNode root) {
        if (root == null) {
            return 0;
        }

        int leftSum = calc(root.left);
        int rightSum = calc(root.right);
        return leftSum-rightSum;
    }

    private static int calc(BTNode node) {
        if (node == null) {
            return 0;
        }

        return(Integer)node.elem+ calc(node.left)+ calc(node.right);
    }

public static Integer levelSum(BTNode root) {
        if (root == null) {
            return 0;
        }


        return dif(root,1);
    }
    public static int dif(BTNode node,int level) {
        if (node ==null) {
            return 0;
        }
        int cur= (Integer) node.elem;
        if (level% 2==1) {
            return cur+dif(node.left, level +1) + dif(node.right, level+ 1);
        } else {
            return -cur+ dif(node.left,level+ 1) + dif(node.right,level+1);
        }
    }

    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) {
            return 0;
        }
        int total= 0;
        if (root.elem>=low && root.elem <= high) {
            total =total+ root.elem;
        }
        if (root.elem>low) {
            total =total+rangeSum(root.left, low, high);
        }
        if (root.elem<high) {
            total=total+ rangeSum(root.right, low, high);
        }
        return total;
    }

public class Task5 {

    public static Integer mirrorSum(BSTNode root) {
        if (root == null) {
            return 0;
        }
        return total(root.left, root.right);
    }
    private static Integer total(BSTNode l, BSTNode r) {
        if (l == null || r == null) {
            return 0;
        }
        int sum= l.elem + r.elem;
        sum= sum+total(l.left, r.right);
        sum= sum+total(l.right, r.left);
        return sum;
    }
}

public class Task6 {

    public static Boolean isBST(BSTNode root) {
        return helper(root, null, null);
    }
    private static Boolean helper(BSTNode node, Integer min, Integer max) {
        if (node == null) {
            return true;
        }


        if ((min != null && node.elem<=min)||(max != null && node.elem >= max)) {
            return false;
        }
        return helper(node.left, min,node.elem) &&helper(node.right, node.elem, max);
    }
