public class Task4 {


    public static void swapChild( BTNode root, int lvl, int M ){
        if(root==null){
            return;
        }
        if(lvl<M){
            BTNode n=root.left;
            root.left=root.right;
            root.right=n;
        }
        swapChild(root.left, lvl+1, M);
        swapChild(root.right, lvl+1, M);
        
    }
}

public class Task5 {
    public static Integer subtractSummation( BTNode root ){
        if(root==null){
            return 0;
        }
        return helper( root.left)-helper( root.right );
        
    }
    private static Integer helper(BTNode root){
        if(root==null){
            return 0;
        }
        Integer x=(Integer)root.elem;


        return x+helper( root.left )+helper( root.right );
        
    }
}

public class Task6 {
    public static Integer levelSum( BTNode root ){
        return helper(root,1);
    }
    private static Integer helper(BTNode root, int lvl){
        if(root==null){
            return 0;
        }
        Integer x=(Integer)root.elem;
        if(lvl%2==0){
            return x+helper(root.left,lvl+1)+helper(root.right,lvl+1);
        }
        else{


            return -x+helper(root.left,lvl+1)+helper(root.right,lvl+1);
        }
    }
}

public class Task4 {

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root==null){
            return 0;
        }
        Integer x=(Integer)root.elem;
        if(x>high){
            return rangeSum(root.left,low,high);
        }
        else if(x<low){
            return rangeSum(root.right,low,high);
        }
        else{
            return x+rangeSum(root.left,low,high)+rangeSum(root.right,low,high);
        }
    }
}



public class Task5 {
    public static Integer mirrorSum( BSTNode root ){
        if(root==null){
            return 0;
        }
        return helper(root.left,root.right);
    }
    private static Integer helper(BSTNode lft, BSTNode rht){
        if(lft==null || rht==null){
            return 0;
        }
        Integer l=(Integer)lft.elem;
        Integer r=(Integer)rht.elem;


        Integer s=l+r;
        s+=helper(lft.left,rht.right);
        s+=helper(lft.right,rht.left);
        return s;
    }
}

public class Task6 {
    public static Boolean isBST( BSTNode root ){
        if(root==null){
            return false;
        }
        Integer r=(Integer)root.elem;
        Integer lft=(Integer)root.left.elem;
        Integer rht=(Integer)root.right.elem;
        if(lft<r && rht>r){
            return true;
        }
        return false;
    }
}
