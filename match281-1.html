    public static void swapChild( BTNode root, int lvl, int M ){
       if(root==null){
        return;
       }
       if(lvl < M){
        BTNode temp = root.left;
        root.left = root.right;
        root.right = temp;
       }
       swapChild(root.left, lvl+1, M);
       swapChild(root.right, lvl+1, M);
    }

    public static Integer subtractSummation( BTNode root ){
        if(root==null){
            return 0;
        }
        int leftSum = helper(root.left);
        int rightSum = helper(root.right);
        return leftSum-rightSum;
    }
    private static int helper(BTNode root){
        if(root == null){
            return 0;
        }
        return (int)root.elem+helper(root.left)+helper(root.right);
    }

    public static Integer levelSum( BTNode root ){
        return helper(root, 0);
    }
    private static int helper(BTNode root, int level){
        if(root==null){
            return 0;
        }
        int sum = 0;
        if(level==0){
            sum-=(int)root.elem;
        }
    else{
            if(level%2==0){
                sum-=(int)root.elem;
            }
        else{
            sum+=(int)root.elem;
        }
    }
        sum+=helper(root.left, level+1);
        sum+=helper(root.right, level+1);
        return sum;
    }

    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        if(root == null){
            return 0;
        }
        int sum = 0;
        if(root.elem>= low && root.elem<=high){
            sum+=root.elem;
        }
        if(root.elem>low){
            sum+= rangeSum(root.left, low, high);
        }
        if(root.elem<high){
            sum+= rangeSum(root.right, low, high);
        }
        return sum; // remove this line
    }

    public static Integer mirrorSum( BSTNode root ){


        if(root == null){
            return 0;
        }
        return helper(root.left, root.right);
    }
    private static Integer helper(BSTNode left, BSTNode right){
        if(left==null || right == null){
            return 0;
        }


        int sum = left.elem + right.elem;
        sum+= helper(left.left, right.right);
        sum+= helper(left.right, right.left);
        return sum;
    }

    public static Boolean isBST( BSTNode root ){
        return helper(root, null, null);
    }
    private static Boolean helper(BSTNode root, Integer min, Integer max){
        if(root == null){
            return true;
        }
        if((min!=null && root.elem<=min) || (max!=null && root.elem>=max)){


            return false;
        }
        return helper(root.left, min, root.elem) && helper(root.right, root.elem, max); 
    }
