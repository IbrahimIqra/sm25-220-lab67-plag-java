public static void swapChild( BTNode root, int lvl, int M ){
        //TO DO
        if(root==null){
            return;
        }
       
        if(lvl<M){
            BTNode temp = root.left;
            root.left = root.right;

public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        //TO DO
        if(root==null){
            return 0;
        }


        Integer x = (Integer)root.elem;
        Integer sum = 0;


        if(x>=low && x<=high){
            sum+=x;
        }


        if(x<high){
            sum+=rangeSum(root.right, low, high);
        }


        if(x>low){
            sum+=rangeSum(root.left, low, high);
        }


        return sum; 
    }

 public static Integer mirrorSum( BSTNode root ){
        //TO DO
        if(root==null){
            return 0;
        }
        return sumHelper(root.left, root.right);// remove this line
    }


 private static Integer sumHelper(BSTNode root1, BSTNode root2){
        if(root1==null || root2 == null){
            return 0;
        }


        int sum = (Integer)root1.elem + (Integer)root2.elem;


        sum+=sumHelper(root1.left, root2.right);
        sum+=sumHelper(root1.right, root2.left);
       
        return sum;
    }
