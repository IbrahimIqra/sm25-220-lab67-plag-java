public class Task4 {
    public static void swapChild(BTNode root, int lvl, int M) {
      if (root==null) return;
        if (lvl >= 2 && lvl <=M) {
            BTNode temp=root.left;
           root.left=root.right;
            root.right=temp;
        }
         swapChild(root.left, lvl + 1,M);
         swapChild(root.right, lvl + 1,M);
    }
}

public class Task5 {
    public static Integer subtractSummation(BTNode root) {
      if (root==null) return 0;
         int left=calculateSum(root.left);
         int right=calculateSum(root.right);
        return left-right;
    }
    public static int calculateSum(BTNode node) {
     if (node==null) return 0;
         int val=(Integer)node.elem;
         int left=calculateSum(node.left);
         int right=calculateSum(node.right);
       return val + left + right;
    }
}

    public static Integer levelSum(BTNode root) {


      return computeLevelSum(root, 0);
    }
    public static int computeLevelSum(BTNode node,int level) {
       if (node==null) return 0;
        int value=(Integer)node.elem;
       if (level % 2==0) {
        value=-value;
        }
         int left =computeLevelSum(node.left,level+1);
         int right =computeLevelSum(node.right,level+1);
       return value+left+right;
    }
}

public class Task4 {
    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
     return addRange(root, low, high);
    }
    public static int addRange(BSTNode node, int low, int high) {
      if (node==null) return 0;
         int val= (Integer)node.elem;
         int total=0;
      if (val >= low && val <= high) {
            total=total+val;
        }
       if (val > low) {
        total=total+addRange(node.left, low, high);
        }
      if (val < high) {
         total=total+addRange(node.right, low, high);
        }
     return total;
    }
}

public class Task5 {
    public static Integer mirrorSum(BSTNode root) {
      if (root==null) return 0;
         return sumMirrors(root.left, root.right);
    }
    public static int sumMirrors(BSTNode leftNode, BSTNode rightNode) {
         if (leftNode==null || rightNode==null) return 0;
          int left=(Integer)leftNode.elem;
          int right=(Integer)rightNode.elem;


          int currentSum=left+right;
           int outer=sumMirrors(leftNode.left, rightNode.right);
          int inner=sumMirrors(leftNode.right, rightNode.left);
        return currentSum + outer + inner;
    }
}

public class Task6 {
    public static Boolean isBST(BSTNode root) {
        return checkBST(root, Integer.MIN_VALUE, Integer.MAX_VALUE);
    }
    public static boolean checkBST(BSTNode node, int min, int max) {
    if (node==null) return true;
      int val=(Integer)node.elem;


     if (val<=min || val>=max) return false;
        boolean left=checkBST(node.left, min, val);
        boolean right=checkBST(node.right, val, max);
       return left && right;
    }
