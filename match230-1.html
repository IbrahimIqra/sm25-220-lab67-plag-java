public class Task4 { 


    public static void swapChild(BTNode root, int lvl,int M) {
        if (root == null) {
        return;
        }
        if (lvl >= M){
             return;
        }

        BTNode Tmpo = root.left;
        root.left = root.right;
        root.right = Tmpo;

    
        swapChild(root.left , lvl+1, M);
        swapChild(root.right, lvl+1, M);
    }
}

 public class Task5 {


    public static Integer subtractSummation(BTNode root) {
        
        int leftsubtreeSum  = sumTreeHelper(root.left);
        int rightsubtreeSum = sumTreeHelper(root.right);

        return leftsubtreeSum - rightsubtreeSum;  
    }

    private static int sumTreeHelper(BTNode root) {
        if (root == null) {
            return 0;
        }
        return (Integer) root.elem + (Integer) sumTreeHelper(root.left) + (Integer)sumTreeHelper(root.right);
    }
   
}

public class Task6 {

    public static Integer levelSum(BTNode root) {
        return SumHelper(root, 0);
    }

    private static int SumHelper(BTNode root, int Lvl) {
        if (root == null) {
            return 0;
        }
        int elem = (Integer) root.elem;

        if (Lvl % 2 == 0) {
          
            return -elem + SumHelper(root.left, Lvl + 1) + SumHelper(root.right, Lvl + 1);
        } 
        
        else {
            return elem + SumHelper(root.left, Lvl + 1) + SumHelper (root.right, Lvl + 1);
        }
    }
}

public class Task4 {



    public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
        
 
        return SumHelper(root, low, high);
    }

    private static int SumHelper(BSTNode root,  int l, int h) {
        if (root == null) {
            return 0;
        }
        int elem =  root.elem;

        if (elem >= l && elem <= h ) {
          
            return elem + SumHelper(root.left,  l , h ) + SumHelper(root.right, l , h);
        } 
        
        else {
            return 0 + SumHelper(root.left,  l ,h) + SumHelper (root.right, l ,h);
        }
    }
}

public class Task6 {

  
   public static Boolean isBST(BSTNode root) {
    return isBSThelper(root, null, null);
}

private static Boolean isBSThelper(BSTNode node, Integer min, Integer max) {
    if (node == null) {
        return true;
    }

    
    if (min != null && node.elem <= min) return false;

    


    if (max != null && node.elem >= max) return false;


    return isBSThelper(node.left, min, node.elem) &&
           isBSThelper(node.right, node.elem, max);
}
