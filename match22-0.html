public static void swapChild( BTNode root, int lvl, int M ){
      if(root==null){
        return;
      }
      if(lvl>=M){
        BTNode temp = root.left;
        root.left = root.right;
        root.right = temp;
      }
      swapChild(root.left, lvl+1, M);
      swapChild(root.right, lvl+1, M);
    }

 public static Integer subtractSummation( BTNode root ){
      if(root==null){
        return 0;
      }
      int leftst_sum = summation(root.left);
      int rightst_sum = summation(root.right);
      return leftst_sum - rightst_sum;
    }
    public static int summation(BTNode node){
      if(node==null){
        return 0;
      }
      return summation(node.left)+summation(node.right)+(Integer)node.elem;
    }


//Lab06 Task-06

public static Integer levelSum( BTNode root ){
      return helper(root,1);
    }


    public static int helper(BTNode node, int lvl){
      if(node==null){
        return 0;
      }
      int val =(Integer)node.elem;
      if(lvl%2==1){
        return -val+helper(node.left,lvl+1)+helper(node.right,lvl+1);


      }
      else{
        return val+helper(node.left,lvl+1)+helper(node.right,lvl+1);
      }
    }



//Lab07 Task04



 public static Integer rangeSum( BSTNode root, Integer low, Integer high ){
      if(root==null){
        return 0;
      }
      int sum = 0;
      if(root.elem>=low && root.elem<=high){
        sum += root.elem;
      }
      if(root.elem>low){
        sum += rangeSum(root.left, low, high);
      }
       if(root.elem<high){
        sum += rangeSum(root.right, low, high);
      }
       return sum;
    }


//Lab07 Task05



public static Integer mirrorSum( BSTNode root ){
      if(root==null){
        return 0;
      }
      return helper(root.left, root.right);
    }
    public static int helper(BSTNode left, BSTNode right){
      if(left==null || right==null){
        return 0;
      }
      int sum = left.elem + right.elem;
      sum+= helper(left.left,right.right);
      sum+= helper(left.right,right.left);
      return sum;
    }

public static Boolean isBST( BSTNode root ){
      return BSTCon(root,Integer.MIN_VALUE, Integer.MAX_VALUE);
    }


    public static Boolean BSTCon(BSTNode node, int min, int max){
      if(node==null){
        return true;
      }
      if(node.elem<=min || node.elem>=max){
        return false;
      }
      return BSTCon(node.left, min, node.elem) && BSTCon(node.right, node.elem, max);
    }
