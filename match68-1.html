    public static void swapChild(BTNode root, int lvl, int M) {
        if (root == null){

       return;
}
        if (lvl>=M) {
            BTNode temp = root.left;
            root.left = root.right;
            root.right = temp;
        }

        swapChild(root.right, lvl+1,M);

        if (root == null) {

        }


        return levelSumHelper(root, 1);

    }
    private static int levelSumHelper(BTNode node, int level) {

        if (node == null) {
            return 0;
        }

        int value = (int) node.elem;

        if (level % 2 != 0) {  


            return value    + levelSumHelper(node.left, level + 1) + levelSumHelper(node.right, level + 1);
        } else {
           
                  return - value
                  + levelSumHelper(node.left, level + 1)
                 + levelSumHelper(node.right, level + 1);
        }
    }

public class Task4 {

   
    public static Integer rangeSum(BSTNode root, Integer low, Integer high) {
        if (root == null) {
            return 0;  
        }

        int value = (int) root.elem;

       
        if (value >= low && value <= high) {
            return value  + rangeSum(root.left, low, high) 
                 + rangeSum(root.right, low, high);
        }

        if (value < low) {
            return rangeSum(root.right, low, high);
        }

        else{
        return rangeSum(root.left, low, high);
    }
  }
}

public class Task5 {

    public static Integer mirrorSum(BSTNode root) {
        if (root == null) {
            return 0;
        }
        return mirrorSumHelper(root.left, root.right);
    }

    private static int mirrorSumHelper(BSTNode leftNode, BSTNode rightNode) {
  
        if (leftNode == null || rightNode == null) {
            return 0;
        }
        int sum = (int) leftNode.elem + (int) rightNode.elem;

        sum +=mirrorSumHelper(leftNode.left, rightNode.right);

        sum+= mirrorSumHelper(leftNode.right, rightNode.left);

        return sum;
    }
}
