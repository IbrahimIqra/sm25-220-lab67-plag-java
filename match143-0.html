  public static void swapChild(BTNode root, int level, int M){
          if(root == null) 
            return;                         //stopping if no node
        if(level >= M){
         BTNode hold = root.left;  
         root.left = root.right;    
         root.right = hold;         
        }
    swapChild(root.left, level + 1, M);         //one step deeper on the left side

   }

     private static int findMirroredSum(BSTNode left, BSTNode right){
         if (left == null || right == null){
          return 0;
        }
    int total = (Integer) left.elem + (Integer) right.elem;



    int outside = findMirrorSum(left.left, right.right);   //left er left and right er right
    int inside = findMirrorSum(left.right, right.left);    //left er right and right er left

    total += outside + inside;
     return total;
    }
}


3)public class Task6 {
    public static Boolean isBST(BSTNode root) {
        return isBSTHelper(root, null, null);
    }
    private static Boolean isBSTHelper(BSTNode node, Integer min, Integer max) {
        if (node == null) {
            return true;
        }
        int value = (Integer) node.elem;
        if (min != null && value <= min) {
            return false;
        }


        if (max != null && value >= max) {
            return false;
        }
        boolean isLeftValid = isBSTHelper(node.left, min, value);
        boolean isRightValid = isBSTHelper(node.right, value, max);
       return isLeftValid && isRightValid;
    }
}
